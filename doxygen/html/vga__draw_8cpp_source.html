00416     Bit32u* temps = (Bit32u*) TempLine;
00417     unsigned int poff = 0;
00418     Bitu skip; /* how much to skip after drawing 4 pixels */
00419 
00420     skip = 4u << vga.config.addr_shift;
00421 
00422     /* *sigh* it looks like DOSBox's VGA scanline code will pass nonzero bits 0-1 in vidstart */
00423     poff += vidstart & 3u;
00424     vidstart &= ~3ul;
00425 
00426     /* hack for Surprise! productions "copper" demo.
00427      * when the demo talks about making the picture waver, what it's doing is diddling
00428      * with the Start Horizontal Retrace register of the CRTC once per scanline.
00429      * ...yeah, really. It's a wonder in retrospect the programmer didn't burn out his
00430      * VGA monitor, and I bet this makes the demo effect unwatchable on LCD flat panels or
00431      * scan converters that rely on the pulses to detect VGA mode changes! */
00432     if (vga_enable_hretrace_effects) {
00433         /* NTS: This is NOT BACKWARDS. It makes sense if you think about it: the monitor
00434          *      begins swinging the electron beam back on horizontal retract, so if the
00435          *      retrace starts sooner, then the blanking on the left side appears to last
00436          *      longer because there are more clocks until active display.
00437          *
00438          *      Also don't forget horizontal total/blank/retrace etc. registers are in
00439          *      character clocks not pixels. In 320x200x256 mode, one character clock is
00440          *      4 pixels.
00441          *
00442          *      Finally, we average it with "weight" because CRTs have "inertia" */
00443         float a = 1.0 / (hretrace_fx_avg_weight + 1);
00444 
00445         hretrace_fx_avg *= 1.0 - a;
00446         hretrace_fx_avg += a * skip * ((int)vga_display_start_hretrace - (int)vga.crtc.start_horizontal_retrace);
00447         int x = (int)floor(hretrace_fx_avg + 0.5);
00448 
00449         vidstart += (Bitu)((int)skip * (x >> 2));
00450         poff += x & 3;
00451     }
00452 
00453     for(Bitu i = 0; i < ((vga.draw.line_length>>(2/*32bpp*/+2/*4 pixels*/))+((poff+3)>>2)); i++) {

00455

00458         *temps++ = vga.dac.xlat32[*ret++];
00459         *temps++ = vga.dac.xlat32[*ret++];
00460         *temps++ = vga.dac.xlat32[*ret++];
00461         /* and skip */

00463     }
00464

01917 void VGA_CheckScanLength(void) {
01918     switch (vga.mode) {
01919     case M_EGA:
01920     case M_LIN4:
01921         if ((machine==MCH_EGA)&&(vga.crtc.mode_control&0x8))
01922             vga.draw.address_add=vga.config.scan_len*16;
01923         else
01924             vga.draw.address_add=vga.config.scan_len*8;
01925         break;
01926     case M_VGA:
01927     case M_LIN8:
01928     case M_LIN15:
01929     case M_LIN16:
01930     case M_LIN24:
01931     case M_LIN32:
01932         if (vga.mode == M_VGA) {
01933             if (svgaCard == SVGA_TsengET3K || svgaCard == SVGA_TsengET4K) {
01934                 /* Observed ET4000AX behavior:
01935                  *    byte mode OR dword mode scans 256-color 4-pixel groups byte by byte from
01936                  *    planar RAM. word mode scans every other byte (skips by two).
01937                  *    We can just scan the buffer normally as linear because of the address
01938                  *    translation carried out by the ET4000 in chained mode:
01939                  *
01940                  *    plane = (addr & 3)   addr = (addr >> 2)
01941                  *
01942                  *    TODO: Validate that this is correct. */
01943                 vga.draw.address_add=vga.config.scan_len*((vga.config.addr_shift == 1)?16:8);
01944             }
01945             else {
01946                 /* Most (almost ALL) VGA clones render chained modes as 4 8-bit planes one DWORD apart.
01947                  * They all act as if writes to chained VGA memory are translated as:
01948                  * addr = ((addr & ~3) << 2) + (addr & 3) */
01949                 vga.draw.address_add=(unsigned int)vga.config.scan_len*((unsigned int)(2*4)<<(unsigned int)vga.config.addr_shift);
01950             }
01951         }
01952         else {
01953             /* the rest (SVGA modes) can be rendered with sanity */
01954             vga.draw.address_add=vga.config.scan_len*(unsigned int)(2u<<vga.config.addr_shift);
01955         }
01956         break;
01957     case M_PC98:
01958         vga.draw.address_add=vga.draw.blocks;
01959         break;
01960     case M_TEXT:
01961     case M_CGA2:
01962     case M_CGA4:
01963     case M_CGA16:
01964     case M_AMSTRAD: // Next line.
01965         if (IS_EGAVGA_ARCH || IS_PC98_ARCH)
01966             vga.draw.address_add=vga.config.scan_len*(unsigned int)(2u<<vga.config.addr_shift);
01967         else
01968             vga.draw.address_add=vga.draw.blocks;
01969         break;
01970     case M_TANDY2:
01971         vga.draw.address_add=vga.draw.blocks/4u;
01972         break;
01973     case M_TANDY4:
01974         vga.draw.address_add=vga.draw.blocks;
01975         break;
01976     case M_TANDY16:
01977         vga.draw.address_add=vga.draw.blocks;
01978         break;
01979     case M_TANDY_TEXT:
01980         vga.draw.address_add=vga.draw.blocks*2;
01981         break;
01982     case M_HERC_TEXT:
01983         vga.draw.address_add=vga.draw.blocks*2;
01984         break;
01985     case M_HERC_GFX:
01986         vga.draw.address_add=vga.draw.blocks;
01987         break;
01988     default:
01989         vga.draw.address_add=vga.draw.blocks*8;
01990         break;
01991     }
