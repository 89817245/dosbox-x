00030 /* Character Generator ports.
00031  * This is in fact officially documented by NEC in
00032  * a 1986 book published about NEC BIOS and BASIC ROM. */
00033 Bitu pc98_a1_read(Bitu port,Bitu iolen) {
00034     switch (port) {
00035         case 0xA9: // an 8-bit I/O port to access font RAM by...
00036             // NOTES: On a PC-9821 Lt2 laptop, the character ROM doesn't seem to latch valid data beyond
00037             //        0xxx5D. Often this reads back as zero, but depending on whatever random data is floating
00038             //        on the bus can read back nonzero. This doesn't apply to 0x0000-0x00FF of course (single wide
00039             //        characters), but only to the double-wide character set where (c & 0x007F) >= 0x5D.
00040             //        This behavior should be emulated. */
00041             return pc98_font_char_read(a1_font_load_addr,a1_font_char_offset & 0xF,(a1_font_char_offset & 0x20) ? 0 : 1);
00042         default:
00043             break;
00044     };
00045 
00046     return ~0;
00047 }

00051  * a 1986 book published about NEC BIOS and BASIC ROM. */
00052 void pc98_a1_write(Bitu port,Bitu val,Bitu iolen) {
00053     switch (port) {
00054         /* A3,A1 (out only) two JIS bytes that make up the char code */
00055         case 0xA1:
00056             a1_font_load_addr &= 0x00FF;
00057             a1_font_load_addr |= (val & 0xFF) << 8;
00058             break;
00059         case 0xA3:
00060             a1_font_load_addr &= 0xFF00;
00061             a1_font_load_addr |= (val & 0xFF);
00062             break;
00063         case 0xA5:
00064             /* From documentation:
00065              *
00066              *    bit [7:6] = Dont care
00067              *    bit [5]   = L/R
00068              *    bit [4]   = 0
00069              *    bit [3:0] = C3-C0
00070              *
00071              * This so far is consistent with real hardware behavior */
00072             a1_font_char_offset = val;
00073             break;
00074         case 0xA7:
00075             /* TODO: Various controls for the text layer */
00076             break;
00077         case 0xA9: // an 8-bit I/O port to access font RAM by...
00078                    // this is what Touhou Project uses to load fonts.
00079                    // never mind decompiling INT 18h on real hardware shows instead
00080                    // a similar sequence with REP MOVSW to A400:0000...
00081                    //
00082                    // there's a restriction noted with INT 18h AH=1Ah where the only
00083                    // codes you can overwrite are 0xxx76 and 0xxx77. I'm guessing that
00084                    // having 512KB of RAM out there dedicated to nothing but fonts
00085                    // is probably not economical to NEC's bottom line, and this
00086                    // restriction makes me wonder if the font is held in ROM except
00087                    // for this narrow sliver of codes, which map to about 8KB of RAM
00088                    // (128*2*16) * 2 = 8192 bytes
00089                    //
00090                    // I'm also guessing that this RAM is not involved with the single-wide
00091                    // character set, which is why writes to 0x0056/0x0057 are redirected to
00092                    // 0x8056/0x8057. Without this hack, Touhou Project 2 will overwrite
00093                    // the letter 'W' when loading it's font data (Level 1 will show "Eastern  ind"
00094                    // instead of "Eastern Wind" for the music title as a result).
00095                    //
00096                    // On real hardware it seems, attempts to write anywhere outside 0xxx56/0xxx57
00097                    // are ignored. They are not remapped. Attempts to write to 0x0056 are ignored
00098                    // by the hardware (since that conflicts with single-wide chars) but you can
00099                    // write to that cell if you write to 0x8056 instead.
00100             if ((a1_font_load_addr & 0x007E) == 0x0056 && (a1_font_load_addr & 0xFF00) != 0x0000)
00101                 pc98_font_char_write(a1_font_load_addr,a1_font_char_offset & 0xF,(a1_font_char_offset & 0x20) ? 0 : 1,val);
00102             else
00103                 LOG_MSG("A1 port attempt to write FONT ROM char 0x%x",a1_font_load_addr);
00104             break;
00105         default:
00106             LOG_MSG("A1 port %lx val %lx unexpected",port,val);
00107             break;
00108     };
00109 }
