00099                         send_event(1);

00101                 case 0xC0:
00102                         snd_seq_ev_set_pgmchange(&ev, chanID, msg[1]);
00103                         send_event(0);

00105                 case 0xD0:
00106                         snd_seq_ev_set_chanpress(&ev, chanID, msg[1]);
