00314 void MixerChannel::lowpassUpdate() {
00315     if (lowpass_freq != 0) {
00316         double timeInterval;
00317         double tau,talpha;
00318 
00319         if (freq_n > freq_d) { // source -> dest mixer rate ratio is > 100%
00320             timeInterval = (double)freq_d_orig / freq_n; // will filter on sample load at source rate
00321             lowpass_on_load = true;
00322             lowpass_on_out = false;
00323         }
00324         else {
00325             timeInterval = (double)1.0 / mixer.freq; // will filter mixer output at mixer rate
00326             lowpass_on_load = false;
00327             lowpass_on_out = true;
00328         }
00329 
00330         tau = 1.0 / (lowpass_freq * 2 * M_PI);
00331         talpha = timeInterval / (tau + timeInterval);
00332         lowpass_alpha = (Bit32s)(talpha * 0x10000); // double -> 16.16 fixed point
00333

00335 //          freq_n,freq_d_orig,timeInterval,tau,talpha,lowpass_on_load,lowpass_on_out);
00336     }

00338         lowpass_on_load = false;

00344     const Bit64s m1 = (Bit64s)in * (Bit64s)lowpass_alpha;

00436     Bitu upto;
00437

00608 inline bool MixerChannel::runSampleInterpolation(const Bitu upto) {
00609     int sample;
00610 
00611     if (msbuffer_o >= upto)
00612         return false;
00613 
00614     while (freq_fslew < freq_d) {
00615         sample = last[0] + (int)(((int64_t)delta[0] * (int64_t)freq_fslew) / (int64_t)freq_d);
00616         msbuffer[msbuffer_o][0] = sample * volmul[0];
00617         sample = last[1] + (int)(((int64_t)delta[1] * (int64_t)freq_fslew) / (int64_t)freq_d);
00618         msbuffer[msbuffer_o][1] = sample * volmul[1];
00619 
00620         freq_f += freq_n;
00621         freq_fslew += freq_nslew;
00622         if ((++msbuffer_o) >= upto)
00623             return false;
00624     }
00625 
00626     current[0] = last[0] + delta[0];
00627     current[1] = last[1] + delta[1];
00628     while (freq_f < freq_d) {
00629         msbuffer[msbuffer_o][0] = current[0] * volmul[0];
00630         msbuffer[msbuffer_o][1] = current[1] * volmul[1];
00631 
00632         freq_f += freq_n;
00633         if ((++msbuffer_o) >= upto)
00634             return false;
00635     }
00636

00750 /* once a millisecond, render 1ms of audio, up to whole samples */
00751 static void MIXER_MixData(Bitu fracs/*render up to*/) {
00752     unsigned int prev_rendered = mixer.samples_rendered_ms.w;
00753     MixerChannel *chan = mixer.channels;
00754     unsigned int whole,frac;
00755     bool endframe = false;
00756 
00757     if (fracs >= ((Bitu)mixer.samples_this_ms.w * mixer.samples_this_ms.fd)) {
00758         fracs = ((Bitu)mixer.samples_this_ms.w * mixer.samples_this_ms.fd);
00759         endframe = true;
00760     }
00761 
00762     whole = (unsigned int)(fracs / mixer.samples_this_ms.fd);
00763     frac = (unsigned int)(fracs % mixer.samples_this_ms.fd);
00764     if (whole <= mixer.samples_rendered_ms.w) return;
00765 
00766     while (chan) {
00767         chan->Mix(whole,fracs);
00768         if (endframe) chan->EndFrame(mixer.samples_this_ms.w);
00769         chan=chan->next;
00770     }
00771 
00772     if (CaptureState & (CAPTURE_WAVE|CAPTURE_VIDEO)) {
00773         Bit32s volscale1 = (Bit32s)(mixer.recordvol[0] * (1 << MIXER_VOLSHIFT));
00774         Bit32s volscale2 = (Bit32s)(mixer.recordvol[1] * (1 << MIXER_VOLSHIFT));
00775         Bit16s convert[1024][2];
00776         Bitu added = whole - prev_rendered;
00777         if (added>1024) added=1024;
00778         Bitu readpos = mixer.work_in + prev_rendered;
00779         for (Bitu i=0;i<added;i++) {
00780             convert[i][0]=MIXER_CLIP(((Bit64s)mixer.work[readpos][0] * (Bit64s)volscale1) >> (MIXER_VOLSHIFT + MIXER_VOLSHIFT));
00781             convert[i][1]=MIXER_CLIP(((Bit64s)mixer.work[readpos][1] * (Bit64s)volscale2) >> (MIXER_VOLSHIFT + MIXER_VOLSHIFT));
00782             readpos++;
00783         }
00784         assert(readpos <= MIXER_BUFSIZE);
00785         CAPTURE_AddWave( mixer.freq, added, (Bit16s*)convert );
00786     }
00787 
00788     if (Mixer_MIXC_Active() && prev_rendered < whole) {
00789         Bitu readpos = mixer.work_in + prev_rendered;
00790         Bitu added = whole - prev_rendered;
00791         Bitu cando = (mixer_capture_write_end - mixer_capture_write) / 2/*bytes/sample*/ / 2/*channels*/;
00792         if (cando > added) cando = added;
00793 
00794         if (cando == 0 && !Mixer_MIXC_AtEnd()) {
00795             Mixer_MIXC_MarkError();
00796         }
00797         else if (cando != 0) {
00798             for (Bitu i=0;i < cando;i++) {
00799                 phys_writew(mixer_capture_write,(Bit16u)MIXER_CLIP(((Bit64s)mixer.work[readpos][0]) >> (MIXER_VOLSHIFT)));
00800                 mixer_capture_write += 2;
00801 
00802                 phys_writew(mixer_capture_write,(Bit16u)MIXER_CLIP(((Bit64s)mixer.work[readpos][1]) >> (MIXER_VOLSHIFT)));
00803                 mixer_capture_write += 2;
00804 
00805                 readpos++;
00806             }
00807 
00808             if (Mixer_MIXC_AtEnd()) {
00809                 if (Mixer_MIXC_ShouldLoop())
00810                     Mixer_MIXC_LoopAround();
00811                 else
00812                     Mixer_MIXC_Stop();
00813             }
00814         }
00815     }
00816 
00817     mixer.samples_rendered_ms.w = whole;
00818     mixer.samples_rendered_ms.fd = frac;
00819     mixer_sample_counter += mixer.samples_rendered_ms.w - prev_rendered;
00820 }

00839 static void MIXER_Mix(void) {
00840     Bitu thr;
00841 
00842     SDL_LockAudio();
00843 
00844     /* render */
00845     assert((mixer.work_in+mixer.samples_per_ms.w) <= MIXER_BUFSIZE);
00846     MIXER_MixData((Bitu)mixer.samples_this_ms.w * (Bitu)mixer.samples_this_ms.fd);
00847     mixer.work_in += mixer.samples_this_ms.w;
00848 
00849     /* how many samples for the next ms? */
00850     mixer.samples_this_ms.w = mixer.samples_per_ms.w;
00851     mixer.samples_this_ms.fn += mixer.samples_per_ms.fn;
00852     if (mixer.samples_this_ms.fn >= mixer.samples_this_ms.fd) {
00853         mixer.samples_this_ms.fn -= mixer.samples_this_ms.fd;
00854         mixer.samples_this_ms.w++;
00855     }
00856 
00857     /* advance. we use in/out & wrap pointers to make sure the rendering code
00858      * doesn't have to worry about circular buffer wraparound. */
00859     thr = mixer.blocksize;
00860     if (thr < mixer.samples_this_ms.w) thr = mixer.samples_this_ms.w;
00861     if ((mixer.work_in+thr) > MIXER_BUFSIZE) {
00862         mixer.work_wrap = mixer.work_in;
00863         mixer.work_in = 0;
00864     }
00865     assert((mixer.work_in+thr) <= MIXER_BUFSIZE);
00866     assert((mixer.work_in+mixer.samples_this_ms.w) <= MIXER_BUFSIZE);
00867     memset(&mixer.work[mixer.work_in][0],0,sizeof(Bit32s)*2*mixer.samples_this_ms.w);
00868     mixer.samples_rendered_ms.fn = 0;
00869     mixer.samples_rendered_ms.w = 0;
00870     SDL_UnlockAudio();
00871     MIXER_FillUp();
00872 }

00874 static void SDLCALL MIXER_CallBack(void * userdata, Uint8 *stream, int len) {
00875     (void)userdata;//UNUSED
00876     Bit32s volscale1 = (Bit32s)(mixer.mastervol[0] * (1 << MIXER_VOLSHIFT));
00877     Bit32s volscale2 = (Bit32s)(mixer.mastervol[1] * (1 << MIXER_VOLSHIFT));
00878     Bitu need = (Bitu)len/MIXER_SSIZE;
00879     Bit16s *output = (Bit16s*)stream;
00880     int remains;
00881     Bit32s *in;
00882 
00883     if (mixer.mute) {
00884         if ((CaptureState & (CAPTURE_WAVE|CAPTURE_VIDEO|CAPTURE_MULTITRACK_WAVE)) != 0)
00885             mixer.work_out = mixer.work_in;
00886         else
00887             mixer.work_out = mixer.work_in = 0;
00888     }
00889 
00890     if (mixer.prebuffer_wait) {
00891         remains = (int)mixer.work_in - (int)mixer.work_out;
00892         if (remains < 0) remains += (int)mixer.work_wrap;
00893         if (remains < 0) remains = 0;
00894 
00895         if ((unsigned int)remains >= mixer.prebuffer_samples)
00896             mixer.prebuffer_wait = false;
00897     }
00898 
00899     if (!mixer.prebuffer_wait && !mixer.mute) {
00900         in = &mixer.work[mixer.work_out][0];
00901         while (need > 0) {
00902             if (mixer.work_out == mixer.work_in) break;
00903             *output++ = MIXER_CLIP((((Bit64s)(*in++)) * (Bit64s)volscale1) >> (MIXER_VOLSHIFT + MIXER_VOLSHIFT));
00904             *output++ = MIXER_CLIP((((Bit64s)(*in++)) * (Bit64s)volscale2) >> (MIXER_VOLSHIFT + MIXER_VOLSHIFT));
00905             mixer.work_out++;
00906             if (mixer.work_out >= mixer.work_wrap) {
00907                 mixer.work_out = 0;
00908                 in = &mixer.work[mixer.work_out][0];
00909             }
00910             need--;
00911         }
00912     }
00913 
00914     if (need > 0)
00915         mixer.prebuffer_wait = true;
00916 
00917     while (need > 0) {
00918         *output++ = 0;
00919         *output++ = 0;
00920         need--;
00921     }
00922 
00923     remains = (int)mixer.work_in - (int)mixer.work_out;
00924     if (remains < 0) remains += (int)mixer.work_wrap;
00925 
00926     if ((unsigned long)remains >= (mixer.blocksize*2UL)) {
00927         /* drop some samples to keep time */
00928         unsigned int drop;
00929 
00930         if ((unsigned long)remains >= (mixer.blocksize*3UL)) // hard drop
00931             drop = ((unsigned int)remains - (unsigned int)(mixer.blocksize));
00932         else // subtle drop
00933             drop = (((unsigned int)remains - (unsigned int)(mixer.blocksize*2)) / 50U) + 1;
00934 
00935         while (drop > 0) {
00936             mixer.work_out++;
00937             if (mixer.work_out >= mixer.work_wrap) mixer.work_out = 0;
00938             drop--;
00939         }
00940     }
00941 }
