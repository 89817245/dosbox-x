00231 inline Bit32s MixerChannel::lowpassStep(Bit32s in,const unsigned int iteration,const unsigned int channel) {
00232         const Bit64s m1 = (Bit64s)in * (Bit64s)lowpass_alpha;
00233         const Bit64s m2 = ((Bit64s)lowpass[iteration][channel] << ((Bit64s)16)) - ((Bit64s)lowpass[iteration][channel] * (Bit64s)lowpass_alpha);
00234         const Bit32s ns = (Bit32s)((m1 + m2) >> (Bit64s)16);

00323         unsigned int patience = 2;
00324         Bitu upto;
00325 
00326         if (whole <= rend_n) return;
00327         assert(whole <= mixer.samples_this_ms.w);
00328         assert(rend_n < mixer.samples_this_ms.w);
00329         Bit32s *outptr = &mixer.work[mixer.work_in+rend_n][0];
00330 
00331         if (!enabled) {
00332                 rend_n = whole;

00335         }
00336 
00337         // HACK: We iterate twice only because of the Sound Blaster emulation. No other emulation seems to need this.
00338         rendering_to_n = whole;
00339         rendering_to_d = frac;
00340         while (msbuffer_o < whole) {
00341                 Bit64u todo = (Bit64u)(whole - msbuffer_o) * (Bit64u)freq_n;
00342                 todo += (Bit64u)freq_f;
00343                 todo += (Bit64u)freq_d - (Bit64u)1;
00344                 todo /= (Bit64u)freq_d;
00345                 if (!current_loaded) todo++;
00346                 handler(todo);
00347 
00348                 if (--patience == 0) break;

00350 
00351         if (msbuffer_o < whole)
00352                 padFillSampleInterpolation(whole);
00353 
00354         upto = whole;

00360 
00361                 while (t_rend_n < whole && t_msbuffer_i < upto) {

00636 static void MIXER_MixData(Bitu fracs/*render up to*/) {
00637         unsigned int prev_rendered = mixer.samples_rendered_ms.w;

00641 
00642         if (fracs >= (Bitu)(mixer.samples_this_ms.w * mixer.samples_this_ms.fd)) {
00643                 fracs = (Bitu)(mixer.samples_this_ms.w * mixer.samples_this_ms.fd);
00644                 endframe = true;
00645         }
00646 
00647         whole = (unsigned int)(fracs / mixer.samples_this_ms.fd);
00648         frac = (unsigned int)(fracs % mixer.samples_this_ms.fd);
00649         if (whole <= mixer.samples_rendered_ms.w) return;

00652                 chan->Mix(whole,fracs);
00653                 if (endframe) chan->EndFrame(mixer.samples_this_ms.w);
00654                 chan=chan->next;
00655         }
00656 
00657         if (CaptureState & (CAPTURE_WAVE|CAPTURE_VIDEO)) {
00658         Bit32s volscale1 = (Bit32s)(mixer.recordvol[0] * (1 << MIXER_VOLSHIFT));
00659         Bit32s volscale2 = (Bit32s)(mixer.recordvol[1] * (1 << MIXER_VOLSHIFT));
00660         Bit16s convert[1024][2];
00661                 Bitu added = whole - prev_rendered;
00662                 if (added>1024) added=1024;
00663                 Bitu readpos = mixer.work_in + prev_rendered;
00664                 for (Bitu i=0;i<added;i++) {
00665             convert[i][0]=MIXER_CLIP(((Bit64s)mixer.work[readpos][0] * (Bit64s)volscale1) >> (MIXER_VOLSHIFT + MIXER_VOLSHIFT));
00666             convert[i][1]=MIXER_CLIP(((Bit64s)mixer.work[readpos][1] * (Bit64s)volscale2) >> (MIXER_VOLSHIFT + MIXER_VOLSHIFT));
00667             readpos++;
00668                 }
00669                 assert(readpos <= MIXER_BUFSIZE);
00670                 CAPTURE_AddWave( mixer.freq, added, (Bit16s*)convert );
00671         }
00672 
00673         mixer.samples_rendered_ms.w = whole;
00674         mixer.samples_rendered_ms.fd = frac;
00675         mixer_sample_counter += mixer.samples_rendered_ms.w - prev_rendered;

00690 static void MIXER_Mix(void) {
00691         Bitu thr;
00692 
00693         SDL_LockAudio();
00694 
00695         /* render */
00696         assert((mixer.work_in+mixer.samples_per_ms.w) <= MIXER_BUFSIZE);
00697         MIXER_MixData((Bitu)mixer.samples_this_ms.w * (Bitu)mixer.samples_this_ms.fd);
00698         mixer.work_in += mixer.samples_this_ms.w;
00699 
00700         /* how many samples for the next ms? */
00701         mixer.samples_this_ms.w = mixer.samples_per_ms.w;
00702         mixer.samples_this_ms.fn += mixer.samples_per_ms.fn;
00703         if (mixer.samples_this_ms.fn >= mixer.samples_this_ms.fd) {
00704                 mixer.samples_this_ms.fn -= mixer.samples_this_ms.fd;
00705                 mixer.samples_this_ms.w++;
00706         }
00707 
00708         /* advance. we use in/out & wrap pointers to make sure the rendering code
00709          * doesn't have to worry about circular buffer wraparound. */
00710         thr = mixer.blocksize;
00711         if (thr < mixer.samples_this_ms.w) thr = mixer.samples_this_ms.w;
00712         if ((mixer.work_in+thr) > MIXER_BUFSIZE) {
00713                 mixer.work_wrap = mixer.work_in;
00714                 mixer.work_in = 0;
00715         }
00716         assert((mixer.work_in+thr) <= MIXER_BUFSIZE);
00717         assert((mixer.work_in+mixer.samples_this_ms.w) <= MIXER_BUFSIZE);
00718         memset(&mixer.work[mixer.work_in][0],0,sizeof(Bit32s)*2*mixer.samples_this_ms.w);
00719         mixer.samples_rendered_ms.fn = 0;
00720         mixer.samples_rendered_ms.w = 0;
00721         SDL_UnlockAudio();
00722         MIXER_FillUp();
00723 }

00725 static void MIXER_CallBack(void * userdata, Uint8 *stream, int len) {
00726     Bit32s volscale1 = (Bit32s)(mixer.mastervol[0] * (1 << MIXER_VOLSHIFT));
00727     Bit32s volscale2 = (Bit32s)(mixer.mastervol[1] * (1 << MIXER_VOLSHIFT));
00728         Bitu need = (Bitu)len/MIXER_SSIZE;
00729         Bit16s *output = (Bit16s*)stream;
00730         int remains;
00731         Bit32s *in;
00732 
00733     if (mixer.prebuffer_wait) {
00734         remains = (int)mixer.work_in - (int)mixer.work_out;
00735         if (remains < 0) remains += mixer.work_wrap;
00736 
00737         if (remains >= mixer.prebuffer_samples)
00738             mixer.prebuffer_wait = false;
00739     }
00740 
00741     if (!mixer.prebuffer_wait) {
00742         in = &mixer.work[mixer.work_out][0];
00743         while (need > 0) {
00744             if (mixer.work_out == mixer.work_in) break;
00745             *output++ = MIXER_CLIP((((Bit64s)(*in++)) * (Bit64s)volscale1) >> (MIXER_VOLSHIFT + MIXER_VOLSHIFT));
00746             *output++ = MIXER_CLIP((((Bit64s)(*in++)) * (Bit64s)volscale2) >> (MIXER_VOLSHIFT + MIXER_VOLSHIFT));
00747             mixer.work_out++;
00748             if (mixer.work_out >= mixer.work_wrap) {
00749                 mixer.work_out = 0;
00750                 in = &mixer.work[mixer.work_out][0];
00751             }
00752             need--;
00753         }
00754     }
00755 
00756     if (need > 0)
00757         mixer.prebuffer_wait = true;
00758 
00759         while (need > 0) {
00760                 *output++ = 0;
00761                 *output++ = 0;
00762                 need--;
00763         }
00764 
00765         remains = (int)mixer.work_in - (int)mixer.work_out;
00766         if (remains < 0) remains += mixer.work_wrap;
00767 
00768         if ((unsigned long)remains >= (mixer.blocksize*2UL)) {
00769                 /* drop some samples to keep time */
00770                 unsigned int drop;
00771 
00772                 if ((unsigned long)remains >= (mixer.blocksize*3UL)) // hard drop
00773                         drop = ((unsigned int)remains - (unsigned int)(mixer.blocksize));
00774                 else // subtle drop
00775                         drop = (((unsigned int)remains - (unsigned int)(mixer.blocksize*2)) / 50U) + 1;
00776 
00777                 while (drop > 0) {
00778                         mixer.work_out++;
00779                         if (mixer.work_out >= mixer.work_wrap) mixer.work_out = 0;
00780                         drop--;
00781                 }
00782         }
