00400       {
00401        BX_ERROR("ne2K: dma read underrun iolen=%d remote_bytes=%d",io_len,BX_NE2K_THIS s.remote_bytes);

00601 {
00602   BX_DEBUG("page 0 write to port %04x, len=%u", (unsigned) offset,
00603            (unsigned) io_len);
00604 
00605   // It appears to be a common practice to use outw on page0 regs...
00606 
00607   // break up outw into two outb's
00608   if (io_len == 2) {
00609     page0_write(offset, (value & 0xff), 1);
00610     page0_write(offset + 1, ((value >> 8) & 0xff), 1);
00611     return;
00612   }
00613 
00614   switch (offset) {
00615   case 0x1:  // PSTART
00616     BX_NE2K_THIS s.page_start = value;
00617     break;
00618 
00619   case 0x2:  // PSTOP
00620         // BX_INFO(("Writing to PSTOP: %02x", value));
00621     BX_NE2K_THIS s.page_stop = value;
00622     break;
00623 
00624   case 0x3:  // BNRY
00625     BX_NE2K_THIS s.bound_ptr = value;
00626     break;
00627 
00628   case 0x4:  // TPSR
00629     BX_NE2K_THIS s.tx_page_start = value;
00630     break;
00631 
00632   case 0x5:  // TBCR0
00633     // Clear out low byte and re-insert
00634     BX_NE2K_THIS s.tx_bytes &= 0xff00;
00635     BX_NE2K_THIS s.tx_bytes |= (value & 0xff);
00636     break;
00637 
00638   case 0x6:  // TBCR1
00639     // Clear out high byte and re-insert
00640     BX_NE2K_THIS s.tx_bytes &= 0x00ff;
00641     BX_NE2K_THIS s.tx_bytes |= ((value & 0xff) << 8);
00642     break;
00643 
00644   case 0x7:  // ISR
00645     value &= 0x7f;  // clear RST bit - status-only bit
00646     // All other values are cleared iff the ISR bit is 1

00648         BX_NE2K_THIS s.ISR.pkt_tx    &= ~((bx_bool)((value & 0x02) == 0x02));

00650     BX_NE2K_THIS s.ISR.tx_err    &= ~((bx_bool)((value & 0x08) == 0x08));

00652     BX_NE2K_THIS s.ISR.cnt_oflow &= ~((bx_bool)((value & 0x20) == 0x20));
00653     BX_NE2K_THIS s.ISR.rdma_done &= ~((bx_bool)((value & 0x40) == 0x40));
00654     value = ((BX_NE2K_THIS s.ISR.rdma_done << 6) |
00655              (BX_NE2K_THIS s.ISR.cnt_oflow << 5) |
00656              (BX_NE2K_THIS s.ISR.overwrite << 4) |
00657              (BX_NE2K_THIS s.ISR.tx_err    << 3) |
00658              (BX_NE2K_THIS s.ISR.rx_err    << 2) |
00659              (BX_NE2K_THIS s.ISR.pkt_tx    << 1) |
00660              (BX_NE2K_THIS s.ISR.pkt_rx));
00661     value &= ((BX_NE2K_THIS s.IMR.rdma_inte << 6) |
00662               (BX_NE2K_THIS s.IMR.cofl_inte << 5) |
00663               (BX_NE2K_THIS s.IMR.overw_inte << 4) |
00664               (BX_NE2K_THIS s.IMR.txerr_inte << 3) |
00665               (BX_NE2K_THIS s.IMR.rxerr_inte << 2) |
00666               (BX_NE2K_THIS s.IMR.tx_inte << 1) |
00667               (BX_NE2K_THIS s.IMR.rx_inte));

00669           PIC_DeActivateIRQ(s.base_irq);
00670       //DEV_pic_lower_irq(BX_NE2K_THIS s.base_irq);
00671     break;
00672 
00673   case 0x8:  // RSAR0
00674     // Clear out low byte and re-insert
00675     BX_NE2K_THIS s.remote_start &= 0xff00;
00676     BX_NE2K_THIS s.remote_start |= (value & 0xff);
00677     BX_NE2K_THIS s.remote_dma = BX_NE2K_THIS s.remote_start;
00678     break;
00679 
00680   case 0x9:  // RSAR1
00681     // Clear out high byte and re-insert
00682     BX_NE2K_THIS s.remote_start &= 0x00ff;
00683     BX_NE2K_THIS s.remote_start |= ((value & 0xff) << 8);
00684     BX_NE2K_THIS s.remote_dma = BX_NE2K_THIS s.remote_start;
00685     break;
00686 
00687   case 0xa:  // RBCR0
00688     // Clear out low byte and re-insert
00689     BX_NE2K_THIS s.remote_bytes &= 0xff00;
00690     BX_NE2K_THIS s.remote_bytes |= (value & 0xff);
00691     break;
00692 
00693   case 0xb:  // RBCR1
00694     // Clear out high byte and re-insert
00695     BX_NE2K_THIS s.remote_bytes &= 0x00ff;
00696     BX_NE2K_THIS s.remote_bytes |= ((value & 0xff) << 8);
00697     break;
00698 
00699   case 0xc:  // RCR
00700     // Check if the reserved bits are set
00701     if (value & 0xc0)
00702       BX_INFO(("RCR write, reserved bits set"));
00703 
00704     // Set all other bit-fields
00705     BX_NE2K_THIS s.RCR.errors_ok = ((value & 0x01) == 0x01);
00706     BX_NE2K_THIS s.RCR.runts_ok  = ((value & 0x02) == 0x02);
00707     BX_NE2K_THIS s.RCR.broadcast = ((value & 0x04) == 0x04);
00708     BX_NE2K_THIS s.RCR.multicast = ((value & 0x08) == 0x08);
00709     BX_NE2K_THIS s.RCR.promisc   = ((value & 0x10) == 0x10);
00710     BX_NE2K_THIS s.RCR.monitor   = ((value & 0x20) == 0x20);
00711 
00712     // Monitor bit is a little suspicious...
00713     if (value & 0x20)
00714       BX_INFO(("RCR write, monitor bit set!"));
00715     break;
00716 
00717   case 0xd:  // TCR
00718     // Check reserved bits
00719     if (value & 0xe0)
00720       BX_ERROR(("TCR write, reserved bits set"));
00721 
00722     // Test loop mode (not supported)
00723     if (value & 0x06) {
00724       BX_NE2K_THIS s.TCR.loop_cntl = (value & 0x6) >> 1;
00725       BX_INFO("TCR write, loop mode %d not supported", BX_NE2K_THIS s.TCR.loop_cntl);
00726     } else {
00727       BX_NE2K_THIS s.TCR.loop_cntl = 0;
00728     }
00729 
00730     // Inhibit-CRC not supported.
00731     if (value & 0x01)
00732       BX_PANIC(("TCR write, inhibit-CRC not supported"));
00733 
00734     // Auto-transmit disable very suspicious
00735     if (value & 0x08)
00736       BX_PANIC(("TCR write, auto transmit disable not supported"));
00737 
00738     // Allow collision-offset to be set, although not used
00739     BX_NE2K_THIS s.TCR.coll_prio = ((value & 0x08) == 0x08);
00740     break;
00741 
00742   case 0xe:  // DCR
00743     // the loopback mode is not suppported yet
00744     if (!(value & 0x08)) {
00745       BX_ERROR(("DCR write, loopback mode selected"));
00746     }
00747     // It is questionable to set longaddr and auto_rx, since they
00748     // aren't supported on the ne2000. Print a warning and continue
00749     if (value & 0x04)
00750       BX_INFO(("DCR write - LAS set ???"));
00751     if (value & 0x10)
00752       BX_INFO(("DCR write - AR set ???"));
00753 
00754     // Set other values.
00755     BX_NE2K_THIS s.DCR.wdsize   = ((value & 0x01) == 0x01);
00756     BX_NE2K_THIS s.DCR.endian   = ((value & 0x02) == 0x02);
00757     BX_NE2K_THIS s.DCR.longaddr = ((value & 0x04) == 0x04); // illegal ?
00758     BX_NE2K_THIS s.DCR.loop     = ((value & 0x08) == 0x08);
00759     BX_NE2K_THIS s.DCR.auto_rx  = ((value & 0x10) == 0x10); // also illegal ?
00760     BX_NE2K_THIS s.DCR.fifo_size = (value & 0x50) >> 5;
00761     break;
00762 
00763   case 0xf:  // IMR
00764     // Check for reserved bit
00765         if (value & 0x80)
00766       BX_PANIC(("IMR write, reserved bit set"));
00767 
00768     // Set other values
00769     BX_NE2K_THIS s.IMR.rx_inte    = ((value & 0x01) == 0x01);
00770     BX_NE2K_THIS s.IMR.tx_inte    = ((value & 0x02) == 0x02);
00771     BX_NE2K_THIS s.IMR.rxerr_inte = ((value & 0x04) == 0x04);
00772     BX_NE2K_THIS s.IMR.txerr_inte = ((value & 0x08) == 0x08);
00773     BX_NE2K_THIS s.IMR.overw_inte = ((value & 0x10) == 0x10);
00774     BX_NE2K_THIS s.IMR.cofl_inte  = ((value & 0x20) == 0x20);
00775     BX_NE2K_THIS s.IMR.rdma_inte  = ((value & 0x40) == 0x40);
00776         if(BX_NE2K_THIS s.ISR.pkt_tx && BX_NE2K_THIS s.IMR.tx_inte) {
00777           LOG_MSG("tx irq retrigger");
00778           PIC_ActivateIRQ(s.base_irq);
00779         }
00780     break;
00781   default:
00782     BX_PANIC("page 0 write, bad offset %0x", offset);
00783   }
00784 }

01184 /*
01185  * mcast_index() - return the 6-bit index into the multicast
01186  * table. Stolen unashamedly from FreeBSD's if_ed.c
01187  */
01188 unsigned
01189 bx_ne2k_c::mcast_index(const void *dst)
01190 {
01191 #define POLYNOMIAL 0x04c11db6
01192   unsigned long crc = 0xffffffffL;
01193   int carry, i, j;
01194   unsigned char b;
01195   unsigned char *ep = (unsigned char *) dst;
01196 
01197   for (i = 6; --i >= 0;) {
01198     b = *ep++;
01199     for (j = 8; --j >= 0;) {
01200       carry = ((crc & 0x80000000L) ? 1 : 0) ^ (b & 0x01);
01201       crc <<= 1;
01202       b >>= 1;
01203       if (carry)
01204         crc = ((crc ^ POLYNOMIAL) | carry);
01205     }
01206   }
01207   return (crc >> 26);
01208 #undef POLYNOMIAL
01209 }

01225  * rx_frame() - called by the platform-specific code when an
01226  * ethernet frame has been received. The destination address
01227  * is tested to see if it should be accepted, and if the
01228  * rx ring has enough room, it is copied into it and
01229  * the receive process is updated
01230  */
01231 void
01232 bx_ne2k_c::rx_frame(const void *buf, unsigned io_len)
01233 {
01234   int pages;
01235   int avail;
01236   unsigned idx;
01237 //  int wrapped;
01238   int nextpage;
01239   unsigned char pkthdr[4];
01240   unsigned char *pktbuf = (unsigned char *) buf;
01241   unsigned char *startptr;
01242   static unsigned char bcast_addr[6] = {0xff,0xff,0xff,0xff,0xff,0xff};
01243 
01244   if(io_len != 60) {
01245         BX_DEBUG("rx_frame with length %d", io_len);
01246   }
01247 
01248         //LOG_MSG("stop=%d, pagestart=%x, dcr_loop=%x, tcr_loopcntl=%x",
01249         //      BX_NE2K_THIS s.CR.stop, BX_NE2K_THIS s.page_start,
01250         //      BX_NE2K_THIS s.DCR.loop, BX_NE2K_THIS s.TCR.loop_cntl);
01251   if ((BX_NE2K_THIS s.CR.stop != 0) ||
01252       (BX_NE2K_THIS s.page_start == 0) /*||
01253       ((BX_NE2K_THIS s.DCR.loop == 0) &&
01254        (BX_NE2K_THIS s.TCR.loop_cntl != 0))*/) {
01255     return;
01256   }
01257 
01258   // Add the pkt header + CRC to the length, and work
01259   // out how many 256-byte pages the frame would occupy
01260   pages = (io_len + 4 + 4 + 255)/256;
01261 
01262   if (BX_NE2K_THIS s.curr_page < BX_NE2K_THIS s.bound_ptr) {
01263     avail = BX_NE2K_THIS s.bound_ptr - BX_NE2K_THIS s.curr_page;    
01264   } else {
01265     avail = (BX_NE2K_THIS s.page_stop - BX_NE2K_THIS s.page_start) -
01266       (BX_NE2K_THIS s.curr_page - BX_NE2K_THIS s.bound_ptr);
01267 //    wrapped = 1;
01268   }
01269 
01270   // Avoid getting into a buffer overflow condition by not attempting
01271   // to do partial receives. The emulation to handle this condition
01272   // seems particularly painful.
01273   if ((avail < pages) 
01274 #if BX_NE2K_NEVER_FULL_RING
01275       || (avail == pages)
01276 #endif
01277       ) {
01278         BX_DEBUG("no space");
01279     return;
01280   }
01281 
01282   if ((io_len < 40/*60*/) && !BX_NE2K_THIS s.RCR.runts_ok) {
01283     BX_DEBUG("rejected small packet, length %d", io_len);
01284     return;
01285   }
01286   // some computers don't care...
01287   if (io_len < 60) io_len=60;
01288 
01289   // Do address filtering if not in promiscuous mode
01290   if (! BX_NE2K_THIS s.RCR.promisc) {
01291     if (!memcmp(buf, bcast_addr, 6)) {
01292       if (!BX_NE2K_THIS s.RCR.broadcast) {
01293         return;
01294       }
01295     } else if (pktbuf[0] & 0x01) {
01296         if (! BX_NE2K_THIS s.RCR.multicast) {
01297             return;
01298         }
01299       idx = mcast_index(buf);
01300       if (!(BX_NE2K_THIS s.mchash[idx >> 3] & (1 << (idx & 0x7)))) {
01301         return;
01302       }
01303     } else if (0 != memcmp(buf, BX_NE2K_THIS s.physaddr, 6)) {
01304       return;
01305     }
01306   } else {
01307       BX_DEBUG(("rx_frame promiscuous receive"));
01308   }
01309 
01310     BX_INFO("rx_frame %d to %x:%x:%x:%x:%x:%x from %x:%x:%x:%x:%x:%x",
01311            io_len,
01312            pktbuf[0], pktbuf[1], pktbuf[2], pktbuf[3], pktbuf[4], pktbuf[5],
01313            pktbuf[6], pktbuf[7], pktbuf[8], pktbuf[9], pktbuf[10], pktbuf[11]);
01314 
01315   nextpage = BX_NE2K_THIS s.curr_page + pages;
01316   if (nextpage >= BX_NE2K_THIS s.page_stop) {
01317     nextpage -= BX_NE2K_THIS s.page_stop - BX_NE2K_THIS s.page_start;
01318   }
01319 
01320   // Setup packet header
01321   pkthdr[0] = 0;        // rx status - old behavior
01322   pkthdr[0] = 1;        // Probably better to set it all the time
01323                         // rather than set it to 0, which is clearly wrong.
01324   if (pktbuf[0] & 0x01) {
01325     pkthdr[0] |= 0x20;  // rx status += multicast packet
01326   }
01327   pkthdr[1] = nextpage; // ptr to next packet
01328   pkthdr[2] = (io_len + 4) & 0xff;      // length-low
01329   pkthdr[3] = (io_len + 4) >> 8;        // length-hi
01330 
01331   // copy into buffer, update curpage, and signal interrupt if config'd
01332   startptr = & BX_NE2K_THIS s.mem[BX_NE2K_THIS s.curr_page * 256 -
01333                                BX_NE2K_MEMSTART];
01334   if ((nextpage > BX_NE2K_THIS s.curr_page) ||
01335       ((BX_NE2K_THIS s.curr_page + pages) == BX_NE2K_THIS s.page_stop)) {
01336     memcpy(startptr, pkthdr, 4);
01337     memcpy(startptr + 4, buf, io_len);
01338     BX_NE2K_THIS s.curr_page = nextpage;
01339   } else {
01340     int endbytes = (BX_NE2K_THIS s.page_stop - BX_NE2K_THIS s.curr_page) 
01341       * 256;
01342     memcpy(startptr, pkthdr, 4);
01343     memcpy(startptr + 4, buf, endbytes - 4);
01344     startptr = & BX_NE2K_THIS s.mem[BX_NE2K_THIS s.page_start * 256 -
01345                                  BX_NE2K_MEMSTART];
01346     memcpy(startptr, (void *)(pktbuf + endbytes - 4),
01347            io_len - endbytes + 8);    
01348     BX_NE2K_THIS s.curr_page = nextpage;
01349   }
01350   
01351   BX_NE2K_THIS s.RSR.rx_ok = 1;
01352   if (pktbuf[0] & 0x80) {
01353     BX_NE2K_THIS s.RSR.rx_mbit = 1;
01354   }
01355 
01356   BX_NE2K_THIS s.ISR.pkt_rx = 1;
01357 
01358   if (BX_NE2K_THIS s.IMR.rx_inte) {
01359         //LOG_MSG("packet rx interrupt");
01360           PIC_ActivateIRQ(s.base_irq);
01361     //DEV_pic_raise_irq(BX_NE2K_THIS s.base_irq);
01362   } //else LOG_MSG("no packet rx interrupt");
01363 
01364 }
