00083 void DOS_GetMemory_Choose() {
00084         if (DOS_PRIVATE_SEGMENT == 0) {
00085                 if (mainline_compatible_mapping) {
00086                         /* DOSBox mainline compatible: private area 0xC800-0xCFFF */
00087                         DOS_PRIVATE_SEGMENT=0xc800;
00088                         DOS_PRIVATE_SEGMENT_END=0xc800 + DOS_PRIVATE_SEGMENT_Size;
00089                 }
00090                 else {
00091                         /* DOSBox-X non-compatible: Position ourself just past the VGA BIOS */
00092                         /* NTS: Code has been arranged so that DOS kernel init follows BIOS INT10h init */
00093                         DOS_PRIVATE_SEGMENT=VGA_BIOS_SEG_END;
00094                         DOS_PRIVATE_SEGMENT_END=DOS_PRIVATE_SEGMENT + DOS_PRIVATE_SEGMENT_Size;
00095                 }
00096 
00097                 if (DOS_PRIVATE_SEGMENT >= 0xA000) {
00098                         memset(GetMemBase()+(DOS_PRIVATE_SEGMENT<<4),0x00,(DOS_PRIVATE_SEGMENT_END-DOS_PRIVATE_SEGMENT)<<4);
00099                         MEM_map_RAM_physmem(DOS_PRIVATE_SEGMENT<<4,(DOS_PRIVATE_SEGMENT_END<<4)-1);
00100                 }
00101 
00102                 LOG(LOG_MISC,LOG_DEBUG)("DOS private segment set to 0x%04x-0x%04x",DOS_PRIVATE_SEGMENT,DOS_PRIVATE_SEGMENT_END-1);
00103         }
