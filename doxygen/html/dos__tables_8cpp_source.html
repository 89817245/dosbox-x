00088 void DOS_GetMemory_Choose() {
00089         if (DOS_PRIVATE_SEGMENT == 0) {
00090         /* DOSBox-X non-compatible: Position ourself just past the VGA BIOS */
00091         /* NTS: Code has been arranged so that DOS kernel init follows BIOS INT10h init */
00092         DOS_PRIVATE_SEGMENT=(Bit16u)VGA_BIOS_SEG_END;
00093         DOS_PRIVATE_SEGMENT_END= (Bit16u)(DOS_PRIVATE_SEGMENT + DOS_PRIVATE_SEGMENT_Size);
00094 
00095         if (IS_PC98_ARCH) {
00096             bool PC98_FM_SoundBios_Enabled(void);
00097 
00098             /* Do not let the private segment overlap with anything else after segment C800:0000 including the SOUND ROM at CC00:0000.
00099              * Limiting to 32KB also leaves room for UMBs if enabled between C800:0000 and the EMS page frame at (usually) D000:0000 */
00100             unsigned int limit = 0xD000;
00101 
00102             if (PC98_FM_SoundBios_Enabled()) {
00103                 // TODO: What about sound BIOSes larger than 16KB?
00104                 if (limit > 0xCC00)
00105                     limit = 0xCC00;
00106             }
00107 
00108             if (DOS_User_Wants_UMBs()) {
00109                 // leave room for UMBs, things are cramped a bit in PC-98 mode
00110                 if (limit > 0xC600)
00111                     limit = 0xC600;
00112             }
00113 
00114             if (DOS_PRIVATE_SEGMENT_END > limit)
00115                 DOS_PRIVATE_SEGMENT_END = limit;
00116 
00117             if (DOS_PRIVATE_SEGMENT >= DOS_PRIVATE_SEGMENT_END)
00118                 E_Exit("Insufficient room in upper memory area for private area");
00119         }
00120 
00121                 if (DOS_PRIVATE_SEGMENT >= 0xA000) {
00122                         memset(GetMemBase()+((Bitu)DOS_PRIVATE_SEGMENT<<4u),0x00,(Bitu)(DOS_PRIVATE_SEGMENT_END-DOS_PRIVATE_SEGMENT)<<4u);
00123                         MEM_map_RAM_physmem((Bitu)DOS_PRIVATE_SEGMENT<<4u,((Bitu)DOS_PRIVATE_SEGMENT_END<<4u)-1u);
00124                 }
00125 
00126                 LOG(LOG_MISC,LOG_DEBUG)("DOS private segment set to 0x%04x-0x%04x",DOS_PRIVATE_SEGMENT,DOS_PRIVATE_SEGMENT_END-1);
00127         }
00128 }

00130 Bit16u DOS_GetMemory(Bit16u pages,const char *who) {
00131         if (who == NULL) who = "";
00132         if (dos_memseg == 0) {
00133                 if (DOS_GetMemory_unmapped) E_Exit("DOS:Attempt to use DOS_GetMemory() when private area was unmapped by BOOT");
00134                 if (DOS_PRIVATE_SEGMENT == 0) DOS_GetMemory_Choose();
00135                 dos_memseg = DOS_PRIVATE_SEGMENT;
00136                 if (dos_memseg == 0) E_Exit("DOS:DOS_GetMemory() before private area has been initialized");
00137         }
00138 
00139         if (((Bitu)pages+(Bitu)dos_memseg) > DOS_PRIVATE_SEGMENT_END) {
00140                 LOG(LOG_DOSMISC,LOG_ERROR)("DOS_GetMemory(%u) failed for '%s' (alloc=0x%04x segment=0x%04x end=0x%04x)",
00141                         pages,who,dos_memseg,DOS_PRIVATE_SEGMENT,DOS_PRIVATE_SEGMENT_END);
00142                 E_Exit("DOS:Not enough memory for internal tables");
00143         }
00144         Bit16u page=dos_memseg;
00145         LOG(LOG_DOSMISC,LOG_DEBUG)("DOS_GetMemory(0x%04x pages,\"%s\") = 0x%04x",pages,who,page);
00146 
00147     {
00148         DOS_GetMemLog_Entry ent;
00149 
00150         ent.segbase = page;
00151         ent.pages = pages;
00152         ent.who = who;
00153 
00154         DOS_GetMemLog.push_back(ent);
00155     }
00156 
00157         dos_memseg+=pages;
00158         return page;
