00129 Bitu XMS_EnableA20(bool enable) {
00130         Bit8u val;
00131 
00132     if (IS_PC98_ARCH) {
00133         // NEC PC-98: Unmask (enable) A20 by writing to port 0xF2.
00134         //            Mask (disable) A20 by writing to port 0xF6.
00135         IO_Write(0xF6,enable ? 0x02 : 0x03); /* 0000 001x  x = mask A20 */
00136     }
00137     else {
00138         // IBM PC/AT: Port 0x92, bit 1, set if A20 enabled
00139         val = IO_Read(0x92);
00140         if (enable) IO_Write(0x92,val | 2);
00141         else            IO_Write(0x92,val & ~2);
00142     }
00143 
00144         return 0;
00145 }

00216 Bitu XMS_AllocateMemory(Bitu size, Bit16u& handle) {    // size = kb
00217         /* Find free handle */
00218         Bit16u index=1;
00219         while (!xms_handles[index].free) {
00220                 if (++index>=XMS_HANDLES) return XMS_OUT_OF_HANDLES;
00221         }
00222         MemHandle mem;
00223         if (size!=0) {
00224                 Bitu pages=(size/4) + ((size & 3) ? 1 : 0);
00225                 mem=MEM_AllocatePages(pages,true);
00226                 if (!mem) return XMS_OUT_OF_SPACE;

00234         xms_handles[index].mem=mem;

00251 Bitu XMS_MoveMemory(PhysPt bpt) {
00252         /* Read the block with mem_read's */
00253         Bitu length=mem_readd(bpt+offsetof(XMS_MemMove,length));
00254 
00255     /* "Length must be even" --Microsoft XMS Spec 3.0 */
00256     if (length & 1u) return XMS_INVALID_LENGTH;
00257 
00258         Bitu src_handle=mem_readw(bpt+offsetof(XMS_MemMove,src_handle));
00259         union {
00260                 RealPt realpt;
00261                 Bit32u offset;
00262         } src,dest;
00263         src.offset=mem_readd(bpt+offsetof(XMS_MemMove,src.offset));
00264         Bitu dest_handle=mem_readw(bpt+offsetof(XMS_MemMove,dest_handle));
00265         dest.offset=mem_readd(bpt+offsetof(XMS_MemMove,dest.offset));
00266         PhysPt srcpt,destpt;
00267         if (src_handle) {
00268                 if (InvalidHandle(src_handle)) {
00269                         return XMS_INVALID_SOURCE_HANDLE;
00270                 }
00271                 if (src.offset>=(xms_handles[src_handle].size*1024U)) {
00272                         return XMS_INVALID_SOURCE_OFFSET;
00273                 }
00274                 if (length>xms_handles[src_handle].size*1024U-src.offset) {
00275                         return XMS_INVALID_LENGTH;
00276                 }
00277                 srcpt=((unsigned int)xms_handles[src_handle].mem*4096U)+src.offset;
00278         } else {
00279                 srcpt=Real2Phys(src.realpt);
00280 
00281         /* Microsoft TEST.C considers it an error to allow real mode pointers + length to
00282          * extend past the end of the 8086-accessible conventional memory area. */
00283         if ((srcpt+length) > 0x10FFF0u) return XMS_INVALID_LENGTH;
00284         }
00285         if (dest_handle) {
00286                 if (InvalidHandle(dest_handle)) {
00287                         return XMS_INVALID_DEST_HANDLE;
00288                 }
00289                 if (dest.offset>=(xms_handles[dest_handle].size*1024U)) {
00290                         return XMS_INVALID_DEST_OFFSET;
00291                 }
00292                 if (length>xms_handles[dest_handle].size*1024U-dest.offset) {
00293                         return XMS_INVALID_LENGTH;
00294                 }
00295                 destpt=((unsigned int)xms_handles[dest_handle].mem*4096U)+dest.offset;
00296         } else {
00297                 destpt=Real2Phys(dest.realpt);
00298 
00299         /* Microsoft TEST.C considers it an error to allow real mode pointers + length to
00300          * extend past the end of the 8086-accessible conventional memory area. */
00301         if ((destpt+length) > 0x10FFF0u) return XMS_INVALID_LENGTH;
00302         }
00303 //      LOG_MSG("XMS move src %X dest %X length %X",srcpt,destpt,length);
00304 
00305     /* we must enable the A20 gate during this copy.
00306      * DOSBox-X masks the A20 line and this will only cause corruption otherwise. */
00307 
00308     if (length != 0) {
00309         bool a20_was_enabled = XMS_GetEnabledA20();
00310 
00311         xms_local_enable_count++;
00312         XMS_EnableA20(true);
00313 
00314         mem_memcpy(destpt,srcpt,length);
00315 
00316         xms_local_enable_count--;
00317         if (!a20_was_enabled) XMS_EnableA20(false);
00318     }
00319 
00320     return 0;
00321 }

00410 Bitu XMS_Handler(void) {
00411     Bitu r;
00412 
00413 //      LOG(LOG_MISC,LOG_ERROR)("XMS: CALL %02X",reg_ah);
00414         switch (reg_ah) {
00415         case XMS_GET_VERSION:                                                                           /* 00 */
00416                 reg_ax=XMS_VERSION;
00417                 reg_bx=XMS_DRIVER_VERSION;
00418                 reg_dx=xms_hma_exists?1:0;
00419                 break;
00420         case XMS_ALLOCATE_HIGH_MEMORY:                                                          /* 01 */
00421                 if (xms_hma_exists) {
00422                         if (xms_hma_application_has_control || DOS_IS_IN_HMA()) {
00423                                 /* hma already controlled by application or DOS kernel */
00424                                 reg_ax=0;
00425                                 reg_bl=HIGH_MEMORY_IN_USE;
00426                         }
00427                         else if (reg_dx < xms_hma_minimum_alloc) {
00428                                 /* not big enough */
00429                                 reg_ax=0;
00430                                 reg_bl=HIGH_MEMORY_NOT_BIG_ENOUGH;
00431                         }
00432                         else { /* program allocation */
00433                                 LOG(LOG_MISC,LOG_DEBUG)("XMS: HMA allocated by application/TSR");
00434                                 xms_hma_application_has_control = true;
00435                                 reg_ax=1;
00436                         }
00437                 }
00438                 else {
00439                         reg_ax=0;
00440                         reg_bl=HIGH_MEMORY_NOT_EXIST;
00441                 }
00442                 break;
00443         case XMS_FREE_HIGH_MEMORY:                                                                      /* 02 */
00444                 if (xms_hma_exists) {
00445                         if (DOS_IS_IN_HMA()) LOG(LOG_MISC,LOG_WARN)("DOS application attempted to free HMA while DOS kernel occupies it!");
00446 
00447                         if (xms_hma_application_has_control) {
00448                                 LOG(LOG_MISC,LOG_DEBUG)("XMS: HMA freed by application/TSR");
00449                                 xms_hma_application_has_control = false;
00450                                 reg_ax=1;
00451                         }
00452                         else {
00453                                 reg_ax=0;
00454                                 reg_bl=HIGH_MEMORY_NOT_ALLOCATED;
00455                         }
00456                 }
00457                 else {
00458                         reg_ax=0;
00459                         reg_bl=HIGH_MEMORY_NOT_EXIST;
00460                 }
00461                 break;
00462         case XMS_GLOBAL_ENABLE_A20:                                                                     /* 03 */
00463         /* This appears to be how Microsoft HIMEM.SYS implements this */
00464         if (!xms_global_enable) {
00465             if ((r=XMS_LocalEnableA20()) == 0)
00466                 xms_global_enable = true;
00467         }
00468         else {
00469             r = 0;
00470         }
00471         SET_RESULT(r);
00472                 break;
00473         case XMS_GLOBAL_DISABLE_A20:                                                            /* 04 */
00474         /* This appears to be how Microsoft HIMEM.SYS implements this */
00475         if (xms_global_enable) {
00476             if ((r=XMS_LocalDisableA20()) == 0)
00477                 xms_global_enable = false;
00478         }
00479         else {
00480             r = 0;
00481         }
00482         SET_RESULT(r);
00483         break;
00484     case XMS_LOCAL_ENABLE_A20:                                                                  /* 05 */
00485         SET_RESULT(XMS_LocalEnableA20());
00486         break;
00487         case XMS_LOCAL_DISABLE_A20:                                                                     /* 06 */
00488         SET_RESULT(XMS_LocalDisableA20());
00489                 break;
00490         case XMS_QUERY_A20:                                                                                     /* 07 */
00491                 reg_ax = (Bit16u)XMS_GetEnabledA20();
00492                 reg_bl = 0;
00493                 break;
00494         case XMS_QUERY_FREE_EXTENDED_MEMORY:                                            /* 08 */
00495                 reg_bl = (Bit8u)XMS_QueryFreeMemory(reg_eax,reg_edx);
00496                 if (reg_eax > 65535) reg_eax = 65535; /* cap sizes for older DOS programs. newer ones use function 0x88 */
00497                 if (reg_edx > 65535) reg_edx = 65535;
00498                 break;
00499         case XMS_ALLOCATE_ANY_MEMORY:                                                           /* 89 */
00500                 { /* Chopping off bits 16-31 to fall through to ALLOCATE_EXTENDED_MEMORY is inaccurate.
00501                      The Extended Memory Specification states you use all of EDX, so programs can request
00502                      64MB or more. Even if DOSBox does not (yet) support >= 64MB of RAM. */
00503                 Bit16u handle = 0;
00504                 SET_RESULT(XMS_AllocateMemory(reg_edx,handle));
00505                 reg_dx = handle;
00506                 } break;
00507         case XMS_ALLOCATE_EXTENDED_MEMORY:                                                      /* 09 */
00508                 {
00509                 Bit16u handle = 0;
00510                 SET_RESULT(XMS_AllocateMemory(reg_dx,handle));
00511                 reg_dx = handle;
00512                 } break;
00513         case XMS_FREE_EXTENDED_MEMORY:                                                          /* 0a */
00514                 SET_RESULT(XMS_FreeMemory(reg_dx));
00515                 break;
00516         case XMS_MOVE_EXTENDED_MEMORY_BLOCK:                                            /* 0b */
00517                 SET_RESULT(XMS_MoveMemory(SegPhys(ds)+reg_si),false);
00518                 break;
00519         case XMS_LOCK_EXTENDED_MEMORY_BLOCK: {                                          /* 0c */
00520                 Bit32u address;
00521                 Bitu res = XMS_LockMemory(reg_dx, address);
00522                 if(res) reg_bl = (Bit8u)res;
00523                 reg_ax = (res==0);
00524                 if (res==0) { // success
00525                         reg_bx=(Bit16u)(address & 0xFFFF);
00526                         reg_dx=(Bit16u)(address >> 16);
00527                 }
00528                 } break;
00529         case XMS_UNLOCK_EXTENDED_MEMORY_BLOCK:                                          /* 0d */
00530                 SET_RESULT(XMS_UnlockMemory(reg_dx));
00531                 break;
00532         case XMS_GET_EMB_HANDLE_INFORMATION:                                            /* 0e */
00533                 SET_RESULT(XMS_GetHandleInformation(reg_dx,reg_bh,reg_bl,reg_edx),false);
00534                 reg_edx &= 0xFFFF;
00535                 break;
00536         case XMS_RESIZE_ANY_EXTENDED_MEMORY_BLOCK:                                      /* 0x8f */
00537                 SET_RESULT(XMS_ResizeMemory(reg_dx, reg_ebx));
00538                 break;
00539         case XMS_RESIZE_EXTENDED_MEMORY_BLOCK:                                          /* 0f */
00540                 SET_RESULT(XMS_ResizeMemory(reg_dx, reg_bx));
00541                 break;
00542         case XMS_ALLOCATE_UMB: {                                                                        /* 10 */
00543                 if (!umb_available) {
00544                         reg_ax=0;
00545                         reg_bl=XMS_FUNCTION_NOT_IMPLEMENTED;
00546                         break;
00547                 }
00548                 Bit16u umb_start=dos_infoblock.GetStartOfUMBChain();
00549                 if (umb_start==0xffff) {
00550                         reg_ax=0;

00577                 }

00579         case XMS_DEALLOCATE_UMB:                                                                        /* 11 */
00580                 if (!umb_available) {

00583                         break;
00584                 }
00585                 if (dos_infoblock.GetStartOfUMBChain()!=0xffff) {
00586                         if (DOS_FreeMemory(reg_dx)) {
00587                                 reg_ax=0x0001;
00588                                 break;
00589                         }
00590                 }
00591                 reg_ax=0x0000;
00592                 reg_bl=UMB_NO_BLOCKS_AVAILABLE;
00593                 break;
00594         case XMS_QUERY_ANY_FREE_MEMORY:                                                         /* 88 */
00595                 reg_bl = (Bit8u)XMS_QueryFreeMemory(reg_eax,reg_edx);
00596                 reg_ecx = (Bit32u)((MEM_TotalPages()*MEM_PAGESIZE)-1);                  // highest known physical memory address
00597                 break;
00598         case XMS_GET_EMB_HANDLE_INFORMATION_EXT: {                                      /* 8e */

00600                 Bitu result = XMS_GetHandleInformation(reg_dx,reg_bh,free_handles,reg_edx);
00601                 if (result != 0) reg_bl = (Bit8u)result;

00604                 } break;

00607                 reg_ax=0;
00608                 reg_bl=XMS_FUNCTION_NOT_IMPLEMENTED;
