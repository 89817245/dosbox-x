00280                         /* record old vector, set our new vector */
00281                         old_int2Fh = RealGetVec(0x2f);
00282                         w = CALLBACK_PhysPointer(int2fdbg_hook_callback);
00283                         RealSetVec(0x2f,CALLBACK_RealPointer(int2fdbg_hook_callback));
00284
