00274                 ToReg[ raw ] = reg;

00278         void MakeTables( void ) {
00279                 Bit8u index = 0;
00280                 memset( ToReg, 0xff, sizeof ( ToReg ) );
00281                 memset( ToRaw, 0xff, sizeof ( ToRaw ) );
00282                 //Select the entries that are valid and the index is the mapping to the index entry
00283                 MakeEntry( 0x01, index );                                       //0x01: Waveform select
00284                 MakeEntry( 0x04, index );                                       //104: Four-Operator Enable
00285                 MakeEntry( 0x05, index );                                       //105: OPL3 Mode Enable
00286                 MakeEntry( 0x08, index );                                       //08: CSW / NOTE-SEL
00287                 MakeEntry( 0xbd, index );                                       //BD: Tremolo Depth / Vibrato Depth / Percussion Mode / BD/SD/TT/CY/HH On
00288                 //Add the 32 byte range that hold the 18 operators
00289                 for ( int i = 0 ; i < 24; i++ ) {
00290                         if ( (i & 7) < 6 ) {
00291                                 MakeEntry(0x20 + i, index );            //20-35: Tremolo / Vibrato / Sustain / KSR / Frequency Multiplication Facto
00292                                 MakeEntry(0x40 + i, index );            //40-55: Key Scale Level / Output Level 
00293                                 MakeEntry(0x60 + i, index );            //60-75: Attack Rate / Decay Rate 
00294                                 MakeEntry(0x80 + i, index );            //80-95: Sustain Level / Release Rate
00295                                 MakeEntry(0xe0 + i, index );            //E0-F5: Waveform Select
00296                         }
00297                 }
00298                 //Add the 9 byte range that hold the 9 channels
00299                 for ( int i = 0 ; i < 9; i++ ) {
00300                         MakeEntry(0xa0 + i, index );                    //A0-A8: Frequency Number
00301                         MakeEntry(0xb0 + i, index );                    //B0-B8: Key On / Block Number / F-Number(hi bits) 
00302                         MakeEntry(0xc0 + i, index );                    //C0-C8: FeedBack Modulation Factor / Synthesis Type
00303                 }
00304                 //Store the amount of bytes the table contains
00305                 RawUsed = index;
00306 //              assert( RawUsed <= 127 );
00307                 delay256 = RawUsed;
00308                 delayShift8 = RawUsed+1; 
00309         }

00323         void AddWrite( Bit32u regFull, Bit8u val ) {
00324                 Bit8u regMask = regFull & 0xff;
00325                 /*
00326                         Do some special checks if we're doing opl3 or dualopl2 commands
00327                         Although you could pretty much just stick to always doing opl3 on the player side
00328                 */
00329                 //Enabling opl3 4op modes will make us go into opl3 mode
00330                 if ( header.hardware != HW_OPL3 && regFull == 0x104 && val && (*cache)[0x105] ) {
00331                         header.hardware = HW_OPL3;
00332                 } 
00333                 //Writing a keyon to a 2nd address enables dual opl2 otherwise
00334                 //Maybe also check for rhythm
00335                 if ( header.hardware == HW_OPL2 && regFull >= 0x1b0 && regFull <=0x1b8 && val ) {
00336                         header.hardware = HW_DUALOPL2;
00337                 }
00338                 Bit8u raw = ToRaw[ regMask ];
00339                 if ( raw == 0xff )
00340                         return;
00341                 if ( regFull & 0x100 )
00342                         raw |= 128;
00343                 AddBuf( raw, val );
00344         }

00346                 Bit16u i;
00347         Bit8u val;
00348                 /* Check the registers to add */
00349                 for (i=0;i<256;i++) {
00350                         //Skip the note on entries

00387         bool DoWrite( Bit32u regFull, Bit8u val ) {
00388                 Bit8u regMask = regFull & 0xff;
00389                 //Check the raw index for this register if we actually have to save it
00390                 if ( handle ) {
00391                         /*
00392                                 Check if we actually care for this to be logged, else just ignore it
00393                         */
00394                         Bit8u raw = ToRaw[ regMask ];
00395                         if ( raw == 0xff ) {
00396                                 return true;
00397                         }
00398                         /* Check if this command will not just replace the same value 
00399                            in a reg that doesn't do anything with it
00400                         */
00401                         if ( (*cache)[ regFull ] == val )
00402                                 return true;
00403                         /* Check how much time has passed */
00404                         Bitu passed = PIC_Ticks - lastTicks;
00405                         lastTicks = (Bit32u)PIC_Ticks;
00406                         header.milliseconds += (Bit32u)passed;
00407 
00408                         //if ( passed > 0 ) LOG_MSG( "Delay %d", passed ) ;
00409                         
00410                         // If we passed more than 30 seconds since the last command, we'll restart the the capture
00411                         if ( passed > 30000 ) {
00412                                 CloseFile();
00413                                 goto skipWrite; 
00414                         }
00415                         while (passed > 0) {
00416                                 if (passed < 257) {                     //1-256 millisecond delay
00417                                         AddBuf( delay256, (Bit8u)(passed - 1));
00418                                         passed = 0;
00419                                 } else {
00420                                         Bitu shift = (passed >> 8);
00421                                         passed -= shift << 8;
00422                                         AddBuf( delayShift8, (Bit8u)(shift - 1));
00423                                 }
00424                         }
00425                         AddWrite( regFull, val );
00426                         return true;
00427                 }
00428 skipWrite:
00429                 //Not yet capturing to a file here
00430                 //Check for commands that would start capturing, if it's not one of them return
00431                 if ( !(
00432                         //note on in any channel 
00433                         ( regMask>=0xb0 && regMask<=0xb8 && (val&0x020) ) ||
00434                         //Percussion mode enabled and a note on in any percussion instrument
00435                         ( regMask == 0xbd && ( (val&0x3f) > 0x20 ) )
00436                 )) {
00437                         return true;
00438                 }
00439                 handle = OpenCaptureFile("Raw Opl",".dro");
00440                 if (!handle)
00441                         return false;
00442                 InitHeader();
00443                 //Prepare space at start of the file for the header
00444                 fwrite( &header, 1, sizeof(header), handle );
00445                 /* write the Raw To Reg table */
00446                 fwrite( &ToReg, 1, RawUsed, handle );
00447                 /* Write the cache of last commands */
00448                 WriteCache( );
00449                 /* Write the command that triggered this */
00450                 AddWrite( regFull, val );
00451                 //Init the timing information for the next commands
00452                 lastTicks = (Bit32u)PIC_Ticks;  
00453                 startTicks = (Bit32u)PIC_Ticks;
00454                 return true;
00455         }

00458                 handle = 0;
00459                 bufUsed = 0;
00460                 MakeTables();
00461         }

00463                 CloseFile();
00464         }

00677                                         ctrl.index = val & 0xff;
00678                                         break;
00679                                 }
00680                         }
00681                         //Fall-through if not handled by control chip
00682                 case MODE_OPL3:
00683                         reg.normal = handler->WriteAddr( (Bit32u)port, (Bit8u)val ) & 0x1ff;
00684                         break;
00685                 case MODE_DUALOPL2:
00686                         //Not a 0x?88 port, when write to a specific side
00687                         if ( !(port & 0x8) ) {
00688                                 Bit8u index = ( port & 2 ) >> 1;
00689                                 reg.dual[index] = val & 0xff;
00690                         } else {
00691                                 reg.dual[0] = val & 0xff;
