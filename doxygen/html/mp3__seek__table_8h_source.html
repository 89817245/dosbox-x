00040 // Note: this C++ struct must match (in binary-form) the "drmp3_seek_point" struct
00041 //       defined in dr_mp3.h.  If that changes, then update this to match, along
00042 //       with adjusting the Serialize() template function that union's the values.
00043 //
00044 struct drmp3_seek_point_serial {
00045     drmp3_uint64 seekPosInBytes;      // Points to the first byte of an MP3 frame.
00046     drmp3_uint64 pcmFrameIndex;       // The index of the PCM frame this seek point targets.
00047     drmp3_uint16 mp3FramesToDiscard;  // The number of whole MP3 frames to be discarded before pcmFramesToDiscard.
00048     drmp3_uint16 pcmFramesToDiscard;
00049     template <class T> void Serialize(T& archive) {
00050             archive & seekPosInBytes & pcmFrameIndex & mp3FramesToDiscard & pcmFramesToDiscard;
00051     }
00052 };
