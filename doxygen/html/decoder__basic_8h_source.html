00131 static bool MakeCodePage(Bitu lin_addr,CodePageHandlerDynRec * &cph) {
00132         Bit8u rdval;
00133         //Ensure page contains memory:
00134         if (GCC_UNLIKELY(mem_readb_checked((PhysPt)lin_addr,&rdval))) return true;
00135 
00136         PageHandler * handler=get_tlb_readhandler((PhysPt)lin_addr);
00137         if (handler->flags & PFLAG_HASCODE) {
00138                 // this is a codepage handler, and the one that we're looking for
00139                 cph=(CodePageHandlerDynRec *)handler;
00140                 return false;
00141         }
00142         if (handler->flags & PFLAG_NOCODE) {
00143                 if (false) { // PAGING_ForcePageInit(lin_addr)) {
00144                         handler=get_tlb_readhandler((PhysPt)lin_addr);
00145                         if (handler->flags & PFLAG_HASCODE) {
00146                                 cph=(CodePageHandlerDynRec *)handler;
00147                                 return false;
00148                         }
00149                 }
00150                 if (handler->flags & PFLAG_NOCODE) {
00151                         LOG_MSG("DYNREC:Can't run code in this page");
00152                         cph=0;
00153                         return false;
00154                 }
00155         } 
00156         Bitu lin_page=lin_addr>>12;
00157         Bitu phys_page=lin_page;
00158         // find the physical page that the linear page is mapped to
00159         if (!PAGING_MakePhysPage(phys_page)) {
00160                 LOG_MSG("DYNREC:Can't find physpage");
00161                 cph=0;
00162                 return false;
00163         }
00164         // find a free CodePage
00165         if (!cache.free_pages) {
00166                 if (cache.used_pages!=decode.page.code) cache.used_pages->ClearRelease();
00167                 else {
00168                         // try another page to avoid clearing our source-crosspage
00169                         if ((cache.used_pages->next) && (cache.used_pages->next!=decode.page.code))
00170                                 cache.used_pages->next->ClearRelease();
00171                         else {
00172                                 LOG_MSG("DYNREC:Invalid cache links");
00173                                 cache.used_pages->ClearRelease();
00174                         }
00175                 }
00176         }
00177         CodePageHandlerDynRec * cpagehandler=cache.free_pages;
00178         cache.free_pages=cache.free_pages->next;
00179 
00180         // adjust previous and next page pointer
00181         cpagehandler->prev=cache.last_page;
00182         cpagehandler->next=0;
00183         if (cache.last_page) cache.last_page->next=cpagehandler;
00184         cache.last_page=cpagehandler;
00185         if (!cache.used_pages) cache.used_pages=cpagehandler;
00186 
00187         // initialize the code page handler and add the handler to the memory page
00188         cpagehandler->SetupAt(phys_page,handler);
00189         MEM_SetPageHandler(phys_page,1,cpagehandler);
00190         PAGING_UnlinkPages(lin_page,1);
00191         cph=cpagehandler;
00192         return false;
00193 }

00195 static void decode_advancepage(void) {
00196         // Advance to the next page
00197         decode.active_block->page.end=4095;
00198         // trigger possible page fault here
00199         decode.page.first++;
00200         Bitu faddr=decode.page.first << 12;
00201         mem_readb((PhysPt)faddr);
00202         MakeCodePage(faddr,decode.page.code);
00203         CacheBlockDynRec * newblock=cache_getblock();
00204         decode.active_block->crossblock=newblock;
00205         newblock->crossblock=decode.active_block;
00206         decode.active_block=newblock;
00207         decode.active_block->page.start=0;
00208         decode.page.code->AddCrossBlock(decode.active_block);
00209         decode.page.wmap=decode.page.code->write_map;
00210         decode.page.invmap=decode.page.code->invalidation_map;
00211         decode.page.index=0;
00212 }

00252 // adjust writemap mask to care for map holes due to special
00253 // codefetch functions
00254 static void INLINE decode_increase_wmapmask(Bitu size) {
00255         Bitu mapidx;
00256         CacheBlockDynRec* activecb=decode.active_block; 
00257         if (GCC_UNLIKELY(!activecb->cache.wmapmask)) {
00258                 // no mask memory yet allocated, start with a small buffer
00259                 activecb->cache.wmapmask=(Bit8u*)malloc(START_WMMEM);
00260                 memset(activecb->cache.wmapmask,0,START_WMMEM);
00261                 activecb->cache.maskstart=(Bit16u)decode.page.index;    // start of buffer is current code position
00262                 activecb->cache.masklen=START_WMMEM;
00263                 mapidx=0;
00264         } else {
00265                 mapidx=decode.page.index-activecb->cache.maskstart;
00266                 if (GCC_UNLIKELY(mapidx+size>=activecb->cache.masklen)) {
00267                         // mask buffer too small, increase
00268                         Bitu newmasklen=activecb->cache.masklen*(Bitu)4;
00269                         if (newmasklen<mapidx+size) newmasklen=((mapidx+size)&~3)*2;
00270                         Bit8u* tempmem=(Bit8u*)malloc(newmasklen);
00271                         memset(tempmem,0,newmasklen);
00272                         memcpy(tempmem,activecb->cache.wmapmask,activecb->cache.masklen);
00273                         free(activecb->cache.wmapmask);
00274                         activecb->cache.wmapmask=tempmem;
00275                         activecb->cache.masklen=(Bit16u)newmasklen;
00276                 }
00277         }
00278         // update mask entries
00279         switch (size) {
00280                 case 1 : activecb->cache.wmapmask[mapidx]+=0x01; break;
00281                 case 2 : (*(Bit16u*)&activecb->cache.wmapmask[mapidx])+=0x0101; break;
00282                 case 4 : (*(Bit32u*)&activecb->cache.wmapmask[mapidx])+=0x01010101; break;
00283         }
00284 }

00289         if (GCC_UNLIKELY(decode.page.index>=4096)) {

00294                 if (decode.page.invmap[decode.page.index] == 0) {
00295                         // position not yet modified

00301                 if (tlb_addr) {
00302                         val=(Bitu)(tlb_addr+decode.code);
00303                         decode_increase_wmapmask(1);

00305                         decode.page.index++;
00306                         return true;
00307                 }
00308         }
00309         // first time decoding or not directly accessible, just fetch the value
00310         val=(Bit32u)decode_fetchb();
