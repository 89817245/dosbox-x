01133 bool CommandLine::GetOpt(std::string &name) {
01134         name.clear();
01135 
01136         /* if we're still doing GNU getopt single-char switches, then parse another and return */
01137         if (GetOptGNUSingleCharCheck(name))
01138                 return true;
01139 
01140         while (opt_scan != cmds.end()) {
01141                 std::string &argv = *opt_scan;
01142                 const char *str = argv.c_str();
01143 
01144                 if ((opt_style == CommandLine::either || opt_style == CommandLine::dos) && *str == '/') {
01145                         /* MS-DOS style /option. Example: /A /OPT /HAX /BLAH */
01146                         name = str+1; /* copy to caller minus leaking slash, then erase/skip */
01147                         if (opt_eat_argv) opt_scan = cmds.erase(opt_scan);
01148                         else opt_scan++;
01149                         return true;
01150                 }
01151                 else if ((opt_style == CommandLine::either || opt_style == CommandLine::gnu || opt_style == CommandLine::gnu_getopt) && *str == '-') {
01152                         str++; /* step past '-' */
01153                         if (str[0] == '-' && str[1] == 0) { /* '--' means to stop parsing */
01154                                 opt_scan = cmds.end();
01155                                 if (opt_eat_argv) opt_scan = cmds.erase(opt_scan);
01156                                 break;
01157                         }
01158 
01159                         if (opt_style == CommandLine::gnu_getopt) {
01160                                 /* --switch => "switch"
01161                                  * -switch => -s -w -i -t -c -h */
01162                                 if (*str == '-') {
01163                                         str++;
01164                                 }
01165                                 else {
01166                                         /* assign to single-char parse then eat the argv */
01167                                         opt_gnu_getopt_singlechar = str;
01168                                         if (opt_eat_argv) opt_scan = cmds.erase(opt_scan);
01169                                         else opt_scan++;
01170 
01171                                         /* if we parse a single-char switch, great */
01172                                         if (GetOptGNUSingleCharCheck(name))
01173                                                 return true;
01174 
01175                                         continue; /* if we're here then there was nothing to parse, continue */
01176                                 }
01177                         }
01178                         else {
01179                                 /* -switch and --switch mean the same thing */
01180                                 if (*str == '-') str++;
01181                         }
01182 
01183                         name = str; /* copy to caller, then erase/skip */
01184                         if (opt_eat_argv) opt_scan = cmds.erase(opt_scan);
01185                         else opt_scan++;
01186                         return true;
01187                 }
01188                 else {
01189                         opt_scan++;
01190                 }
01191         }
01192 
01193         return false;
01194 }
