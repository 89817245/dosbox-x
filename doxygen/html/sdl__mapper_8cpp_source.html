01471     static void ProcessInput(Bit16s x, Bit16s y, float deadzone, DOSBox_Vector2& joy)
01472     {
01473         // http://www.third-helix.com/2013/04/12/doing-thumbstick-dead-zones-right.html
01474 
01475         joy = DOSBox_Vector2((x + 0.5f) / 32767.5f, (y + 0.5f) / 32767.5f);
01476 
01477         float m = joy.magnitude();
01478         DOSBox_Vector2 n = joy.normalized();
01479         joy = m < deadzone ? DOSBox_Vector2() : n * ((m - deadzone) / (1.0f - deadzone));
01480 
01481         DOSBox_Vector2 min = DOSBox_Vector2(-1.0f, -1.0f);
01482         DOSBox_Vector2 max = DOSBox_Vector2(+1.0f, +1.0f);
01483         joy = joy.clamp(min, max);

03144 static void CreateStringBind(char * line,bool loading=false) {
03145     std::string o_line = line;
03146 
03147     line=trim(line);
03148     if (*line == 0) return;
03149     char * eventname=StripWord(line);
03150     CEvent * event;
03151     for (CEventVector_it ev_it=events.begin();ev_it!=events.end();ev_it++) {
03152         if (!strcasecmp((*ev_it)->GetName(),eventname)) {
03153             event=*ev_it;
03154             goto foundevent;
03155         }
03156     }
03157 
03158     if (loading) {
03159         /* NTS: StripWord() updates line pointer after ASCIIZ snipping the event name */
03160         pending_string_binds[eventname] = line; /* perhaps code will register it later (i.e. Herc pal change) */
03161         LOG(LOG_MISC,LOG_WARN)("Can't find matching event for %s = %s yet. It may exist later when registered elsewhere in this emulator.",eventname,line);
03162     }
03163     else {
03164         LOG(LOG_MISC,LOG_WARN)("Can't find matching event for %s",eventname);
03165     }
03166 
03167     return ;
03168 foundevent:
03169     CBind * bind;
03170     for (char * bindline=StripWord(line);*bindline;bindline=StripWord(line)) {
03171         for (CBindGroup_it it=bindgroups.begin();it!=bindgroups.end();it++) {
03172             bind=(*it)->CreateConfigBind(bindline);
03173             if (bind) {
03174                 event->AddBind(bind);
03175                 bind->SetFlags(bindline);
03176                 event->update_menu_shortcut();
03177                 break;
03178             }
03179         }
03180     }
03181 }
