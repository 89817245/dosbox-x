02437 static void CreateStringBind(char * line,bool loading=false) {
02438         line=trim(line);
02439         char * eventname=StripWord(line);
02440         CEvent * event;
02441         for (CEventVector_it ev_it=events.begin();ev_it!=events.end();ev_it++) {
02442                 if (!strcasecmp((*ev_it)->GetName(),eventname)) {
02443                         event=*ev_it;
02444                         goto foundevent;
02445                 }
02446         }
02447 
02448     if (loading) {
02449         /* NTS: StripWord() updates line pointer after ASCIIZ snipping the event name */
02450         pending_string_binds[eventname] = line; /* perhaps code will register it later (i.e. Herc pal change) */
02451         LOG(LOG_MISC,LOG_WARN)("Can't find matching event for %s = %s yet. It may exist later when registered elsewhere in this emulator.",eventname,line);
02452     }
02453     else {
02454         LOG(LOG_MISC,LOG_WARN)("Can't find matching event for %s",eventname);
02455     }
02456 
02457         return ;
02458 foundevent:
02459         CBind * bind;
02460         for (char * bindline=StripWord(line);*bindline;bindline=StripWord(line)) {
02461                 for (CBindGroup_it it=bindgroups.begin();it!=bindgroups.end();it++) {
02462                         bind=(*it)->CreateConfigBind(bindline);
02463                         if (bind) {
02464                                 event->AddBind(bind);
02465                                 bind->SetFlags(bindline);
02466                                 break;

02895 static void InitializeJoysticks(void) {

02903                         // depending on the number of physically attached joysticks
02904                         if (mapper.sticks.num>1) {
02905                                 // more than one joystick present; if all are acceptable use 2axis
02906                                 // to allow emulation of two joysticks
02907                                 bool first_usable=false;
02908                                 SDL_Joystick* tmp_stick1=SDL_JoystickOpen(0);
02909                                 if (tmp_stick1) {

02916                                 SDL_Joystick* tmp_stick2=SDL_JoystickOpen(1);
02917                                 if (tmp_stick2) {
02918                                         if ((SDL_JoystickNumAxes(tmp_stick2)>1) || (SDL_JoystickNumButtons(tmp_stick2)>0)) {
02919                                                 second_usable=true;
02920                                         }
02921                                         SDL_JoystickClose(tmp_stick2);
02922                                 }
02923                                 // choose joystick type now that we know which physical joysticks are usable
02924                                 if (first_usable) {
02925                                         if (second_usable) {
02926                                                 joytype=JOY_2AXIS;

02928                                         } else {
02929                                                 joytype=JOY_4AXIS;

02931                                         }
02932                                 } else if (second_usable) {
02933                                         joytype=JOY_4AXIS_2;
02934                                         LOG_MSG("One joystick reported, initializing with 4axis_2");
02935                                 }
