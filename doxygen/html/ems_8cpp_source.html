00752         MemHandle src_handle = 0,dest_handle = 0;

00754         if (!region.src_type) {
00755                 src_mem=region.src_page_seg*16+region.src_offset;
00756         } else {

00796                         } else {
00797                                 if (toread<dest_remain) {
00798                                         MEM_BlockRead((dest_handle*MEM_PAGE_SIZE)+dest_off,buf_dest,toread);
00799                                 } else {
00800                                         MEM_BlockRead((dest_handle*MEM_PAGE_SIZE)+dest_off,buf_dest,dest_remain);
00801                                         MEM_BlockRead((MEM_NextHandle(dest_handle)*MEM_PAGE_SIZE),&buf_dest[dest_remain],toread-dest_remain);
00802                                 }
00803                         }
00804                         /* Write to the source */
00805                         if (!region.src_type) {
00806                                 MEM_BlockWrite(src_mem,buf_dest,toread);
00807                         } else {
00808                                 if (toread<src_remain) {
00809                                         MEM_BlockWrite((src_handle*MEM_PAGE_SIZE)+src_off,buf_dest,toread);
00810                                 } else {
00811                                         MEM_BlockWrite((src_handle*MEM_PAGE_SIZE)+src_off,buf_dest,src_remain);
00812                                         MEM_BlockWrite((MEM_NextHandle(src_handle)*MEM_PAGE_SIZE),&buf_dest[src_remain],toread-src_remain);
00813                                 }
00814                         }
00815                 }

00818                         MEM_BlockWrite(dest_mem,buf_src,toread);
00819                 } else {
00820                         if (toread<dest_remain) {

00822                         } else {
00823                                 MEM_BlockWrite((dest_handle*MEM_PAGE_SIZE)+dest_off,buf_src,dest_remain);
00824                                 MEM_BlockWrite((MEM_NextHandle(dest_handle)*MEM_PAGE_SIZE),&buf_src[dest_remain],toread-dest_remain);
00825                         }

01236 static Bitu V86_Monitor() {
01237         /* Calculate which interrupt did occur */
01238         Bitu int_num=(mem_readw(SegPhys(ss)+(reg_esp & cpu.stack.mask))-0x2803);
01239 
01240         /* See if Exception 0x0d and not Interrupt 0x0d */
01241         if ((int_num==(0x0d*4)) && ((reg_sp&0xffff)!=0x1fda)) {
01242                 /* Protection violation during V86-execution,
01243                    needs intervention by monitor (depends on faulting opcode) */
01244 
01245                 reg_esp+=6;             // skip ip of CALL and error code of EXCEPTION 0x0d
01246 
01247                 /* Get address of faulting instruction */
01248                 Bit16u v86_cs=mem_readw(SegPhys(ss)+((reg_esp+4) & cpu.stack.mask));
01249                 Bit16u v86_ip=mem_readw(SegPhys(ss)+((reg_esp+0) & cpu.stack.mask));
01250                 Bit8u v86_opcode=mem_readb((v86_cs<<4)+v86_ip);
01251 //              LOG_MSG("v86 monitor caught protection violation at %x:%x, opcode=%x",v86_cs,v86_ip,v86_opcode);
01252                 switch (v86_opcode) {
01253                         case 0x0f:              // double byte opcode
01254                                 v86_opcode=mem_readb((v86_cs<<4)+v86_ip+1);
01255                                 switch (v86_opcode) {
01256                                         case 0x20: {    // mov reg,CRx
01257                                                 Bitu rm_val=mem_readb((v86_cs<<4)+v86_ip+2);
01258                                                 Bitu which=(rm_val >> 3) & 7;
01259                                                 if ((rm_val<0xc0) || (rm_val>=0xe8))
01260                                                         E_Exit("Invalid opcode 0x0f 0x20 %x caused a protection fault!",static_cast<unsigned int>(rm_val));
01261                                                 Bit32u crx=CPU_GET_CRX(which);
01262                                                 switch (rm_val&7) {
01263                                                         case 0: reg_eax=crx;    break;
01264                                                         case 1: reg_ecx=crx;    break;
01265                                                         case 2: reg_edx=crx;    break;
01266                                                         case 3: reg_ebx=crx;    break;
01267                                                         case 4: reg_esp=crx;    break;
01268                                                         case 5: reg_ebp=crx;    break;
01269                                                         case 6: reg_esi=crx;    break;
01270                                                         case 7: reg_edi=crx;    break;
01271                                                 }
01272                                                 mem_writew(SegPhys(ss)+((reg_esp+0) & cpu.stack.mask),v86_ip+3);
01273                                                 }
01274                                                 break;
01275                                         case 0x22: {    // mov CRx,reg
01276                                                 Bitu rm_val=mem_readb((v86_cs<<4)+v86_ip+2);
01277                                                 Bitu which=(rm_val >> 3) & 7;
01278                                                 if ((rm_val<0xc0) || (rm_val>=0xe8))
01279                                                         E_Exit("Invalid opcode 0x0f 0x22 %x caused a protection fault!",static_cast<unsigned int>(rm_val));
01280                                                 Bit32u crx=0;
01281                                                 switch (rm_val&7) {
01282                                                         case 0: crx=reg_eax;    break;
01283                                                         case 1: crx=reg_ecx;    break;
01284                                                         case 2: crx=reg_edx;    break;
01285                                                         case 3: crx=reg_ebx;    break;
01286                                                         case 4: crx=reg_esp;    break;
01287                                                         case 5: crx=reg_ebp;    break;
01288                                                         case 6: crx=reg_esi;    break;
01289                                                         case 7: crx=reg_edi;    break;
01290                                                 }
01291                                                 if (which==0) crx|=1;   // protection bit always on
01292                                                 CPU_SET_CRX(which,crx);
01293                                                 mem_writew(SegPhys(ss)+((reg_esp+0) & cpu.stack.mask),v86_ip+3);
01294                                                 }
01295                                                 break;
01296                                         default:
01297                                                 E_Exit("Unhandled opcode 0x0f %x caused a protection fault!",v86_opcode);
01298                                 }
01299                                 break;
01300                         case 0xe4:              // IN AL,Ib
01301                                 if (!VCPI_trapio_r(mem_readb((v86_cs<<4)+v86_ip+1),1))
01302                                         reg_al=(Bit8u)(IO_ReadB(mem_readb((v86_cs<<4)+v86_ip+1))&0xff);
01303                                 mem_writew(SegPhys(ss)+((reg_esp+0) & cpu.stack.mask),v86_ip+2);
01304                                 break;
01305                         case 0xe5:              // IN AX,Ib
01306                                 if (!VCPI_trapio_r(mem_readb((v86_cs<<4)+v86_ip+1),2))
01307                                         reg_ax=(Bit16u)(IO_ReadW(mem_readb((v86_cs<<4)+v86_ip+1))&0xffff);
01308                                 mem_writew(SegPhys(ss)+((reg_esp+0) & cpu.stack.mask),v86_ip+2);
01309                                 break;
01310                         case 0xe6:              // OUT Ib,AL
01311                                 if (!VCPI_trapio_w(mem_readb((v86_cs<<4)+v86_ip+1),reg_al,1))
01312                                         IO_WriteB(mem_readb((v86_cs<<4)+v86_ip+1),reg_al);
01313                                 mem_writew(SegPhys(ss)+((reg_esp+0) & cpu.stack.mask),v86_ip+2);
01314                                 break;
01315                         case 0xe7:              // OUT Ib,AX
01316                                 if (!VCPI_trapio_w(mem_readb((v86_cs<<4)+v86_ip+1),reg_ax,2))
01317                                         IO_WriteW(mem_readb((v86_cs<<4)+v86_ip+1),reg_ax);
01318                                 mem_writew(SegPhys(ss)+((reg_esp+0) & cpu.stack.mask),v86_ip+2);
01319                                 break;
01320                         case 0xec:              // IN AL,DX
01321                                 if (!VCPI_trapio_r(reg_dx,1))
01322                                         reg_al=(Bit8u)(IO_ReadB(reg_dx)&0xff);
01323                                 mem_writew(SegPhys(ss)+((reg_esp+0) & cpu.stack.mask),v86_ip+1);
01324                                 break;
01325                         case 0xed:              // IN AX,DX
01326                                 if (!VCPI_trapio_r(reg_dx,2))
01327                                         reg_ax=(Bit16u)(IO_ReadW(reg_dx)&0xffff);
01328                                 mem_writew(SegPhys(ss)+((reg_esp+0) & cpu.stack.mask),v86_ip+1);
01329                                 break;
01330                         case 0xee:              // OUT DX,AL
01331                                 if (!VCPI_trapio_w(reg_dx,reg_al,1))
01332                                         IO_WriteB(reg_dx,reg_al);
01333                                 mem_writew(SegPhys(ss)+((reg_esp+0) & cpu.stack.mask),v86_ip+1);
01334                                 break;
01335                         case 0xef:              // OUT DX,AX
01336                                 if (!VCPI_trapio_w(reg_dx,reg_ax,2))
01337                                         IO_WriteW(reg_dx,reg_ax);
01338                                 mem_writew(SegPhys(ss)+((reg_esp+0) & cpu.stack.mask),v86_ip+1);
01339                                 break;
01340                         case 0xf0:              // LOCK prefix
01341                                 mem_writew(SegPhys(ss)+((reg_esp+0) & cpu.stack.mask),v86_ip+1);
01342                                 break;
01343                         case 0xf4:              // HLT
01344                                 reg_flags|=FLAG_IF;
01345                                 CPU_HLT(reg_eip);
01346                                 mem_writew(SegPhys(ss)+((reg_esp+0) & cpu.stack.mask),v86_ip+1);
01347                                 break;
01348                         default:
01349                                 E_Exit("Unhandled opcode %x caused a protection fault!",v86_opcode);
01350                 }
01351                 return CBRET_NONE;
01352         }
01353 
01354         /* Get address to interrupt handler */
01355         Bit16u vint_vector_seg=mem_readw(SegValue(ds)+int_num+2);
01356         Bit16u vint_vector_ofs=mem_readw(int_num);
01357         if (reg_sp!=0x1fda) reg_esp+=(2+3*4);   // Interrupt from within protected mode
01358         else reg_esp+=2;
01359 
01360         /* Read entries that were pushed onto the stack by the interrupt */
01361         Bit16u return_ip=mem_readw(SegPhys(ss)+(reg_esp & cpu.stack.mask));
01362         Bit16u return_cs=mem_readw(SegPhys(ss)+((reg_esp+4) & cpu.stack.mask));
01363         Bit32u return_eflags=mem_readd(SegPhys(ss)+((reg_esp+8) & cpu.stack.mask));
01364 
01365         /* Modify stack to call v86-interrupt handler */
01366         mem_writed(SegPhys(ss)+(reg_esp & cpu.stack.mask),vint_vector_ofs);
01367         mem_writed(SegPhys(ss)+((reg_esp+4) & cpu.stack.mask),vint_vector_seg);
01368         mem_writed(SegPhys(ss)+((reg_esp+8) & cpu.stack.mask),return_eflags&(~(FLAG_IF|FLAG_TF)));
01369 
01370         /* Adjust SP of v86-stack */
01371         Bit16u v86_ss=mem_readw(SegPhys(ss)+((reg_esp+0x10) & cpu.stack.mask));
01372         Bit16u v86_sp=mem_readw(SegPhys(ss)+((reg_esp+0x0c) & cpu.stack.mask))-6;
01373         mem_writew(SegPhys(ss)+((reg_esp+0x0c) & cpu.stack.mask),v86_sp);
01374 
01375         /* Return to original code after v86-interrupt handler */
01376         mem_writew((v86_ss<<4)+v86_sp+0,return_ip);
01377         mem_writew((v86_ss<<4)+v86_sp+2,return_cs);
01378         mem_writew((v86_ss<<4)+v86_sp+4,(Bit16u)(return_eflags&0xffff));
01379         return CBRET_NONE;
01380 }

01391 static void SetupVCPI() {
01392         /* The EMM OS handle is often located just above the 1MB boundary.
01393          * And we're about to write that area directly. So for obvious
01394          * reasons we should enable the A20 gate now. This fixes random
01395          * crashes in v86 mode when a20=mask as opposed to a20=fast. */
01396         if ((emm_handles[vcpi.ems_handle].mem<<12) & (1<<20)) {
01397                 LOG(LOG_MISC,LOG_DEBUG)("EMS:EMM OS handle is associated with memory on an odd megabyte. Enabling A20 gate to avoid corrupting DOS state");
01398                 XMS_EnableA20(true);
01399         }
01400 
01401         vcpi.enabled=false;
01402 
01403         vcpi.ems_handle=0;      // use EMM system handle for VCPI data
01404 
01405         vcpi.enabled=true;
01406 
01407         vcpi.pic1_remapping=0x08;       // master PIC base
01408         vcpi.pic2_remapping=0x70;       // slave PIC base
01409 
01410         vcpi.private_area=emm_handles[vcpi.ems_handle].mem<<12;
01411 
01412         /* GDT */
01413         mem_writed(vcpi.private_area+0x0000,0x00000000);        // descriptor 0
01414         mem_writed(vcpi.private_area+0x0004,0x00000000);        // descriptor 0
01415 
01416         Bit32u ldt_address=(vcpi.private_area+0x1000);
01417         Bit16u ldt_limit=0xff;
