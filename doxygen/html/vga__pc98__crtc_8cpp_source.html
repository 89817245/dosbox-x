00093         case 0x00:      // 0x70: Character row, CG start scanline
00094             pc98_text_first_row_scanline_start = (unsigned char)val & 0x1F;

00099         case 0x04:      // 0x74: Character row, number of CG scanlines to display

00101             break;
00102         case 0x06:
00103             pc98_text_row_scroll_lines = (unsigned char)val & 0x1F;
00104             break;
00105         case 0x08:
00106             pc98_text_row_scroll_count_start = (unsigned char)val & 0x1F;
00107             break;
00108         case 0x0A:
00109             pc98_text_row_scroll_num_lines = (unsigned char)val & 0x1F;
00110             break;
00111         case 0x0C:      // 0x7C: mode reg / vram operation mode (also, reset tile counter)
00112             if (enable_pc98_grcg) {
00113                 pc98_gdc_tile_counter = 0;
00114                 pc98_gdc_modereg = (uint8_t)val;
00115                 /* bit 7: 1=GRGC active  0=GRGC invalid
00116                  * bit 6: 1=Read/Modify/Write when writing  0=TCR mode at read, TDW mode at write */
00117                 pc98_gdc_vramop &= ~(3 << VOPBIT_GRCG);
00118                 pc98_gdc_vramop |= (val & 0xC0) >> (6 - VOPBIT_GRCG);
00119             }
00120             break;
00121         case 0x0E:      // 0x7E: tile data
00122             if (enable_pc98_grcg) {
00123                 pc98_gdc_tiles[pc98_gdc_tile_counter].b[0] = (uint8_t)val;
00124                 pc98_gdc_tiles[pc98_gdc_tile_counter].b[1] = (uint8_t)val;
00125                 pc98_gdc_tile_counter = (pc98_gdc_tile_counter + 1) & 3;
00126             }
00127             break;
00128         default:
00129             LOG_MSG("PC98 CRTC w: port=0x%02X val=0x%02X unknown",(unsigned int)port,(unsigned int)val);
00130             break;
00131     }

00160 void pc98_port6A_command_write(unsigned char b) {
00161     switch (b) {
00162         case 0x00: // 16-color (analog) disable
00163             gdc_analog = false;
00164             pc98_gdc_vramop &= ~(1 << VOPBIT_ANALOG);
00165             VGA_SetupHandlers();   // confirmed on real hardware: this disables access to E000:0000
00166             pc98_update_palette(); // Testing on real hardware shows that the "digital" and "analog" palettes are completely different.
00167                                    // They're both there in hardware, but one or another is active depending on analog enable.
00168                                    // Also, the 4th bitplane at E000:0000 disappears when switched off from the display and from CPU access.
00169             break;
00170         case 0x01: // or enable
00171             if (enable_pc98_16color) {
00172                 gdc_analog = true;
00173                 pc98_gdc_vramop |= (1 << VOPBIT_ANALOG);
00174                 VGA_SetupHandlers();   // confirmed on real hardware: this enables access to E000:0000
00175                 pc98_update_palette(); // Testing on real hardware shows that the "digital" and "analog" palettes are completely different.
00176                                        // They're both there in hardware, but one or another is active depending on analog enable.
00177                                        // Also, the 4th bitplane at E000:0000 disappears when switched off from the display and from CPU access.
00178             }
00179             break;
00180         case 0x04:
00181             if (egc_enable_enable)
00182                 pc98_gdc_vramop &= ~(1 << VOPBIT_EGC);
00183             break;
00184         case 0x05:
00185             if (enable_pc98_egc && egc_enable_enable)
00186                 pc98_gdc_vramop |= (1 << VOPBIT_EGC);
00187             break;
00188         case 0x06:
00189             egc_enable_enable = false;
00190             break;
00191         case 0x07:
00192             egc_enable_enable = true;
00193             break;
00194         case 0x0A: // TODO
00195         case 0x0B: // TODO
00196             // TODO
00197             break;
00198         case 0x20: // 256-color mode disable
00199             if (enable_pc98_egc && egc_enable_enable) {
00200                 pc98_gdc_vramop &= ~(1 << VOPBIT_VGA);
00201                 VGA_SetupHandlers(); // memory mapping presented to the CPU changes
00202                 pc98_update_palette();
00203                 pc98_update_page_ptrs();
00204             }
00205             break;
00206         case 0x21: // 256-color mode enable
00207             if (enable_pc98_egc && egc_enable_enable && enable_pc98_256color) {
00208                 pc98_gdc_vramop |= (1 << VOPBIT_VGA);
00209                 VGA_SetupHandlers(); // memory mapping presented to the CPU changes
00210                 pc98_update_palette();
00211                 pc98_update_page_ptrs();
00212             }
00213             break;
00214         case 0x40: // CRT mode
00215         case 0x41: // Plasma/LCD mode
00216             pc98_crt_mode = (b&1)==0;
00217             break;
00218         case 0x68: // 128KB VRAM boundary
00219             // TODO: Any conditions?
00220             pc98_256kb_boundary = false;
00221             VGA_SetupHandlers(); // memory mapping presented to the CPU changes
00222             break;
00223         case 0x69: // 256KB VRAM boundary
00224             // TODO: Any conditions?
00225             pc98_256kb_boundary = true;
00226             VGA_SetupHandlers(); // memory mapping presented to the CPU changes
00227             break;
00228         case 0x82: // GDC Clock #1   0=2.5MHz   1=5MHz
00229         case 0x83:
00230             gdc_clock_1 = !!(b&1);
00231             gdc_clock_check();
00232             break;
00233         case 0x84: // GDC Clock #2   0=2.5MHz   1=5MHz
00234         case 0x85:
00235             gdc_clock_2 = !!(b&1);

00241             break;

00276 /* Port 0x9A0 readback.
00277  * This is needed to report the GDC setting so that Windows 3.1 doesn't change the
00278  * GDC to 5MHz arbitrarily and leave PC-98 games confused afterward */
00279 Bitu pc98_read_9a0(Bitu /*port*/,Bitu /*iolen*/) {
00280     Bitu retval = 0;
00281 
00282     /* bit 0 depends on what was selected by writing to port 9a0.
00283      * according to undocumented 9821 (not verified), unknown registers and 0x00 will return 0xff here. */
00284     switch (sel_9a0) {
00285         case 0x01:      // color/monochrome
00286             if (true) retval |= 1u;//FIXME
00287             break;
00288         case 0x02:      // odd raster mask
00289             if (pc98_graphics_hide_odd_raster_200line) retval |= 1u;
00290             break;
00291         case 0x03:      // display on/off
00292             if (pc98_display_enable) retval |= 1u;
00293             break;
00294         case 0x04:      // palette mode (used by Sim City 2000)
00295             if (pc98_gdc_vramop & (1 << VOPBIT_ANALOG)) retval |= 1u;
00296             break;
00297         default:
00298             retval |= 0xFF;//FIXME: Is this true?
00299             break;
00300     }
00301 
00302     /* bit 1: graphic GDC clock frequency as set in hardware at this moment */
00303     if (gdc_5mhz_mode)
00304         retval |= 0x02;
00305 
00306     return retval;
00307 }
