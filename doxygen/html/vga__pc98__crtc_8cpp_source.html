00037 void pc98_crtc_write(Bitu port,Bitu val,Bitu iolen) {
00038     switch (port&0xE) {
00039         case 0x0C:      // 0x7C: mode reg / vram operation mode (also, reset tile counter)
00040             if (enable_pc98_grcg) {
00041                 pc98_gdc_tile_counter = 0;
00042                 pc98_gdc_modereg = val;
00043                 pc98_gdc_vramop &= ~(3 << VOPBIT_GRCG);
00044                 pc98_gdc_vramop |= (val & 0xC0) >> (6 - VOPBIT_GRCG);
00045             }
00046             break;
00047         case 0x0E:      // 0x7E: tile data
00048             if (enable_pc98_grcg) {
00049                 pc98_gdc_tiles[pc98_gdc_tile_counter].b[0] = val;
00050                 pc98_gdc_tiles[pc98_gdc_tile_counter].b[1] = val;
00051                 pc98_gdc_tile_counter = (pc98_gdc_tile_counter + 1) & 3;
00052             }
00053             break;
00054         default:
00055             LOG_MSG("PC98 CRTC w: port=0x%02X val=0x%02X unknown",(unsigned int)port,(unsigned int)val);
00056             break;
00057     };
