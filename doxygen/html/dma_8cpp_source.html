00075  *
00076  * NTS: Don't forget, from 8237 datasheet: The DMA chip transfers a byte (or word if 16-bit) of data,
00077  *      and THEN increments or decrements the address. So in decrement mode, "address" is still the
00078  *      first byte before decrementing. */

00080         Bit8u * write=(Bit8u *) data;

00084         offset <<= dma16;
00085         Bit32u dma_wrap = (((0xffff<<dma16)+dma16)&DMA16_ADDRMASK) | dma_wrapping;
00086 
00087         if (dma16) {
00088                 /* I'm going to assume by how ISA DMA works that you can't just copy bytes backwards,
00089                  * because 16-bit DMA means one 16-bit WORD transferred per DMA memory cycle on the ISA Bus.
00090                  *
00091                  * I have yet to see a DOS program use this mode of ISA DMA, so this remains unimplemented.
00092                  *
00093                  * Data to transfer from the device:
00094                  *
00095                  * 0x1234 0x5678 0x9ABC 0xDEF0
00096                  *
00097                  * Becomes stored to memory by DMA like this (one 16-bit WORD at a time):
00098                  *
00099                  * 0xDEF0 0x9ABC 0x5678 0x1234
00100                  *
00101                  * it does NOT become:
00102                  *
00103                  * 0xF0DE 0xBC9A 0x7856 0x3412 */
00104                 LOG(LOG_DMACONTROL,LOG_WARN)("16-bit decrementing DMA not implemented");
00105         }
00106         else {
00107                 for ( ; size ; size--, offset--) {
00108                         offset &= dma_wrap;
00109                         Bitu page = highpart_addr_page+(offset >> 12);
00110                         /* care for EMS pageframe etc. */
00111                         if (page < EMM_PAGEFRAME4K) page = paging.firstmb[page];
00112                         else if (page < EMM_PAGEFRAME4K+0x10) page = ems_board_mapping[page];
00113                         else if (page < LINK_START) page = paging.firstmb[page];

00227 void DmaController::WriteControllerReg(Bitu reg,Bitu val,Bitu /*len*/) {
00228         DmaChannel * chan;
00229         switch (reg) {
00230         /* set base address of DMA transfer (1st byte low part, 2nd byte high part) */
00231         case 0x0:case 0x2:case 0x4:case 0x6:
00232                 UpdateEMSMapping();
00233                 chan=GetChannel((Bit8u)(reg >> 1));
00234                 flipflop=!flipflop;
00235                 if (flipflop) {
00236                         chan->baseaddr=(chan->baseaddr&0xff00)|val;
00237                         chan->curraddr=(chan->curraddr&0xff00)|val;
00238                 } else {
00239                         chan->baseaddr=(chan->baseaddr&0x00ff)|(val << 8);
00240                         chan->curraddr=(chan->curraddr&0x00ff)|(val << 8);
00241                 }
00242                 break;
00243         /* set DMA transfer count (1st byte low part, 2nd byte high part) */
00244         case 0x1:case 0x3:case 0x5:case 0x7:
00245                 UpdateEMSMapping();
00246                 chan=GetChannel((Bit8u)(reg >> 1));
00247                 flipflop=!flipflop;
00248                 if (flipflop) {
00249                         chan->basecnt=(chan->basecnt&0xff00)|val;
00250                         chan->currcnt=(chan->currcnt&0xff00)|val;
00251                 } else {
00252                         chan->basecnt=(chan->basecnt&0x00ff)|(val << 8);
00253                         chan->currcnt=(chan->currcnt&0x00ff)|(val << 8);
00254                 }
00255                 break;
00256         case 0x8:               /* Comand reg not used */
00257                 break;
00258         case 0x9:               /* Request registers, memory to memory */
00259                 //TODO Warning?
00260                 break;
00261         case 0xa:               /* Mask Register */
00262                 if ((val & 0x4)==0) UpdateEMSMapping();
00263                 chan=GetChannel(val & 3);
00264                 chan->SetMask((val & 0x4)>0);
00265                 break;
00266         case 0xb:               /* Mode Register */
00267                 UpdateEMSMapping();
00268                 chan=GetChannel(val & 3);
00269                 chan->autoinit=(val & 0x10) > 0;
00270                 chan->increment=(!allow_decrement_mode) || ((val & 0x20) == 0); /* 0=increment 1=decrement */
00271                 //TODO Maybe other bits? Like bits 6-7 to select demand/single/block/cascade mode? */
00272                 break;
00273         case 0xc:               /* Clear Flip/Flip */
00274                 flipflop=false;
00275                 break;
00276         case 0xd:               /* Master Clear/Reset */
00277                 for (Bit8u ct=0;ct<4;ct++) {
00278                         chan=GetChannel(ct);
00279                         chan->SetMask(true);
00280                         chan->tcount=false;
00281                 }
00282                 flipflop=false;
00283                 break;
00284         case 0xe:               /* Clear Mask register */               
00285                 UpdateEMSMapping();
00286                 for (Bit8u ct=0;ct<4;ct++) {
00287                         chan=GetChannel(ct);
00288                         chan->SetMask(false);
00289                 }
00290                 break;
00291         case 0xf:               /* Multiple Mask register */
00292                 UpdateEMSMapping();
00293                 for (Bit8u ct=0;ct<4;ct++) {
00294                         chan=GetChannel(ct);
00295                         chan->SetMask(val & 1);
00296                         val>>=1;
00297                 }
00298                 break;
00299         }

00302 Bitu DmaController::ReadControllerReg(Bitu reg,Bitu /*len*/) {
00303         DmaChannel * chan;Bitu ret;
00304         switch (reg) {
00305         /* read base address of DMA transfer (1st byte low part, 2nd byte high part) */
00306         case 0x0:case 0x2:case 0x4:case 0x6:
00307                 chan=GetChannel((Bit8u)(reg >> 1));
00308                 flipflop=!flipflop;
00309                 if (flipflop) {
00310                         return chan->curraddr & 0xff;
00311                 } else {
00312                         return (chan->curraddr >> 8) & 0xff;
00313                 }
00314         /* read DMA transfer count (1st byte low part, 2nd byte high part) */
00315         case 0x1:case 0x3:case 0x5:case 0x7:
00316                 chan=GetChannel((Bit8u)(reg >> 1));
00317                 flipflop=!flipflop;
00318                 if (flipflop) {
00319                         return chan->currcnt & 0xff;
00320                 } else {
00321                         return (chan->currcnt >> 8) & 0xff;
00322                 }
00323         case 0x8:               /* Status Register */
00324                 ret=0;
00325                 for (Bit8u ct=0;ct<4;ct++) {
00326                         chan=GetChannel(ct);
00327                         if (chan->tcount) ret|=1 << ct;
00328                         chan->tcount=false;
00329                         if (chan->request) ret|=1 << (4+ct);
00330                 }
00331                 return ret;
00332         case 0xc:               /* Clear Flip/Flip (apparently most motherboards will treat read OR write as reset) */
00333                 flipflop=false;
00334                 break;
00335         default:
00336                 LOG(LOG_DMACONTROL,LOG_NORMAL)("Trying to read undefined DMA port %x",(int)reg);
00337                 break;
00338         }
00339         return 0xffffffff;
00340 }
