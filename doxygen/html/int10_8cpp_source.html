01005     bmp_ofs = host_readd((HostPt)(tmp+10)); // bOffBits
01006 
01007     if (fread(tmp,40,1,fp) != 1) goto fail; // BITMAPINFOHEADER
01008     if (host_readd((HostPt)(tmp+0)) != 40) goto fail; // biSize == 40 or else
01009     if (host_readd((HostPt)(tmp+4)) != 2048) goto fail; // biWidth == 2048 or else
01010     if (host_readd((HostPt)(tmp+8)) != 2048) goto fail; // biHeight == 2048 or else
01011     if (host_readw((HostPt)(tmp+12)) != 1) goto fail; // biPlanes == 1 or else
01012     if (host_readw((HostPt)(tmp+14)) != 1) goto fail; // biBitCount == 1 or else
01013     if (host_readd((HostPt)(tmp+16)) != 0) goto fail; // biCompression == 0 or else
01014 
01015     /* first row is 8x16 single width */
01016     fseek(fp,long(bmp_ofs+((2048u-16u)*(2048u/8u))),SEEK_SET); /* arrrgh bitmaps are upside-down */
01017     if (fread(tmp,(2048/8)*16,1,fp) != 1) goto fail;
01018     for (lowbyte=0;lowbyte < 256;lowbyte++) {
01019         for (r=0;r < 16;r++) {
01020             vga.draw.font[(lowbyte*16)+r] = tmp[lowbyte+((15-r/*upside-down!*/)*(2048/8))] ^ 0xFF/*ANEX86 has inverse color scheme*/;
01021         }
01022     }
01023     /* everything after is 16x16 fullwidth.
01024      * note: 2048 / 16 = 128 */
01025     for (hibyte=1;hibyte < 128;hibyte++) {
01026         fseek(fp,long(bmp_ofs+((2048u-(16u*hibyte)-16u)*(2048u/8u))),SEEK_SET); /* arrrgh bitmaps are upside-down */
01027         if (fread(tmp,(2048/8)*16,1,fp) != 1) goto fail;
01028 
01029         for (lowbyte=0;lowbyte < 128;lowbyte++) {
01030             for (r=0;r < 16;r++) {
01031                 unsigned int i;
01032                 unsigned int o;
01033 
01034                 /* NTS: fullwidth is 16x16 128 chars across.
01035                  * each row of the char bitmap is TWO bytes. */
01036                 i = (lowbyte*2)+((15-r/*upside-down!*/)*(2048/8));
01037                 o = ((((hibyte*128)+lowbyte)*16)+r)*2;
01038 
01039                 assert((i+2) <= sizeof(tmp));
01040                 assert((o+2) <= sizeof(vga.draw.font));
01041 
01042                 vga.draw.font[o+0] = tmp[i+0] ^ 0xFF;
01043                 vga.draw.font[o+1] = tmp[i+1] ^ 0xFF;
01044             }
01045         }
01046     }
01047 
01048     LOG_MSG("ANEX86.BMP/FREECG98.BMP font loaded");
01049     fclose(fp);
01050     return true;
01051 fail:
01052     LOG_MSG("ANEX86.BMP/FREECG98.BMP invalid, ignoring");
01053     fclose(fp);
01054     return false;
