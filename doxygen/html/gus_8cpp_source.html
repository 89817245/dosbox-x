00134 static INLINE Bit32s GetSample(Bit32u Delta, Bit32u CurAddr, bool eightbit) {
00135         Bit32u useAddr;
00136         Bit32u holdAddr;
00137         useAddr = CurAddr >> WAVE_FRACT;
00138         if (eightbit) {
00139                 if (Delta >= (1 << WAVE_FRACT)) {
00140                         Bit32s tmpsmall = (Bit8s)GUSRam[useAddr];
00141                         return tmpsmall << 8;
00142                 } else {
00143                         // Interpolate
00144                         Bit32s w1 = ((Bit8s)GUSRam[useAddr+0]) << 8;
00145                         Bit32s w2 = ((Bit8s)GUSRam[useAddr+1]) << 8;
00146                         Bit32s diff = w2 - w1;
00147                         return (w1+((diff*(Bit32s)(CurAddr&WAVE_FRACT_MASK ))>>WAVE_FRACT));
00148                 }
00149         } else {
00150                 // Formula used to convert addresses for use with 16-bit samples
00151                 holdAddr = useAddr & 0xc0000L;
00152                 useAddr = useAddr & 0x1ffffL;
00153                 useAddr = useAddr << 1;
00154                 useAddr = (holdAddr | useAddr);
00155 
00156                 if(Delta >= (1 << WAVE_FRACT)) {
00157                         return (GUSRam[useAddr+0] | (((Bit8s)GUSRam[useAddr+1]) << 8));
00158                 } else {
00159                         // Interpolate
00160                         Bit32s w1 = (GUSRam[useAddr+0] | (((Bit8s)GUSRam[useAddr+1]) << 8));
00161                         Bit32s w2 = (GUSRam[useAddr+2] | (((Bit8s)GUSRam[useAddr+3]) << 8));
00162                         Bit32s diff = w2 - w1;
00163                         return (w1+((diff*(Bit32s)(CurAddr&WAVE_FRACT_MASK ))>>WAVE_FRACT));
00164                 }
00165         }

00294                 Bit32u WaveExtra = 0;
00295                 bool endcondition;
00296 
00297                 if ((WaveCtrl & 0x3) == 0/*voice is running*/) {
00298                         /* NTS: WaveAddr and WaveAdd are unsigned.
00299                          *      If WaveAddr <= WaveAdd going backwards, WaveAddr becomes negative, which as an unsigned integer,
00300                          *      means carrying down from the highest possible value of the integer type. Which means that if the
00301                          *      start position is less than WaveAdd the WaveAddr will skip over the start pointer and continue
00302                          *      playing downward from the top of the GUS memory, without stopping/looping as expected.
00303                          *
00304                          *      This "bug" was implemented on purpose because real Gravis Ultrasound hardware acts this way. */
00305                         if (WaveCtrl & 0x40/*backwards (direction)*/) {
00306                                 /* unsigned int subtract, mask, compare. will miss start pointer if WaveStart <= WaveAdd.
00307                                  * This bug is deliberate, accurate to real GUS hardware, do not fix. */
00308                                 WaveAddr -= WaveAdd;
00309                                 WaveAddr &= ((Bitu)1 << ((Bitu)WAVE_FRACT + (Bitu)20/*1MB*/)) - 1;
00310                                 endcondition = (WaveAddr < WaveStart)?true:false;
00311                                 if (endcondition) WaveExtra = WaveStart - WaveAddr;
00312                         }
00313                         else {
00314                                 WaveAddr += WaveAdd;
00315                                 endcondition = (WaveAddr > WaveEnd)?true:false;
00316                                 WaveAddr &= ((Bitu)1 << ((Bitu)WAVE_FRACT + (Bitu)20/*1MB*/)) - 1;
00317                                 if (endcondition) WaveExtra = WaveAddr - WaveEnd;
00318                         }
00319 
00320                         if (endcondition) {
00321                                 if (WaveCtrl & 0x20) /* generate an IRQ if requested */
00322                                         myGUS.WaveIRQ |= irqmask;
00323 
00324                                 if ((RampCtrl & 0x04/*roll over*/) && !(WaveCtrl & 0x08/*looping*/)) {
00325                                         /* "3.11. Rollover feature
00326                                          * 
00327                                          * Each voice has a 'rollover' feature that allows an application to be notified when a voice's playback position passes
00328                                          * over a particular place in DRAM.  This is very useful for getting seamless digital audio playback.  Basically, the GF1
00329                                          * will generate an IRQ when a voice's current position is  equal to the end position.  However, instead of stopping or
00330                                          * looping back to the start position, the voice will continue playing in the same direction.  This means that there will be
00331                                          * no pause (or gap) in the playback.  Note that this feature is enabled/disabled through the voice's VOLUME control
00332                                          * register (since there are no more bits available in the voice control registers).   A voice's loop enable bit takes
00333                                          * precedence over the rollover.  This means that if a voice's loop enable is on, it will loop when it hits the end position,
00334                                          * regardless of the state of the rollover enable."
00335                                          *
00336                                          * Despite the confusing description above, that means that looping takes precedence over rollover. If not looping, then
00337                                          * rollover means to fire the IRQ but keep moving. If looping, then fire IRQ and carry out loop behavior. Gravis Ultrasound
00338                                          * Windows 3.1 drivers expect this behavior, else Windows WAVE output will not work correctly. */
00339                                 }

00347                                         }
00348                                 }
00349                         }

00355                                 endcondition = (WaveAddr <= WaveStart)?true:false;
00356                         else
00357                                 endcondition = (WaveAddr >= WaveEnd)?true:false;
00358 
00359                         if (endcondition)

00361                 }

00581         }

00591          * This is how "Juice" by Psychic Link is able to play music by GUS timer even though the demo never enables the

00606

01626         Bitu dmalimit = myGUS.memsize;
01627         int step = 0,docount = 0;
01628         bool dma16xlate;
01629         Bitu holdAddr;
01630 
01631         // FIXME: What does the GUS do if the DMA address goes beyond the end of memory?
01632 
01633         /* is this a DMA transfer where the GUS memory address is to be translated for 16-bit PCM?
01634          * Note that there is plenty of code out there that transfers 16-bit PCM with 8-bit DMA,
01635          * and such transfers DO NOT involve the memory address translation.
01636          *
01637          * MODE    XLATE?    NOTE
01638          * 0x00    NO        8-bit PCM, 8-bit DMA (Most DOS programs)
01639          * 0x04    DEPENDS   8-bit PCM, 16-bit DMA (Star Control II, see comments below)
01640          * 0x40    NO        16-bit PCM, 8-bit DMA (Windows 3.1, Quake, for 16-bit PCM over 8-bit DMA)
01641          * 0x44    YES       16-bit PCM, 16-bit DMA (Windows 3.1, Quake, for 16-bit PCM over 16-bit DMA)
01642          *
01643          * Mode 0x04 is marked DEPENDS. It DEPENDS on whether the assigned DMA channel is 8-bit (NO) or 16-bit (YES).
01644          * Mode 0x04 does not appear to be used by any other DOS application or Windows drivers, only by an erratic
01645          * bug in Star Control II. FIXME: But, what does REAL hardware do? Drag out the old Pentium 100MHz with the
01646          * GUS classic and test! --J.C.
01647          *
01648          * Star Control II has a bug where, if the GUS DMA channel is 8-bit (DMA channel 0-3), it will upload
01649          * it's samples to GUS RAM, one DMA transfer per sample, and sometimes set bit 2. Setting bit 2 incorrectly
01650          * tells the GUS it's a 16-bit wide DMA transfer when the DMA channel is 8-bit. But, if the DMA channel is
01651          * 16-bit (DMA channel 5-7), Star Control II will correctly set bit 2 in all cases and samples will
01652          * transfer correctly to GUS RAM.
01653          * */
01654 
01655         /* FIXME: So, if the GUS DMA channel is 8-bit and Star Control II writes mode 0x04 (8-bit PCM, 16-bit DMA), what does the GF1 do?
01656          *        I'm guessing so far that something happens within the GF1 to transfer as 8-bit anyway, clearly the developers of Star Control II
01657          *        did not hear any audible sign that an invalid DMA control was being used. Perhaps the hardware engineers of the GF1 figured
01658          *        out that case and put something in the silicon to ignore the invalid DMA control state. I won't have any answers until
01659          *        I pull out an old Pentium box with a GUS classic and check. --J.C.
01660          *
01661          *        DMA transfers noted by Star Control II that are the reason for this hack (gusdma=1):
01662          *
01663          *        LOG:  157098507 DEBUG MISC:GUS DMA: terminal count reached. DMAControl=0x21
01664          *        LOG:  157098507 DEBUG MISC:GUS DMA transfer 1981 bytes, GUS RAM address 0x0 8-bit DMA 8-bit PCM (ctrl=0x21)
01665          *        LOG:  157098507 DEBUG MISC:GUS DMA: terminal count reached. DMAControl=0x21
01666          *        LOG:  157100331 DEBUG MISC:GUS DMA: terminal count reached. DMAControl=0x21
01667          *        LOG:  157100331 DEBUG MISC:GUS DMA transfer 912 bytes, GUS RAM address 0x7c0 8-bit DMA 8-bit PCM (ctrl=0x21)
01668          *        LOG:  157100331 DEBUG MISC:GUS DMA: terminal count reached. DMAControl=0x21
01669          *        LOG:  157100470 DEBUG MISC:GUS DMA: terminal count reached. DMAControl=0x25
01670          *        LOG:  157100470 DEBUG MISC:GUS DMA transfer 1053 bytes, GUS RAM address 0xb50 16-bit DMA 8-bit PCM (ctrl=0x25)    <-- What?

01672          *        LOG:  157102251 DEBUG MISC:GUS DMA: terminal count reached. DMAControl=0x21
01673          *        LOG:  157102251 DEBUG MISC:GUS DMA transfer 1597 bytes, GUS RAM address 0xf80 8-bit DMA 8-bit PCM (ctrl=0x21)
01674          *        LOG:  157102251 DEBUG MISC:GUS DMA: terminal count reached. DMAControl=0x21
01675          *        LOG:  157104064 DEBUG MISC:GUS DMA: terminal count reached. DMAControl=0x21
01676          *        LOG:  157104064 DEBUG MISC:GUS DMA transfer 2413 bytes, GUS RAM address 0x15c0 8-bit DMA 8-bit PCM (ctrl=0x21)
01677          *        LOG:  157104064 DEBUG MISC:GUS DMA: terminal count reached. DMAControl=0x21
01678          *
01679          *        (end list)
01680          *
01681          *        Noted: Prior to this hack, the samples played by Star Control II sounded more random and often involved leftover
01682          *               sample data in GUS RAM, where with this fix, the music now sounds identical to what is played when using
01683          *               it's Sound Blaster support. */
01684         if (myGUS.dma1 < 4/*8-bit DMA channel*/ && (myGUS.DMAControl & 0x44) == 0x04/*8-bit PCM, 16-bit DMA*/)
01685                 dma16xlate = false; /* Star Control II hack: 8-bit PCM, 8-bit DMA, ignore the bit that says it's 16-bit wide */
01686         else
01687                 dma16xlate = (myGUS.DMAControl & 0x4) ? true : false;
01688 
01689         if (dma16xlate) {
01690                 // 16-bit wide DMA. The GUS SDK specifically mentions that 16-bit DMA is translated
01691                 // to GUS RAM the same way you translate the play pointer. Eugh. But this allows
01692                 // older demos to work properly even if you set the GUS DMA to a 16-bit channel (5)
01693                 // instead of the usual 8-bit channel (1).
01694                 holdAddr = dmaaddr & 0xc0000L;
01695                 dmaaddr = dmaaddr & 0x1ffffL;
01696                 dmaaddr = dmaaddr << 1;
01697                 dmaaddr = (holdAddr | dmaaddr);
01698                 dmalimit = ((dmaaddr & 0xc0000L) | 0x3FFFFL) + 1;
01699         }
01700 
01701         if (dmaaddr < dmalimit)
01702                 docount = dmalimit - dmaaddr;
01703 
01704         docount /= (chan->DMA16+1);

01707 
01708         // hack: some programs especially Gravis Ultrasound MIDI playback like to upload by DMA but never clear the DMA TC flag on the DMA controller.

01711

01713                 if ((myGUS.DMAControl & 0x2) == 0) {

01721                                         // 8-bit data
01722                                         for(i=dmaaddr;i<(dmaaddr+read);i++) GUSRam[i] ^= 0x80;
01723                                 } else {
01724                                         // 16-bit data
01725                                         for(i=dmaaddr+1;i<(dmaaddr+read);i+=2) GUSRam[i] ^= 0x80;
01726                                 }
01727                         }
01728 
01729                         step = read;
01730                 } else {
01731                         //Read data out of UltraSound
01732                         int wd = chan->Write(docount,&GUSRam[dmaaddr]);
01733                         //Check for 16 or 8bit channel
01734                         wd*=(chan->DMA16+1);
01735 
01736                         step = wd;
01737                 }
01738         }
01739 
01740         LOG(LOG_MISC,LOG_DEBUG)("GUS DMA transfer %lu bytes, GUS RAM address 0x%lx %u-bit DMA %u-bit PCM (ctrl=0x%02x) tcount=%u",
01741                 (unsigned long)step,(unsigned long)dmaaddr,myGUS.DMAControl & 0x4 ? 16 : 8,myGUS.DMAControl & 0x40 ? 16 : 8,myGUS.DMAControl,chan->tcount);
01742 
01743         if (step > 0) {
01744                 dmaaddr += (unsigned int)step;
01745 
01746                 if (dma16xlate) {
01747                         holdAddr = dmaaddr & 0xc0000L;
01748                         dmaaddr = dmaaddr & 0x3ffffL;
01749                         dmaaddr = dmaaddr >> 1;
01750                         dmaaddr = (holdAddr | dmaaddr);
01751                 }
01752 
01753                 myGUS.dmaAddr = dmaaddr >> 4;
01754                 myGUS.dmaAddrOffset = dmaaddr & 0xF;
01755         }
01756 
01757         if (chan->tcount) {
01758                 LOG(LOG_MISC,LOG_DEBUG)("GUS DMA transfer hit Terminal Count, setting DMA TC IRQ pending");
01759 
01760                 /* Raise the TC irq, and stop DMA */
01761                 myGUS.IRQStatus |= 0x80;
01762                 saved_tcount = true;
01763                 GUS_CheckIRQ();
01764                 GUS_StopDMA();
01765         }
01766 
01767         chan->tcount = saved_tcount;
01768 }
