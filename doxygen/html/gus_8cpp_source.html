00155 static INLINE Bit32s GetSample(Bit32u Delta, Bit32u CurAddr, bool eightbit) {
00156         Bit32u useAddr;
00157         Bit32u holdAddr;
00158         useAddr = CurAddr >> WAVE_FRACT;
00159         if (eightbit) {
00160                 if (Delta >= (1 << WAVE_FRACT)) {
00161                         Bit32s tmpsmall = (Bit8s)GUSRam[useAddr];
00162                         return tmpsmall << 8;
00163                 } else {
00164                         // Interpolate
00165                         Bit32s w1 = ((Bit8s)GUSRam[useAddr+0]) << 8;
00166                         Bit32s w2 = ((Bit8s)GUSRam[useAddr+1]) << 8;
00167                         Bit32s diff = w2 - w1;
00168                         return (w1+((diff*(Bit32s)(CurAddr&WAVE_FRACT_MASK ))>>WAVE_FRACT));
00169                 }
00170         } else {
00171                 // Formula used to convert addresses for use with 16-bit samples
00172                 holdAddr = useAddr & 0xc0000L;
00173                 useAddr = useAddr & 0x1ffffL;
00174                 useAddr = useAddr << 1;
00175                 useAddr = (holdAddr | useAddr);
00176 
00177                 if(Delta >= (1 << WAVE_FRACT)) {
00178                         return (GUSRam[useAddr+0] | (((Bit8s)GUSRam[useAddr+1]) << 8));
00179                 } else {
00180                         // Interpolate
00181                         Bit32s w1 = (GUSRam[useAddr+0] | (((Bit8s)GUSRam[useAddr+1]) << 8));

00319                          *      means carrying down from the highest possible value of the integer type. Which means that if the
00320                          *      start position is less than WaveAdd the WaveAddr will skip over the start pointer and continue
00321                          *      playing downward from the top of the GUS memory, without stopping/looping as expected.
00322                          *
00323                          *      This "bug" was implemented on purpose because real Gravis Ultrasound hardware acts this way. */

00330                                 if (endcondition) WaveExtra = WaveStart - WaveAddr;
00331                         }

00336                                 if (endcondition) WaveExtra = WaveAddr - WaveEnd;

00347                                          * over a particular place in DRAM.  This is very useful for getting seamless digital audio playback.  Basically, the GF1
00348                                          * will generate an IRQ when a voice's current position is  equal to the end position.  However, instead of stopping or

00498          *
00499          *      Global Data Low (3X4) is either a 16-bit transfer, or the low half of a 16-bit transfer with 8-bit I/O.
00500          *
00501          *      Global Data High (3X5) is either an 8-bit transfer for one of the GF registers or the high part of a 16-bit wide register with 8-bit I/O.

00504          *      LOWER 8 bits, when the code should have been checking the UPPER 8 bits. Programming error #2 was the mis-interpetation of bit 0 (bit 8 of

00506          *      it backwards. */

01684 void GUS_DMA_Event_Transfer(DmaChannel *chan,Bitu dmawords) {
01685         Bitu dmaaddr = (Bitu)(myGUS.dmaAddr << 4ul) + (Bitu)myGUS.dmaAddrOffset;
01686         Bitu dmalimit = myGUS.memsize;
01687     unsigned int docount = 0;
01688         unsigned int step = 0;
01689         bool dma16xlate;
01690         Bitu holdAddr;
01691 
01692         // FIXME: What does the GUS do if the DMA address goes beyond the end of memory?
01693 
01694         /* is this a DMA transfer where the GUS memory address is to be translated for 16-bit PCM?
01695          * Note that there is plenty of code out there that transfers 16-bit PCM with 8-bit DMA,
01696          * and such transfers DO NOT involve the memory address translation.
01697          *
01698          * MODE    XLATE?    NOTE
01699          * 0x00    NO        8-bit PCM, 8-bit DMA (Most DOS programs)
01700          * 0x04    DEPENDS   8-bit PCM, 16-bit DMA (Star Control II, see comments below)
01701          * 0x40    NO        16-bit PCM, 8-bit DMA (Windows 3.1, Quake, for 16-bit PCM over 8-bit DMA)
01702          * 0x44    YES       16-bit PCM, 16-bit DMA (Windows 3.1, Quake, for 16-bit PCM over 16-bit DMA)
01703          *
01704          * Mode 0x04 is marked DEPENDS. It DEPENDS on whether the assigned DMA channel is 8-bit (NO) or 16-bit (YES).
01705          * Mode 0x04 does not appear to be used by any other DOS application or Windows drivers, only by an erratic
01706          * bug in Star Control II. FIXME: But, what does REAL hardware do? Drag out the old Pentium 100MHz with the
01707          * GUS classic and test! --J.C.
01708          *
01709          * Star Control II has a bug where, if the GUS DMA channel is 8-bit (DMA channel 0-3), it will upload
01710          * it's samples to GUS RAM, one DMA transfer per sample, and sometimes set bit 2. Setting bit 2 incorrectly
01711          * tells the GUS it's a 16-bit wide DMA transfer when the DMA channel is 8-bit. But, if the DMA channel is
01712          * 16-bit (DMA channel 5-7), Star Control II will correctly set bit 2 in all cases and samples will
01713          * transfer correctly to GUS RAM.
01714          * */
01715 
01716         /* FIXME: So, if the GUS DMA channel is 8-bit and Star Control II writes mode 0x04 (8-bit PCM, 16-bit DMA), what does the GF1 do?
01717          *        I'm guessing so far that something happens within the GF1 to transfer as 8-bit anyway, clearly the developers of Star Control II
01718          *        did not hear any audible sign that an invalid DMA control was being used. Perhaps the hardware engineers of the GF1 figured
01719          *        out that case and put something in the silicon to ignore the invalid DMA control state. I won't have any answers until
01720          *        I pull out an old Pentium box with a GUS classic and check. --J.C.
01721          *
01722          *        DMA transfers noted by Star Control II that are the reason for this hack (gusdma=1):
01723          *
01724          *        LOG:  157098507 DEBUG MISC:GUS DMA: terminal count reached. DMAControl=0x21
01725          *        LOG:  157098507 DEBUG MISC:GUS DMA transfer 1981 bytes, GUS RAM address 0x0 8-bit DMA 8-bit PCM (ctrl=0x21)
01726          *        LOG:  157098507 DEBUG MISC:GUS DMA: terminal count reached. DMAControl=0x21
01727          *        LOG:  157100331 DEBUG MISC:GUS DMA: terminal count reached. DMAControl=0x21
01728          *        LOG:  157100331 DEBUG MISC:GUS DMA transfer 912 bytes, GUS RAM address 0x7c0 8-bit DMA 8-bit PCM (ctrl=0x21)
01729          *        LOG:  157100331 DEBUG MISC:GUS DMA: terminal count reached. DMAControl=0x21
01730          *        LOG:  157100470 DEBUG MISC:GUS DMA: terminal count reached. DMAControl=0x25
01731          *        LOG:  157100470 DEBUG MISC:GUS DMA transfer 1053 bytes, GUS RAM address 0xb50 16-bit DMA 8-bit PCM (ctrl=0x25)    <-- What?
01732          *        LOG:  157100470 DEBUG MISC:GUS DMA: terminal count reached. DMAControl=0x25
01733          *        LOG:  157102251 DEBUG MISC:GUS DMA: terminal count reached. DMAControl=0x21
01734          *        LOG:  157102251 DEBUG MISC:GUS DMA transfer 1597 bytes, GUS RAM address 0xf80 8-bit DMA 8-bit PCM (ctrl=0x21)
01735          *        LOG:  157102251 DEBUG MISC:GUS DMA: terminal count reached. DMAControl=0x21
01736          *        LOG:  157104064 DEBUG MISC:GUS DMA: terminal count reached. DMAControl=0x21
01737          *        LOG:  157104064 DEBUG MISC:GUS DMA transfer 2413 bytes, GUS RAM address 0x15c0 8-bit DMA 8-bit PCM (ctrl=0x21)
01738          *        LOG:  157104064 DEBUG MISC:GUS DMA: terminal count reached. DMAControl=0x21
01739          *
01740          *        (end list)
01741          *
01742          *        Noted: Prior to this hack, the samples played by Star Control II sounded more random and often involved leftover
01743          *               sample data in GUS RAM, where with this fix, the music now sounds identical to what is played when using
01744          *               it's Sound Blaster support. */
01745         if (myGUS.dma1 < 4/*8-bit DMA channel*/ && (myGUS.DMAControl & 0x44) == 0x04/*8-bit PCM, 16-bit DMA*/)
01746                 dma16xlate = false; /* Star Control II hack: 8-bit PCM, 8-bit DMA, ignore the bit that says it's 16-bit wide */
01747         else
01748                 dma16xlate = (myGUS.DMAControl & 0x4) ? true : false;
01749 
01750         if (dma16xlate) {
01751                 // 16-bit wide DMA. The GUS SDK specifically mentions that 16-bit DMA is translated
01752                 // to GUS RAM the same way you translate the play pointer. Eugh. But this allows
01753                 // older demos to work properly even if you set the GUS DMA to a 16-bit channel (5)
01754                 // instead of the usual 8-bit channel (1).
01755                 holdAddr = dmaaddr & 0xc0000L;
01756                 dmaaddr = dmaaddr & 0x1ffffL;
01757                 dmaaddr = dmaaddr << 1;
01758                 dmaaddr = (holdAddr | dmaaddr);
01759                 dmalimit = ((dmaaddr & 0xc0000L) | 0x3FFFFL) + 1;
01760         }
01761 
01762         if (dmaaddr < dmalimit)
01763                 docount = (unsigned int)(dmalimit - dmaaddr);
01764 
01765         docount /= (chan->DMA16+1u);
01766         if (docount > (chan->currcnt+1u)) docount = chan->currcnt+1u;
01767         if ((Bitu)docount > dmawords) docount = dmawords;
01768 
01769         // hack: some programs especially Gravis Ultrasound MIDI playback like to upload by DMA but never clear the DMA TC flag on the DMA controller.
01770         bool saved_tcount = chan->tcount;
01771         chan->tcount = false;
01772 
01773         if (docount > 0) {
01774                 if ((myGUS.DMAControl & 0x2) == 0) {
01775                         Bitu read=(Bitu)chan->Read((Bitu)docount,&GUSRam[dmaaddr]);
01776                         //Check for 16 or 8bit channel
01777                         read*=(chan->DMA16+1u);
01778                         if((myGUS.DMAControl & 0x80) != 0) {
01779                                 //Invert the MSB to convert twos compliment form
01780                                 Bitu i;
01781                                 if((myGUS.DMAControl & 0x40) == 0) {
01782                                         // 8-bit data
01783                                         for(i=dmaaddr;i<(dmaaddr+read);i++) GUSRam[i] ^= 0x80;
01784                                 } else {
01785                                         // 16-bit data
01786                                         for(i=dmaaddr+1;i<(dmaaddr+read);i+=2) GUSRam[i] ^= 0x80;
01787                                 }
01788                         }
01789 
01790                         step = read;
01791                 } else {
01792                         //Read data out of UltraSound
01793                         Bitu wd = (Bitu)chan->Write((Bitu)docount,&GUSRam[dmaaddr]);
01794                         //Check for 16 or 8bit channel
01795                         wd*=(chan->DMA16+1u);
01796 
01797                         step = wd;
01798                 }
01799         }
01800 
01801         LOG(LOG_MISC,LOG_DEBUG)("GUS DMA transfer %lu bytes, GUS RAM address 0x%lx %u-bit DMA %u-bit PCM (ctrl=0x%02x) tcount=%u",
01802                 (unsigned long)step,(unsigned long)dmaaddr,myGUS.DMAControl & 0x4 ? 16 : 8,myGUS.DMAControl & 0x40 ? 16 : 8,myGUS.DMAControl,chan->tcount);
01803 
01804         if (step > 0) {
01805                 dmaaddr += (unsigned int)step;
01806 
01807                 if (dma16xlate) {
01808                         holdAddr = dmaaddr & 0xc0000L;
01809                         dmaaddr = dmaaddr & 0x3ffffL;
01810                         dmaaddr = dmaaddr >> 1;
01811                         dmaaddr = (holdAddr | dmaaddr);
01812                 }
01813 
01814                 myGUS.dmaAddr = dmaaddr >> 4;
01815                 myGUS.dmaAddrOffset = dmaaddr & 0xF;
01816         }
01817 
01818         if (chan->tcount) {
01819                 LOG(LOG_MISC,LOG_DEBUG)("GUS DMA transfer hit Terminal Count, setting DMA TC IRQ pending");
01820 
01821                 /* Raise the TC irq, and stop DMA */
01822                 myGUS.IRQStatus |= 0x80;
01823                 saved_tcount = true;
01824                 GUS_CheckIRQ();
01825                 GUS_StopDMA();
01826         }
01827 
01828         chan->tcount = saved_tcount;
01829 }

01900 static void GUS_CallBack(Bitu len) {
01901         memset(&MixTemp,0,len*8);
01902         Bitu i;
01903         Bit16s * buf16 = (Bit16s *)MixTemp;
01904         Bit32s * buf32 = (Bit32s *)MixTemp;
01905 
01906         if ((GUS_reset_reg & 0x01) == 0x01) {
01907                 for(i=0;i<myGUS.ActiveChannels;i++)
01908                         guschan[i]->generateSamples(buf32,len);
01909         }
01910 
01911     // FIXME: I wonder if the GF1 chip DAC had more than 16 bits precision
01912     //        to render louder than 100% volume without clipping, and if so,
01913     //        how many extra bits?
01914     //
01915     //        If not, then perhaps clipping and saturation were not a problem
01916     //        unless the volume was set to maximum?
01917     //
01918     //        Time to pull out the GUS MAX and test this theory: what happens
01919     //        if you play samples that would saturate at maximum volume at 16-bit
01920     //        precision? Does it audibly clip or is there some headroom like some
01921     //        sort of 17-bit DAC?
01922     //
01923     //        One way to test is to play a sample on one channel while another
01924     //        channel is set to play a single sample at maximum volume (to see
01925     //        if it makes the audio grungy like a waveform railed to one side).
01926     //
01927     //        Past experience with GUS cards says that at full volume their line
01928     //        out jacks can be quite loud when connected to a speaker.
01929     //
01930     //        While improving this code, a better audio compression function
01931     //        could be implemented that does proper envelope tracking and volume
01932     //        control for better results than this.
01933     //
01934     //        --J.C.
01935 
01936     for(i=0;i<len*2;i++) {
01937         Bit32s sample=((buf32[i] >> 13)*AutoAmp)>>9;
01938         if (sample>32767) {
01939             sample=32767;
01940             if (enable_autoamp) AutoAmp -= 4; /* dampen faster than recovery */
01941         } else if (sample<-32768) {
01942             sample=-32768;
01943             if (enable_autoamp) AutoAmp -= 4; /* dampen faster than recovery */
01944         }
01945         else if (AutoAmp < myGUS.masterVolumeMul) {
01946             AutoAmp++; /* recovery back to 100% normal volume */
01947         }
01948

01950     }
01951 
01952         gus_chan->AddSamples_s16(len,buf16);
01953         CheckVoiceIrq();
