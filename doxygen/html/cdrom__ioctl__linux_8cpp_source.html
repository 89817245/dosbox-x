00071         } else {
00072                 ret = lseek(cdrom_fd, sector * CD_FRAMESIZE, SEEK_SET);
00073                 if (ret >= 0) ret = read(cdrom_fd, buf, buflen);
00074                 if (ret != buflen) ret = -1;
00075         }
00076         close(cdrom_fd);
00077 
00078         MEM_BlockWrite(buffer, buf, buflen);
00079         delete[] buf;
00080
