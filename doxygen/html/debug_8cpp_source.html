00518                         if (data != 0xCC) {
00519                                 oldData = data;
00520                                 mem_writeb(location,0xCC);
00521                         } else if (!active) {
00522                                 // Another activate breakpoint is already here.
00523                                 // Find it, and copy its oldData value
00524                                 CBreakpoint *bp = FindOtherActiveBreakpoint(location, this);

00533                 } else {
00534                         if (mem_readb(location) == 0xCC) {
00535                                 if (oldData == 0xCC)
00536                                         DEBUG_ShowMsg("DEBUG: Internal error while deactivating breakpoint.\n");

00539                                 bool otherActive = (FindOtherActiveBreakpoint(location, this) != 0);
00540 
00541                                 // If so, remove 0xCC and set old value

00615         if (BPoints.empty()) return false;
00616 
00617         // Search matching breakpoint
00618         std::list<CBreakpoint*>::iterator i;

00620         for(i=BPoints.begin(); i != BPoints.end(); ++i) {
00621                 bp = (*i);
00622                 if ((bp->GetType()==BKPNT_PHYSICAL) && bp->IsActive() && (bp->GetSegment()==seg) && (bp->GetOffset()==off)) {
00623                         // Found, 
00624                         if (bp->GetOnce()) {
00625                                 // delete it, if it should only be used once
00626                                 (BPoints.erase)(i);
00627                                 bp->Activate(false);
00628                                 delete bp;
00629                         } else {
00630                                 // Also look for once-only breakpoints at this address
00631                                 bp = FindPhysBreakpoint(seg, off, true);
00632                                 if (bp) {
00633                                         BPoints.remove(bp);
00634                                         bp->Activate(false);
00635                                         delete bp;

00637                         }

02576 char* AnalyzeInstruction(char* inst, bool saveSelector) {
02577         static char result[256];
02578         
02579         char instu[256];
02580         char prefix[3];
02581         Bit16u seg;
02582 
02583         strcpy(instu,inst);
02584         upcase(instu);
02585 
02586         result[0] = 0;
02587         char* pos = strchr(instu,'[');
02588         if (pos) {
02589                 // Segment prefix ?
02590                 if (*(pos-1)==':') {
02591                         char* segpos = pos-3;
02592                         prefix[0] = tolower(*segpos);
02593                         prefix[1] = tolower(*(segpos+1));
02594                         prefix[2] = 0;
02595                         seg = (Bit16u)GetHexValue(segpos,segpos);
02596                 } else {
02597                         if (strstr(pos,"SP") || strstr(pos,"BP")) {
02598                                 seg = SegValue(ss);
02599                                 strcpy(prefix,"ss");
02600                         } else {
02601                                 seg = SegValue(ds);
02602                                 strcpy(prefix,"ds");
02603                         }
02604                 }
02605 
02606                 pos++;
02607                 Bit32u adr = GetHexValue(pos,pos);
02608                 while (*pos!=']') {
02609                         if (*pos=='+') {
02610                                 pos++;
02611                                 adr += GetHexValue(pos,pos);
02612                         } else if (*pos=='-') {
02613                                 pos++;
02614                                 adr -= GetHexValue(pos,pos); 
02615                         } else 
02616                                 pos++;
02617                 }
02618                 Bit32u address = (Bit32u)GetAddress(seg,adr);
02619                 if (!(get_tlb_readhandler(address)->flags & PFLAG_INIT)) {
02620                         static char outmask[] = "%s:[%04X]=%02X";
02621                         
02622                         if (cpu.pmode) outmask[6] = '8';
02623                                 switch (DasmLastOperandSize()) {
02624                                 case 8 : {      Bit8u val = mem_readb(address);
02625                                                         outmask[12] = '2';
02626                                                         sprintf(result,outmask,prefix,adr,val);
02627                                                 }       break;
02628                                 case 16: {      Bit16u val = mem_readw(address);
02629                                                         outmask[12] = '4';
02630                                                         sprintf(result,outmask,prefix,adr,val);
02631                                                 }       break;
02632                                 case 32: {      Bit32u val = mem_readd(address);
02633                                                         outmask[12] = '8';
02634                                                         sprintf(result,outmask,prefix,adr,val);
02635                                                 }       break;
02636                         }
02637                 } else {
02638                         sprintf(result,"[illegal]");
02639                 }
02640                 // Variable found ?
02641                 CDebugVar* var = CDebugVar::FindVar(address);
02642                 if (var) {
02643                         // Replace occurrence
02644                         char* pos1 = strchr(inst,'[');
02645                         char* pos2 = strchr(inst,']');
02646                         if (pos1 && pos2) {
02647                                 char temp[256];
02648                                 strcpy(temp,pos2);                              // save end
02649                                 pos1++; *pos1 = 0;                              // cut after '['
02650                                 strcat(inst,var->GetName());    // add var name
02651                                 strcat(inst,temp);                              // add end
02652                         }
02653                 }
02654                 // show descriptor info, if available
02655                 if ((cpu.pmode) && saveSelector) {
02656                         strcpy(curSelectorName,prefix);
02657                 }
02658         }
02659         // If it is a callback add additional info
02660         pos = strstr(inst,"callback");
02661         if (pos) {
02662                 pos += 9;
02663                 Bitu nr = GetHexValue(pos,pos);
02664                 const char* descr = CALLBACK_GetDescription(nr);
02665                 if (descr) {
02666                         strcat(inst,"  ("); strcat(inst,descr); strcat(inst,")");
02667                 }
02668         }
02669         // Must be a jump
02670         if (instu[0] == 'J')
02671         {
02672                 bool jmp = false;
02673                 switch (instu[1]) {
02674                 case 'A' :      {       jmp = (get_CF()?false:true) && (get_ZF()?false:true); // JA
02675                                         }       break;
02676                 case 'B' :      {       if (instu[2] == 'E') {
02677                                                         jmp = (get_CF()?true:false) || (get_ZF()?true:false); // JBE
02678                                                 } else {
02679                                                         jmp = get_CF()?true:false; // JB
02680                                                 }
02681                                         }       break;
02682                 case 'C' :      {       if (instu[2] == 'X') {
02683                                                         jmp = reg_cx == 0; // JCXZ
02684                                                 } else {
02685                                                         jmp = get_CF()?true:false; // JC
02686                                                 }
02687                                         }       break;
02688                 case 'E' :      {       jmp = get_ZF()?true:false; // JE
02689                                         }       break;
02690                 case 'G' :      {       if (instu[2] == 'E') {
02691                                                         jmp = (get_SF()?true:false)==(get_OF()?true:false); // JGE
02692                                                 } else {
02693                                                         jmp = (get_ZF()?false:true) && ((get_SF()?true:false)==(get_OF()?true:false)); // JG
02694                                                 }
02695                                         }       break;
02696                 case 'L' :      {       if (instu[2] == 'E') {
02697                                                         jmp = (get_ZF()?true:false) || ((get_SF()?true:false)!=(get_OF()?true:false)); // JLE
02698                                                 } else {
02699                                                         jmp = (get_SF()?true:false)!=(get_OF()?true:false); // JL
02700                                                 }
02701                                         }       break;
02702                 case 'M' :      {       jmp = true; // JMP
02703                                         }       break;
02704                 case 'N' :      {       switch (instu[2]) {
02705                                                 case 'B' :      
02706                                                 case 'C' :      {       jmp = get_CF()?false:true;      // JNB / JNC
02707                                                                         }       break;
02708                                                 case 'E' :      {       jmp = get_ZF()?false:true;      // JNE
02709                                                                         }       break;
02710                                                 case 'O' :      {       jmp = get_OF()?false:true;      // JNO
02711                                                                         }       break;
02712                                                 case 'P' :      {       jmp = get_PF()?false:true;      // JNP
02713                                                                         }       break;
02714                                                 case 'S' :      {       jmp = get_SF()?false:true;      // JNS
02715                                                                         }       break;
02716                                                 case 'Z' :      {       jmp = get_ZF()?false:true;      // JNZ
02717                                                                         }       break;
02718                                                 }
02719                                         }       break;
02720                 case 'O' :      {       jmp = get_OF()?true:false; // JO
02721                                         }       break;
02722                 case 'P' :      {       if (instu[2] == 'O') {
02723                                                         jmp = get_PF()?false:true; // JPO
02724                                                 } else {
02725                                                         jmp = get_SF()?true:false; // JP / JPE
02726                                                 }
02727                                         }       break;
02728                 case 'S' :      {       jmp = get_SF()?true:false; // JS
02729                                         }       break;
02730                 case 'Z' :      {       jmp = get_ZF()?true:false; // JZ
02731                                         }       break;
02732                 }
02733                 if (jmp) {
02734                         pos = strchr(instu,'$');
02735                         if (pos) {
02736                                 pos = strchr(instu,'+');
02737                                 if (pos) {
02738                                         strcpy(result,"(down)");
02739                                 } else {
02740                                         strcpy(result,"(up)");
02741                                 }
02742                         }
02743                 } else {
02744                         sprintf(result,"(no jmp)");
02745                 }
02746         }
02747         return result;
02748 }

03337 // Display the content of the MCB chain starting with the MCB at the specified segment.
03338 static void LogMCBChain(Bit16u mcb_segment) {
03339         DOS_MCB mcb(mcb_segment);
03340         char filename[9]; // 8 characters plus a terminating NUL
03341         const char *psp_seg_note;
03342         Bit16u DOS_dataOfs = static_cast<Bit16u>(dataOfs); //Realmode addressing only
03343         PhysPt dataAddr = PhysMake(dataSeg,DOS_dataOfs);// location being viewed in the "Data Overview"
03344 
03345         // loop forever, breaking out of the loop once we've processed the last MCB
03346         while (true) {
03347                 // verify that the type field is valid
03348                 if (mcb.GetType()!=0x4d && mcb.GetType()!=0x5a) {
03349                         LOG(LOG_MISC,LOG_ERROR)("MCB chain broken at %04X:0000!",mcb_segment);
03350                         return;
03351                 }
03352 
03353                 mcb.GetFileName(filename);
03354 
03355                 // some PSP segment values have special meanings
03356                 switch (mcb.GetPSPSeg()) {
03357                         case MCB_FREE:
03358                                 psp_seg_note = "(free)";
03359                                 break;
03360                         case MCB_DOS:
03361                                 psp_seg_note = "(DOS)";
03362                                 break;
03363                         default:
03364                                 psp_seg_note = "";
03365                 }
03366 
03367                 LOG(LOG_MISC,LOG_ERROR)("   %04X  %12u     %04X %-7s  %s",mcb_segment,mcb.GetSize() << 4,mcb.GetPSPSeg(), psp_seg_note, filename);
03368 
03369                 // print a message if dataAddr is within this MCB's memory range
03370                 PhysPt mcbStartAddr = PhysMake(mcb_segment+1,0);
03371                 PhysPt mcbEndAddr = PhysMake(mcb_segment+1+mcb.GetSize(),0);
03372                 if (dataAddr >= mcbStartAddr && dataAddr < mcbEndAddr) {
03373                         LOG(LOG_MISC,LOG_ERROR)("   (data addr %04hX:%04X is %u bytes past this MCB)",dataSeg,DOS_dataOfs,dataAddr - mcbStartAddr);
03374                 }
03375 
03376                 // if we've just processed the last MCB in the chain, break out of the loop
03377                 if (mcb.GetType()==0x5a) {
03378                         break;
03379                 }
03380                 // else, move to the next MCB in the chain
03381                 mcb_segment+=mcb.GetSize()+1;
03382                 mcb.SetPt(mcb_segment);
03383         }
