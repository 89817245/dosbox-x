01685         static char result[256];
01686         
01687         char instu[256];
01688         char prefix[3];
01689         Bit16u seg;
01690 
01691         strcpy(instu,inst);
01692         upcase(instu);
01693 
01694         result[0] = 0;
01695         char* pos = strchr(instu,'[');
01696         if (pos) {
01697                 // Segment prefix ?
01698                 if (*(pos-1)==':') {
01699                         char* segpos = pos-3;
01700                         prefix[0] = tolower(*segpos);
01701                         prefix[1] = tolower(*(segpos+1));
01702                         prefix[2] = 0;
01703                         seg = (Bit16u)GetHexValue(segpos,segpos);
01704                 } else {
01705                         if (strstr(pos,"SP") || strstr(pos,"BP")) {
01706                                 seg = SegValue(ss);
01707                                 strcpy(prefix,"ss");
01708                         } else {
01709                                 seg = SegValue(ds);
01710                                 strcpy(prefix,"ds");
01711                         };
01712                 };
01713 
01714                 pos++;
01715                 Bit32u adr = GetHexValue(pos,pos);
01716                 while (*pos!=']') {
01717                         if (*pos=='+') {
01718                                 pos++;
01719                                 adr += GetHexValue(pos,pos);
01720                         } else if (*pos=='-') {
01721                                 pos++;
01722                                 adr -= GetHexValue(pos,pos); 
01723                         } else 
01724                                 pos++;
01725                 };
01726                 Bit32u address = GetAddress(seg,adr);
01727                 if (!(get_tlb_readhandler(address)->flags & PFLAG_INIT)) {
01728                         static char outmask[] = "%s:[%04X]=%02X";
01729                         
01730                         if (cpu.pmode) outmask[6] = '8';
01731                                 switch (DasmLastOperandSize()) {
01732                                 case 8 : {      Bit8u val = mem_readb(address);
01733                                                         outmask[12] = '2';
01734                                                         sprintf(result,outmask,prefix,adr,val);
01735                                                 }       break;
01736                                 case 16: {      Bit16u val = mem_readw(address);
01737                                                         outmask[12] = '4';
01738                                                         sprintf(result,outmask,prefix,adr,val);
01739                                                 }       break;
01740                                 case 32: {      Bit32u val = mem_readd(address);
01741                                                         outmask[12] = '8';
01742                                                         sprintf(result,outmask,prefix,adr,val);
01743                                                 }       break;
01744                         }
01745                 } else {
01746                         sprintf(result,"[illegal]");
01747                 }
01748                 // Variable found ?
01749                 CDebugVar* var = CDebugVar::FindVar(address);
01750                 if (var) {
01751                         // Replace occurence
01752                         char* pos1 = strchr(inst,'[');
01753                         char* pos2 = strchr(inst,']');
01754                         if (pos1 && pos2) {
01755                                 char temp[256];
01756                                 strcpy(temp,pos2);                              // save end
01757                                 pos1++; *pos1 = 0;                              // cut after '['
01758                                 strcat(inst,var->GetName());    // add var name
01759                                 strcat(inst,temp);                              // add end
01760                         };
01761                 };
01762                 // show descriptor info, if available
01763                 if ((cpu.pmode) && saveSelector) {
01764                         strcpy(curSelectorName,prefix);
01765                 };
01766         };
01767         // If it is a callback add additional info
01768         pos = strstr(inst,"callback");
01769         if (pos) {
01770                 pos += 9;
01771                 Bitu nr = GetHexValue(pos,pos);
01772                 const char* descr = CALLBACK_GetDescription(nr);
01773                 if (descr) {
01774                         strcat(inst,"  ("); strcat(inst,descr); strcat(inst,")");
01775                 }
01776         };
01777         // Must be a jump
01778         if (instu[0] == 'J')
01779         {
01780                 bool jmp = false;
01781                 switch (instu[1]) {
01782                 case 'A' :      {       jmp = (get_CF()?false:true) && (get_ZF()?false:true); // JA
01783                                         }       break;
01784                 case 'B' :      {       if (instu[2] == 'E') {
01785                                                         jmp = (get_CF()?true:false) || (get_ZF()?true:false); // JBE
01786                                                 } else {
01787                                                         jmp = get_CF()?true:false; // JB
01788                                                 }
01789                                         }       break;
01790                 case 'C' :      {       if (instu[2] == 'X') {
01791                                                         jmp = reg_cx == 0; // JCXZ
01792                                                 } else {
01793                                                         jmp = get_CF()?true:false; // JC
01794                                                 }
01795                                         }       break;
01796                 case 'E' :      {       jmp = get_ZF()?true:false; // JE
01797                                         }       break;
01798                 case 'G' :      {       if (instu[2] == 'E') {
01799                                                         jmp = (get_SF()?true:false)==(get_OF()?true:false); // JGE
01800                                                 } else {
01801                                                         jmp = (get_ZF()?false:true) && ((get_SF()?true:false)==(get_OF()?true:false)); // JG
01802                                                 }
01803                                         }       break;
01804                 case 'L' :      {       if (instu[2] == 'E') {
01805                                                         jmp = (get_ZF()?true:false) || ((get_SF()?true:false)!=(get_OF()?true:false)); // JLE
01806                                                 } else {
01807                                                         jmp = (get_SF()?true:false)!=(get_OF()?true:false); // JL
01808                                                 }
01809                                         }       break;
01810                 case 'M' :      {       jmp = true; // JMP
01811                                         }       break;
01812                 case 'N' :      {       switch (instu[2]) {
01813                                                 case 'B' :      
01814                                                 case 'C' :      {       jmp = get_CF()?false:true;      // JNB / JNC
01815                                                                         }       break;
01816                                                 case 'E' :      {       jmp = get_ZF()?false:true;      // JNE
01817                                                                         }       break;
01818                                                 case 'O' :      {       jmp = get_OF()?false:true;      // JNO
01819                                                                         }       break;
01820                                                 case 'P' :      {       jmp = get_PF()?false:true;      // JNP
01821                                                                         }       break;
01822                                                 case 'S' :      {       jmp = get_SF()?false:true;      // JNS
01823                                                                         }       break;
01824                                                 case 'Z' :      {       jmp = get_ZF()?false:true;      // JNZ
01825                                                                         }       break;
01826                                                 }
01827                                         }       break;
01828                 case 'O' :      {       jmp = get_OF()?true:false; // JO
01829                                         }       break;
01830                 case 'P' :      {       if (instu[2] == 'O') {
01831                                                         jmp = get_PF()?false:true; // JPO
01832                                                 } else {
01833                                                         jmp = get_SF()?true:false; // JP / JPE
01834                                                 }
01835                                         }       break;
01836                 case 'S' :      {       jmp = get_SF()?true:false; // JS
01837                                         }       break;
01838                 case 'Z' :      {       jmp = get_ZF()?true:false; // JZ
01839                                         }       break;
01840                 }
01841                 if (jmp) {
01842                         pos = strchr(instu,'$');
01843                         if (pos) {
01844                                 pos = strchr(instu,'+');
01845                                 if (pos) {
01846                                         strcpy(result,"(down)");
01847                                 } else {
01848                                         strcpy(result,"(up)");
01849                                 }
01850                         }
01851                 } else {
01852                         sprintf(result,"(no jmp)");
01853                 }
01854         }
01855         return result;
01856 };
