00172 CacheBlockDynRec * LinkBlocks(BlockReturn ret) {
00173         CacheBlockDynRec * block=NULL;
00174         // the last instruction was a control flow modifying instruction
00175         Bit32u temp_ip=SegPhys(cs)+reg_eip;
00176         CodePageHandlerDynRec * temp_handler=(CodePageHandlerDynRec *)get_tlb_readhandler(temp_ip);
00177         if (temp_handler->flags & PFLAG_HASCODE) {
00178                 // see if the target is an already translated block
00179                 block=temp_handler->FindCacheBlock(temp_ip & 4095);
00180                 if (!block) return NULL;
00181 
00182                 // found it, link the current block to
00183                 cache.block.running->LinkTo(ret==BR_Link2,block);
00184                 return block;
00185         }
00186         return NULL;

00189 /*
00190         The core tries to find the block that should be executed next.
00191         If such a block is found, it is run, otherwise the instruction
00192         stream starting at ip_point is translated (see decoder.h) and
00193         makes up a new code block that will be run.
00194         When control is returned to CPU_Core_Dynrec_Run (which might
00195         be right after the block is run, or somewhen long after that
00196         due to the direct cacheblock linking) the returncode decides
00197         the next action. This might be continuing the translation and
00198         execution process, or returning from the core etc.
00199 */

00206 Bits CPU_Core_Dynrec_Run(void) {
00207     if (CPU_Cycles <= 0)
00208             return CBRET_NONE;
00209 
00210     /* Dynamic core is NOT compatible with the way page faults
00211      * in the guest are handled in this emulator. Do not use
00212      * dynamic core if paging is enabled. Do not comment this
00213      * out, even if it happens to work for a minute, a half
00214      * hour, a day, because it will turn around and cause
00215      * Windows 95 to crash when you've become most comfortable
00216      * with the idea that it works. This code cannot handle
00217      * the sudden context switch of a page fault and it never
00218      * will. Don't do it. You have been warned. */
00219     if (paging.enabled && !use_dynamic_core_with_paging) {
00220         if (paging_warning) {
00221             LOG_MSG("Dynamic core warning: The guest OS/Application has just switched on 80386 paging, which is not supported by the dynamic core. The normal core will be used until paging is switched off again.");
00222             paging_warning = false;
00223         }
00224 
00225         return CPU_Core_Normal_Run();
00226     }
00227 
00228         for (;;) {
00229                 // Determine the linear address of CS:EIP
00230                 PhysPt ip_point=SegPhys(cs)+reg_eip;
00231                 #if C_HEAVY_DEBUG
00232                         if (DEBUG_HeavyIsBreakpoint()) return (Bits)debugCallback;
00233                 #endif
00234 
00235                 CodePageHandlerDynRec * chandler=0;
00236                 // see if the current page is present and contains code
00237                 if (GCC_UNLIKELY(MakeCodePage(ip_point,chandler))) {
00238                         // page not present, throw the exception
00239                         CPU_Exception(cpu.exception.which,cpu.exception.error);
00240                         continue;
00241                 }
00242 
00243                 // page doesn't contain code or is special
00244                 if (GCC_UNLIKELY(!chandler)) return CPU_Core_Normal_Run();
00245 
00246                 // find correct Dynamic Block to run
00247                 CacheBlockDynRec * block=chandler->FindCacheBlock(ip_point&4095);
00248                 if (!block) {
00249                         // no block found, thus translate the instruction stream
00250                         // unless the instruction is known to be modified
00251                         if (!chandler->invalidation_map || (chandler->invalidation_map[ip_point&4095]<4)) {
00252                                 // translate up to 32 instructions
00253                                 block=CreateCacheBlock(chandler,ip_point,32);
00254                         } else {
00255                                 // let the normal core handle this instruction to avoid zero-sized blocks
00256                                 cpu_cycles_count_t old_cycles=CPU_Cycles;
00257                                 CPU_Cycles=1;
00258                                 Bits nc_retcode=CPU_Core_Normal_Run();
00259                                 if (!nc_retcode) {
00260                                         CPU_Cycles=old_cycles-1;
00261                                         continue;
00262                                 }
00263                                 CPU_CycleLeft+=old_cycles;
00264                                 return nc_retcode;
00265                         }
00266                 }
00267 
00268 run_block:
00269                 cache.block.running=0;
00270                 // now we're ready to run the dynamic code block
00271 //              BlockReturn ret=((BlockReturn (*)(void))(block->cache.start))();
00272                 BlockReturn ret=core_dynrec.runcode(block->cache.start);
00273 
00274         if (sizeof(CPU_Cycles) > 4) {
00275             // HACK: All dynrec cores for each processor assume CPU_Cycles is 32-bit wide.
00276             //       The purpose of this hack is to sign-extend the lower 32 bits so that
00277             //       when CPU_Cycles goes negative it doesn't suddenly appear as a very
00278             //       large integer value.
00279             //
00280             //       This hack is needed for dynrec to work on x86_64 targets.
00281             CPU_Cycles = (Bits)((int32_t)CPU_Cycles);
00282         }
00283 
00284                 switch (ret) {
00285                 case BR_Iret:
00286 #if C_DEBUG
00287 #if C_HEAVY_DEBUG
00288                         if (DEBUG_HeavyIsBreakpoint()) return (Bits)debugCallback;
00289 #endif
00290 #endif
00291                         if (!GETFLAG(TF)) {
00292                                 if (GETFLAG(IF) && PIC_IRQCheck) return CBRET_NONE;
00293                                 break;
00294                         }
00295                         // trapflag is set, switch to the trap-aware decoder
00296                         cpudecoder=CPU_Core_Dynrec_Trap_Run;
00297                         return CBRET_NONE;
00298 
00299                 case BR_Normal:
00300                         // the block was exited due to a non-predictable control flow
00301                         // modifying instruction (like ret) or some nontrivial cpu state
00302                         // changing instruction (for example switch to/from pmode),
00303                         // or the maximum number of instructions to translate was reached
00304 #if C_DEBUG
00305 #if C_HEAVY_DEBUG
00306                         if (DEBUG_HeavyIsBreakpoint()) return (Bits)debugCallback;
00307 #endif
00308 #endif
00309                         break;
00310 
00311                 case BR_Cycles:
00312                         // cycles went negative, return from the core to handle
00313                         // external events, schedule the pic...
00314 #if C_DEBUG
00315 #if C_HEAVY_DEBUG
00316                         if (DEBUG_HeavyIsBreakpoint()) return (Bits)debugCallback;
00317 #endif
00318 #endif
00319                         return CBRET_NONE;
00320 
00321                 case BR_CallBack:
00322                         // the callback code is executed in dosbox.conf, return the callback number
00323                         FillFlags();
00324                         return (Bits)core_dynrec.callback;
00325 
00326                 case BR_SMCBlock:
00327 //                      LOG_MSG("selfmodification of running block at %x:%x",SegValue(cs),reg_eip);
00328                         cpu.exception.which=0;
00329                         // fallthrough, let the normal core handle the block-modifying instruction
00330                 case BR_Opcode:
00331                         // some instruction has been encountered that could not be translated
00332                         // (thus it is not part of the code block), the normal core will
00333                         // handle this instruction
00334                         CPU_CycleLeft+=CPU_Cycles;
00335                         CPU_Cycles=1;
00336                         return CPU_Core_Normal_Run();
00337 
00338 #if (C_DEBUG)
00339                 case BR_OpcodeFull:
00340                         CPU_CycleLeft+=CPU_Cycles;
00341                         CPU_Cycles=1;
00342                         return CPU_Core_Full_Run();
00343 #endif
00344 
00345                 case BR_Link1:
00346                 case BR_Link2:
00347                         block=LinkBlocks(ret);
00348                         if (block) goto run_block;
00349                         break;
00350 
00351                 default:
00352                         E_Exit("Invalid return code %d", ret);
00353                 }
00354         }
00355         return CBRET_NONE;
00356 }

00358 Bits CPU_Core_Dynrec_Trap_Run(void) {
00359         Bits oldCycles = CPU_Cycles;
00360         CPU_Cycles = 1;
00361         cpu.trap_skip = false;
00362 
00363         // let the normal core execute the next (only one!) instruction
00364         Bits ret=CPU_Core_Normal_Run();
00365 
00366         // trap to int1 unless the last instruction deferred this
00367         // (allows hardware interrupts to be served without interaction)
00368         if (!cpu.trap_skip) CPU_HW_Interrupt(1);
00369 
00370         CPU_Cycles = oldCycles-1;
00371         // continue (either the trapflag was clear anyways, or the int1 cleared it)
00372         cpudecoder = &CPU_Core_Dynrec_Run;
00373 
00374         return ret;
00375 }
