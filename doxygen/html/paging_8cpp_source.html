00272 // PAGING_NewPageFault
00273 // lin_addr, page_addr: the linear and page address the fault happened at
00274 // prepare_only: true in case the calling core handles the fault, else the PageFaultCore does
00275 static void PAGING_NewPageFault(PhysPt lin_addr, Bitu page_addr, bool prepare_only, Bitu faultcode) {
00276         paging.cr2=lin_addr;
00277         //LOG_MSG("FAULT q%d, code %x",  pf_queue.used, faultcode);
00278         //PrintPageInfo("FA+",lin_addr,faultcode, prepare_only);
00279 
00280         if (prepare_only) {
00281                 cpu.exception.which = EXCEPTION_PF;
00282                 cpu.exception.error = faultcode;
00283         } else if (dosbox_allow_nonrecursive_page_fault) {
00284                 throw GuestPageFaultException(lin_addr,page_addr,faultcode);
00285         } else {
00286                 // Save the state of the cpu cores
00287                 LazyFlags old_lflags;
00288                 memcpy(&old_lflags,&lflags,sizeof(LazyFlags));
00289                 CPU_Decoder * old_cpudecoder;
00290                 old_cpudecoder=cpudecoder;
00291                 cpudecoder=&PageFaultCore;
00292                 LOG(LOG_PAGING,LOG_NORMAL)("Recursive PageFault for %lx used=%d",(unsigned long)lin_addr,(int)pf_queue.used);
00293                 if (pf_queue.used >= PF_QUEUESIZE) E_Exit("PF queue overrun.");
00294                 if (pf_queue.used != 0) LOG_MSG("Warning: PAGING_NewPageFault() more than one level, now using level %d\n",(int)pf_queue.used+1);
00295                 PF_Entry * entry=&pf_queue.entries[pf_queue.used++];
00296                 entry->cs=SegValue(cs);
00297                 entry->eip=reg_eip;
00298                 entry->page_addr=page_addr;
00299                 entry->mpl=cpu.mpl;
00300                 cpu.mpl=3;
00301                 CPU_Exception(EXCEPTION_PF,faultcode);
00302                 DOSBOX_RunMachine();
00303                 pf_queue.used--;
00304                 memcpy(&lflags,&old_lflags,sizeof(LazyFlags));
00305                 cpudecoder=old_cpudecoder;
00306         }

00840         Bitu xlat_index = linkmode | (paging.wp? 8:0) | ((cpu.cpl==3)? 4:0);
00841         Bit8u outcome = xlat_mapping[xlat_index];
00842 
00843         // get the physpage handler we are going to map 
00844         PageHandler * handler=MEM_GetPageHandler(phys_page);
00845         Bitu lin_base=lin_page << 12;
00846 
00847 //      LOG_MSG("MAPPG %s",lnm[outcome]);

00852                 LOG(LOG_PAGING,LOG_NORMAL)("Not enough paging links, resetting cache");
00853                 PAGING_ClearTLB();

00855         // re-use some of the unused bits in the phys_page variable
00856         // needed in the exception handler and foiler so they can replace themselves appropriately

00860         paging.tlb.phys_page[lin_page]= phys_page | (linkmode<< 30) | (dirty? PHYSPAGE_DITRY:0);

00862         case ACMAP_RW:

00865                         handler->GetHostReadPt(phys_page)-lin_base;
00866         else paging.tlb.read[lin_page]=0;
00867         paging.tlb.readhandler[lin_page]=handler;
00868                 
00869                 // write
00870                 if (dirty) { // in case it is already dirty we don't need to check
00871                         if (handler->getFlags() & PFLAG_WRITEABLE) paging.tlb.write[lin_page] = 
00872                                 handler->GetHostWritePt(phys_page)-lin_base;
00873                         else paging.tlb.write[lin_page]=0;
00874         paging.tlb.writehandler[lin_page]=handler;
00875                 } else {
00876                         paging.tlb.writehandler[lin_page]= &foiling_handler;
00877                         paging.tlb.write[lin_page]=0;
00878                 }
00879                 break;
00880         case ACMAP_RE:
00881                 // read
00882                 if (handler->getFlags() & PFLAG_READABLE) paging.tlb.read[lin_page] = 
00883                         handler->GetHostReadPt(phys_page)-lin_base;
00884                 else paging.tlb.read[lin_page]=0;
00885                 paging.tlb.readhandler[lin_page]=handler;
00886                 // exception
00887                 paging.tlb.writehandler[lin_page]= &exception_handler;
00888                 paging.tlb.write[lin_page]=0;
00889                 break;
00890         case ACMAP_EE:
00891                 paging.tlb.readhandler[lin_page]= &exception_handler;
00892                 paging.tlb.writehandler[lin_page]= &exception_handler;
00893                 paging.tlb.read[lin_page]=0;
00894                 paging.tlb.write[lin_page]=0;
00895                 break;
00896 }
00897 
00898         switch(linkmode) {
00899         case ACCESS_KR:
00900                 paging.kr_links.entries[paging.kr_links.used++]=lin_page;
00901                 break;
00902         case ACCESS_KRW:
00903                 paging.krw_links.entries[paging.krw_links.used++]=lin_page;
00904                 break;
