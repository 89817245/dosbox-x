00786 Bit16u device_CON::GetInformation(void) {
00787         if (dos_con_use_int16_to_detect_input || IS_PC98_ARCH) {
00788                 Bit16u ret = 0x80D3; /* No Key Available */
00789 
00790                 /* DOSBox-X behavior: Use INT 16h AH=0x11 Query keyboard status/preview key.
00791                  * The reason we do this is some DOS programs actually rely on hooking INT 16h
00792                  * to manipulate, hide, or transform what the DOS CON driver sees as well as
00793                  * itself. Perhaps the most disgusting example of this behavior would be the
00794                  * SCANDISK.EXE utility in Microsoft MS-DOS 6.22, which apparently relies on
00795                  * hooking INT 16h in this way to catch the Escape, CTRL+C, and some other
00796                  * scan codes in order to "eat" the scan codes before they get back to DOS.
00797                  * The reason they can get away with it apparently and still respond properly
00798                  * to those keys, is because the MS-DOS 6.22 CON driver always calls INT 16h
00799                  * AH=0x11 first before calling INT 16h AH=0x10 to fetch the scan code.
00800                  *
00801                  * Without this fix, SCANDISK.EXE does not respond properly to Escape and
00802                  * a few other keys. Pressing Escape will do nothing until you hit any other
00803                  * key, at which point it suddenly acts upon the Escape key.
00804                  *
00805                  * Since Scandisk is using INT 21h AH=0x0B to query STDIN during this time,
00806                  * this implementation is a good "halfway" compromise in that this call
00807                  * will trigger the INT 16h AH=0x11 hook it relies on. */
00808                 if (readcache || dev_con_pos < dev_con_max) return 0x8093; /* key available */
00809 
00810                 Bitu saved_ax = reg_ax;
00811 
00812                 reg_ah = (IS_EGAVGA_ARCH)?0x11:0x1; // check for keystroke
00813 
00814         /* FIXME: PC-98 emulation should eventually use CONIO emulation that
00815          *        better emulates the actual platform. The purpose of this
00816          *        hack is to allow our code to call into INT 16h without
00817          *        setting up an INT 16h vector */
00818         if (IS_PC98_ARCH)
00819             INT16_Handler_Wrap();
00820         else
00821             CALLBACK_RunRealInt(0x16);
00822 
00823         if (!GETFLAG(ZF)) { /* key is present, waiting to be returned on AH=0x10 or AH=0x00 */
00824             if (IS_PC98_ARCH && reg_al == 0) {
00825                 /* some scan codes are ignored by CON, and wouldn't read anything.
00826                  * while we're at it, take the scan code and convert it into ANSI here
00827                  * so that Read() returns it immediately instead of doing this conversion itself.
00828                  * This way we never block when we SAID a key was available that gets ignored. */
