00311 void MEM_CalloutObject::Install(Bitu page,Bitu pagemask/*MEMMASK_ISA_10BIT, etc.*/,MEM_CalloutHandler *handler) {
00312         if(!installed) {
00313         if (pagemask == 0 || (pagemask & ~0xFFFFFFFU)) {
00314             LOG(LOG_MISC,LOG_ERROR)("MEM_CalloutObject::Install: Page mask %x is invalid",(unsigned int)pagemask);
00315             return;
00316         }
00317 
00318         /* we need a mask for the distance between aliases of the port, and the range of I/O ports. */
00319         /* only the low part of the mask where bits are zero, not the upper.
00320          * This loop is the reason portmask cannot be ~0 else it would become an infinite loop.
00321          * This also serves to check that the mask is a proper combination of ISA masking and
00322          * I/O port range (example: IOMASK_ISA_10BIT & (~0x3) == 0x3FF & 0xFFFFFFFC = 0x3FC for a device with 4 I/O ports).
00323          * A proper mask has (from MSB to LSB):
00324          *   - zero or more 0 bits from MSB
00325          *   - 1 or more 1 bits in the middle
00326          *   - zero or more 0 bits to LSB */
00327         {
00328             Bitu m = 1;
00329             Bitu test;
00330 
00331             /* compute range mask from zero bits at LSB */
00332             range_mask = 0;
00333             test = pagemask ^ 0xFFFFFFFU;
00334             while ((test & m) == m) {
00335                 range_mask = m;
00336                 m = (m << 1) + 1;
00337             }
00338 
00339             /* DEBUG */
00340             if ((pagemask & range_mask) != 0 ||
00341                 ((range_mask + 1) & range_mask) != 0/* should be a mask, therefore AND by itself + 1 should be zero (think (0xF + 1) & 0xF = 0x10 & 0xF = 0x0) */) {
00342                 LOG(LOG_MISC,LOG_ERROR)("MEM_CalloutObject::Install: pagemask(%x) & range_mask(%x) != 0 (%x). You found a corner case that broke this code, fix it.",
00343                     (unsigned int)pagemask,
00344                     (unsigned int)range_mask,
00345                     (unsigned int)(pagemask & range_mask));
00346                 return;
00347             }
00348 
00349             /* compute alias mask from middle 1 bits */
00350             alias_mask = range_mask;
00351             test = pagemask + range_mask; /* will break if portmask & range_mask != 0 */
00352             while ((test & m) == m) {
00353                 alias_mask = m;
00354                 m = (m << 1) + 1;
00355             }
00356 
00357             /* any bits after that should be zero. */
00358             /* confirm this by XORing portmask by (alias_mask ^ range_mask). */
00359             /* we already confirmed that portmask & range_mask == 0. */
00360             /* Example:
00361              *
00362              *    Sound Blaster at port 220-22Fh with 10-bit ISA decode would be 0x03F0 therefore:
00363              *      portmask =    0x03F0
00364              *      range_mask =  0x000F
00365              *      alias_mask =  0x03FF
00366              *
00367              *      portmask ^ range_mask = 0x3FF
00368              *      portmask ^ range_mask ^ alias_mask = 0x0000
00369              *
00370              * Example of invalid portmask 0x13F0:
00371              *      portmask =    0x13F0
00372              *      range_mask =  0x000F
00373              *      alias_mask =  0x03FF
00374              *
00375              *      portmask ^ range_mask = 0x13FF
00376              *      portmask ^ range_mask ^ alias_mask = 0x1000 */
00377             if ((pagemask ^ range_mask ^ alias_mask) != 0 ||
00378                 ((alias_mask + 1) & alias_mask) != 0/* should be a mask, therefore AND by itself + 1 should be zero */) {
00379                 LOG(LOG_MISC,LOG_ERROR)("MEM_CalloutObject::Install: pagemask(%x) ^ range_mask(%x) ^ alias_mask(%x) != 0 (%x). Invalid portmask.",
00380                     (unsigned int)pagemask,
00381                     (unsigned int)range_mask,
00382                     (unsigned int)alias_mask,
00383                     (unsigned int)(pagemask ^ range_mask ^ alias_mask));
00384                 return;
00385             }
00386 
00387             if (page & range_mask) {
00388                 LOG(LOG_MISC,LOG_ERROR)("MEM_CalloutObject::Install: page %x and page mask %x not aligned (range_mask %x)",
00389                     (unsigned int)page,(unsigned int)pagemask,(unsigned int)range_mask);
00390                 return;
00391             }
00392         }
00393 
00394                 installed=true;
00395                 m_base=page;
00396         mem_mask=pagemask;
00397         m_handler=handler;
00398 
00399         /* add this object to the callout array.
00400          * don't register any I/O handlers. those will be registered during the "slow path"
00401          * callout process when the CPU goes to access them. to encourage that to happen,
00402          * we invalidate the I/O ranges */
00403         LOG(LOG_MISC,LOG_DEBUG)("MEM_CalloutObject::Install added device with page=0x%x mem_mask=0x%x rangemask=0x%x aliasmask=0x%x",
00404             (unsigned int)page,(unsigned int)mem_mask,(unsigned int)range_mask,(unsigned int)alias_mask);
00405 
00406         InvalidateCachedHandlers();
00407     }
00408 }

00416 /* callers maintain a handle to it.
00417  * if they need to touch it, they get a pointer, which they then have to put back.
00418  * The way DOSBox/DOSbox-X code handles MEM callbacks it's common to declare an MEM object,
00419  * call the install, but then never touch it again, so this should work fine.
00420  *
00421  * this allows us to maintain ready-made MEM callout objects to return quickly rather
00422  * than write more complicated code where the caller has to make an MEM_CalloutObject
00423  * and then call install and we have to add it's pointer to a list/vector/whatever.
00424  * It also avoids problems where if we have to resize the vector, the pointers become
00425  * invalid, because callers have only handles and they have to put all the pointers
00426  * back in order for us to resize the vector. */
00427 MEM_Callout_t MEM_AllocateCallout(MEM_Type_t t) {
00428     if (t < MEM_TYPE_MIN || t >= MEM_TYPE_MAX)
00429         return MEM_Callout_t_none;
00430 
00431     MEM_callout_vector &vec = MEM_callouts[t - MEM_TYPE_MIN];
00432 
00433 try_again:
00434     while (vec.alloc_from < vec.size()) {
00435         MEM_CalloutObject &obj = vec[vec.alloc_from];
00436 
00437         if (!obj.alloc) {
00438             obj.alloc = true;
00439             assert(obj.isInstalled() == false);
00440             return MEM_Callout_t_comb(t,vec.alloc_from++); /* make combination, then increment alloc_from */
00441         }
00442 
00443         vec.alloc_from++;
00444     }
00445 
00446     /* okay, double the size of the vector within reason.
00447      * if anyone has pointers out to our elements, then we cannot resize. vector::resize() may invalidate them. */
00448     if (vec.size() < 4096 && vec.getcounter == 0) {
00449         size_t nsz = vec.size() * 2;
00450 
00451         LOG(LOG_MISC,LOG_WARN)("MEM_AllocateCallout type %u expanding array to %u",(unsigned int)t,(unsigned int)nsz);
00452         vec.alloc_from = vec.size(); /* allocate from end of old vector size */
00453         vec.resize(nsz);
00454         goto try_again;
00455     }
00456 
00457     LOG(LOG_MISC,LOG_WARN)("MEM_AllocateCallout type %u no free entries",(unsigned int)t);
00458     return MEM_Callout_t_none;
00459 }

00792 INLINE Bitu BestMatch(Bitu size) {
00793         Bitu index=XMS_START;   
00794         Bitu first=0;
00795         Bitu best=0xfffffff;
00796         Bitu best_first=0;
00797         while (index<memory.reported_pages) {
00798                 /* Check if we are searching for first free page */
00799                 if (!first) {
00800                         /* Check if this is a free page */
00801                         if (!memory.mhandles[index]) {
00802                                 first=index;    
00803                         }
00804                 } else {
00805                         /* Check if this still is used page */

00824         }
