00081

00083     d = (((unsigned int)tm->tm_year - 80u) << 9u) + (((unsigned int)tm->tm_mon + 1u) << 5u) + (unsigned int)tm->tm_mday;

01174         (bootbuffer.headcount > ((filesize <= 3000) ? 64 : 255))) {
01175         LOG_MSG("Rejecting image, boot sector has weird values not consistent with FAT filesystem");
01176                 created_successfully = false;
01177         return;
01178     }
01179 
01180     /* work at this point in logical sectors */
01181         sector_size = loadedDisk->getSectSize();

01184      * that indicates 1024 bytes per sector. */
01185     if (pc98_512_to_1024_allow &&
01186          bootbuffer.bytespersector != getSectSize() &&
01187          bootbuffer.bytespersector >  getSectSize() &&
01188         (bootbuffer.bytespersector %  getSectSize()) == 0) {
01189         unsigned int ratioshift = 1;
01190

01194         unsigned int ratio = 1u << ratioshift;
01195 
01196         LOG_MSG("Disk indicates %u bytes/sector, FAT filesystem indicates %u bytes/sector. Ratio=%u:1 shift=%u",
01197                 getSectSize(),bootbuffer.bytespersector,ratio,ratioshift);

01242      *      most common appears to be 512 byte/sector HDI images formatted with 1024 byte/sector FAT filesystems.

01247                 (bootbuffer.rootdirentries == 0) ||

01263      * PC-98 games are normally distributed in on the internet.

01269      * 15        15 & 14       01111 AND 01110     RESULT: 01110 (15)

01276         return;
01277     }

01280      * the bytespersector value of the MS-DOS filesystem. if they don't match, problems
01281      * will result. */

01283         LOG_MSG("FAT bytes/sector %u does not match disk image bytes/sector %u",
01284             (unsigned int)bootbuffer.bytespersector,
01285             (unsigned int)getSectSize());
01286                 created_successfully = false;
01287         return;
01288     }
01289 
01290         /* Filesystem must be contiguous to use absolute sectors, otherwise CHS will be used */
01291         absolute = IS_PC98_ARCH || ((bootbuffer.headcount == headscyl) && (bootbuffer.sectorspertrack == cylsector));
01292 
01293         /* Determine FAT format, 12, 16 or 32 */
01294 
01295         /* Get size of root dir in sectors */
01296         Bit32u RootDirSectors = ((bootbuffer.rootdirentries * 32u) + (bootbuffer.bytespersector - 1u)) / bootbuffer.bytespersector;
01297         Bit32u DataSectors;
01298         if(bootbuffer.totalsectorcount != 0) {
01299                 DataSectors = (Bitu)bootbuffer.totalsectorcount - ((Bitu)bootbuffer.reservedsectors + ((Bitu)bootbuffer.fatcopies * (Bitu)bootbuffer.sectorsperfat) + (Bitu)RootDirSectors);
01300         } else {
01301                 DataSectors = (Bitu)bootbuffer.totalsecdword - ((Bitu)bootbuffer.reservedsectors + ((Bitu)bootbuffer.fatcopies * (Bitu)bootbuffer.sectorsperfat) + (Bitu)RootDirSectors);
01302 
01303         }
01304         CountOfClusters = DataSectors / bootbuffer.sectorspercluster;
01305 
01306         firstDataSector = ((Bitu)bootbuffer.reservedsectors + ((Bitu)bootbuffer.fatcopies * (Bitu)bootbuffer.sectorsperfat) + (Bitu)RootDirSectors) + (Bitu)partSectOff;
01307         firstRootDirSect = (Bitu)bootbuffer.reservedsectors + ((Bitu)bootbuffer.fatcopies * (Bitu)bootbuffer.sectorsperfat) + (Bitu)partSectOff;

01310                 /* Volume is FAT12 */
01311                 LOG_MSG("Mounted FAT volume is FAT12 with %d clusters", CountOfClusters);

01313         } else if (CountOfClusters < 65525) {
01314                 LOG_MSG("Mounted FAT volume is FAT16 with %d clusters", CountOfClusters);
01315                 fattype = FAT16;
01316         } else {

01318                 fattype = FAT32;
01319         }
01320 
01321         /* There is no cluster 0, this means we are in the root directory */
01322         cwdDirCluster = 0;
01323 
01324         memset(fatSectBuffer,0,1024);
01325         curFatSect = 0xffffffff;
01326
