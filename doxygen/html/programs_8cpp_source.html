00228         if (env_seg == 0) {

00276 bool EnvPhys_ScanUntilNextString(PhysPt &env_scan,PhysPt env_fence) {
00277         /* scan until end of block or NUL */
00278         while (env_scan < env_fence && mem_readb(env_scan) != 0) env_scan++;
00279 
00280         /* if we hit the fence, that's something to warn about. */
00281         if (env_scan >= env_fence) {
00282                 LOG_MSG("Warning: environment string scan hit the end of the environment block without terminating NUL\n");
00283                 return false;
00284         }
00285 
00286         /* if we stopped at anything other than a NUL, that's something to warn about */
00287         if (mem_readb(env_scan) != 0) {
00288                 LOG_MSG("Warning: environment string scan scan stopped without hitting NUL\n");
00289                 return false;
