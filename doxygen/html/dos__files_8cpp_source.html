00618 bool DOS_OpenFileExtended(char const * name, Bit16u flags, Bit16u createAttr, Bit16u action, Bit16u *entry, Bit16u* status) {
00619 // FIXME: Not yet supported : Bit 13 of flags (int 0x24 on critical error)
00620         Bit16u result = 0;
00621         if (action==0) {
00622                 // always fail setting
00623                 DOS_SetError(DOSERR_FUNCTION_NUMBER_INVALID);
00624                 return false;
00625         } else {
00626                 if (((action & 0x0f)>2) || ((action & 0xf0)>0x10)) {
00627                         // invalid action parameter
00628                         DOS_SetError(DOSERR_FUNCTION_NUMBER_INVALID);
00629                         return false;
00630                 }
00631         }

00633                 // File already exists
00634                 switch (action & 0x0f) {
00635                         case 0x00:              // failed
00636                                 DOS_SetError(DOSERR_FILE_ALREADY_EXISTS);
00637                                 return false;
00638                         case 0x01:              // file open (already done)
00639                                 result = 1;
00640                                 break;
00641                         case 0x02:              // replace
00642                                 DOS_CloseFile(*entry);
00643                                 if (!DOS_CreateFile(name, createAttr, entry)) return false;
00644                                 result = 3;
00645                                 break;
00646                         default:
00647                                 DOS_SetError(DOSERR_FUNCTION_NUMBER_INVALID);
00648                                 E_Exit("DOS: OpenFileExtended: Unknown action.");
00649                                 break;
00650                 }
00651         } else {
00652                 // File doesn't exist
00653                 if ((action & 0xf0)==0) {
00654                         // uses error code from failed open
00655                         return false;
00656                 }
00657                 // Create File
00658                 if (!DOS_CreateFile(name, createAttr, entry)) {
00659                         // uses error code from failed create
00660                         return false;
00661                 }
00662                 result = 2;
00663         }
00664         // success
00665         *status = result;
00666         return true;

00828 Bit8u FCB_Parsename(Bit16u seg,Bit16u offset,Bit8u parser ,char *string, Bit8u *change) {
00829         char * string_begin=string;

00832                 // default drive forced, this intentionally invalidates an extended FCB
00833                 mem_writeb(PhysMake(seg,offset),0);
00834         }
00835         DOS_FCB fcb(seg,offset,false);  // always a non-extended FCB
00836         bool hasdrive,hasname,hasext,finished;
00837         hasdrive=hasname=hasext=finished=false;
00838         Bitu index=0;
00839         Bit8u fill=' ';
00840 /* First get the old data from the fcb */
00841 #ifdef _MSC_VER
00842 #pragma pack (1)
00843 #endif
00844         union {
00845                 struct {
00846                         char drive[2];
00847                         char name[9];
00848                         char ext[4];
00849                 } GCC_ATTRIBUTE (packed) part;
00850                 char full[DOS_FCBNAME];
00851         } fcb_name;
00852 #ifdef _MSC_VER
00853 #pragma pack()
00854 #endif
00855         /* Get the old information from the previous fcb */
00856         fcb.GetName(fcb_name.full);
00857         fcb_name.part.drive[0]-='A'-1;fcb_name.part.drive[1]=0;
00858         fcb_name.part.name[8]=0;fcb_name.part.ext[3]=0;
00859         /* Strip of the leading sepetaror */
00860         if((parser & PARSE_SEP_STOP) && *string)  {       //ignore leading seperator
00861                 char sep[] = FCB_SEP;char a[2];
00862                 a[0]= *string;a[1]='\0';
00863                 if (strcspn(a,sep)==0) string++;
00864         } 
00865         /* strip leading spaces */
00866         while((*string==' ')||(*string=='\t')) string++;
00867         /* Check for a drive */
00868         if (string[1]==':') {
00869                 fcb_name.part.drive[0]=0;
00870                 hasdrive=true;
00871                 if (isalpha(string[0]) && Drives[ascii_toupper(string[0])-'A']) {
00872                         fcb_name.part.drive[0]=(char)(ascii_toupper(string[0])-'A'+1);
00873                 } else ret=0xff;
00874                 string+=2;
00875         }
00876         /* Special checks for . and .. */
00877         if (string[0]=='.') {
00878                 string++;
00879                 if (!string[0]) {
00880                         hasname=true;
00881                         ret=PARSE_RET_NOWILD;
00882                         strcpy(fcb_name.part.name,".       ");
00883                         goto savefcb;
00884                 }
00885                 if (string[1]=='.' && !string[1])       {
00886                         string++;
00887                         hasname=true;
00888                         ret=PARSE_RET_NOWILD;
00889                         strcpy(fcb_name.part.name,"..      ");
00890                         goto savefcb;
00891                 }
00892                 goto checkext;
00893         }
00894         /* Copy the name */     
00895         hasname=true;finished=false;fill=' ';index=0;
00896         while (index<8) {
00897                 if (!finished) {
00898                         if (string[0]=='*') {fill='?';fcb_name.part.name[index]='?';if (!ret) ret=1;finished=true;}
00899                         else if (string[0]=='?') {fcb_name.part.name[index]='?';if (!ret) ret=1;}
00900             else if (IS_PC98_ARCH && shiftjis_lead_byte(string[0])) {
00901                 /* Shift-JIS is NOT ASCII and SHOULD NOT be converted to uppercase like ASCII */
00902                 fcb_name.part.name[index]=string[0];
00903                 string++;
00904                 index++;
00905                 if (index >= 8) break;
00906 
00907                 /* should be trailing byte of Shift-JIS */
00908                 if ((unsigned char)string[0] < 32 || (unsigned char)string[0] >= 127) continue;
00909 
00910                 fcb_name.part.name[index]=string[0];
00911             }
00912                         else if (isvalid(string[0])) {fcb_name.part.name[index]=(char)(ascii_toupper(string[0]));}
00913                         else { finished=true;continue; }
00914                         string++;
00915                 } else {
00916                         fcb_name.part.name[index]=fill;
00917                 }
00918                 index++;
00919         }
00920         if (!(string[0]=='.')) goto savefcb;
00921         string++;
00922 checkext:
00923         /* Copy the extension */
00924         hasext=true;finished=false;fill=' ';index=0;
00925         while (index<3) {
00926                 if (!finished) {
00927                         if (string[0]=='*') {fill='?';fcb_name.part.ext[index]='?';finished=true;}
00928                         else if (string[0]=='?') {fcb_name.part.ext[index]='?';if (!ret) ret=1;}

00930             else if (IS_PC98_ARCH && shiftjis_lead_byte(string[0])) {
00931                 /* Shift-JIS is NOT ASCII and SHOULD NOT be converted to uppercase like ASCII */
00932                 fcb_name.part.ext[index]=string[0];
00933                 string++;
00934                 index++;
00935                 if (index >= 3) break;
00936 
00937                 /* should be trailing byte of Shift-JIS */
00938                 if ((unsigned char)string[0] < 32 || (unsigned char)string[0] >= 127) continue;
00939 
00940                 fcb_name.part.ext[index]=string[0];
00941             }
00942                         else { finished=true;continue; }
00943                         string++;
00944                 } else {
00945                         fcb_name.part.ext[index]=fill;
00946                 }
00947                 index++;
00948         }
00949 savefcb:
00950         if (!hasdrive & !(parser & PARSE_DFLT_DRIVE)) fcb_name.part.drive[0] = 0;
00951         if (!hasname & !(parser & PARSE_BLNK_FNAME)) strcpy(fcb_name.part.name,"        ");
00952         if (!hasext & !(parser & PARSE_BLNK_FEXT)) strcpy(fcb_name.part.ext,"   ");
00953         fcb.SetName(fcb_name.part.drive[0],fcb_name.part.name,fcb_name.part.ext);
00954         *change=(Bit8u)(string-string_begin);

01062 Bit8u DOS_FCBRead(Bit16u seg,Bit16u offset,Bit16u recno) {
01063         DOS_FCB fcb(seg,offset);
01064         Bit8u fhandle,cur_rec;Bit16u cur_block,rec_size;
01065         fcb.GetSeqData(fhandle,rec_size);
01066         if (fhandle==0xff && rec_size!=0) {
01067                 if (!DOS_FCBOpen(seg,offset)) return FCB_READ_NODATA;
01068                 LOG(LOG_FCB,LOG_WARN)("Reopened closed FCB");
01069                 fcb.GetSeqData(fhandle,rec_size);
01070         }
01071         fcb.GetRecord(cur_block,cur_rec);
01072         Bit32u pos=((cur_block*128)+cur_rec)*rec_size;
01073         if (!DOS_SeekFile(fhandle,&pos,DOS_SEEK_SET)) return FCB_READ_NODATA; 
01074         Bit16u toread=rec_size;
01075         if (!DOS_ReadFile(fhandle,dos_copybuf,&toread)) return FCB_READ_NODATA;
01076         if (toread==0) return FCB_READ_NODATA;
01077         if (toread < rec_size) { //Zero pad copybuffer to rec_size
01078                 Bitu i = toread;
01079                 while(i < rec_size) dos_copybuf[i++] = 0;
01080         }
01081         MEM_BlockWrite(Real2Phys(dos.dta())+recno*rec_size,dos_copybuf,rec_size);
01082         if (++cur_rec>127) { cur_block++;cur_rec=0; }
01083         fcb.SetRecord(cur_block,cur_rec);
01084         if (toread==rec_size) return FCB_SUCCESS;
01085         if (toread==0) return FCB_READ_NODATA;
01086         return FCB_READ_PARTIAL;
01087 }

01152 Bit8u DOS_FCBRandomRead(Bit16u seg,Bit16u offset,Bit16u * numRec,bool restore) {
01153 /* if restore is true :random read else random blok read. 
01154  * random read updates old block and old record to reflect the random data
01155  * before the read!!!!!!!!! and the random data is not updated! (user must do this)
01156  * Random block read updates these fields to reflect the state after the read!
01157  */
01158         DOS_FCB fcb(seg,offset);

01160         Bit16u old_block=0;
01161         Bit8u old_rec=0;
01162         Bit8u error=0;

01164

01166         fcb.GetRandom(random);
01167         fcb.SetRecord((Bit16u)(random / 128),(Bit8u)(random & 127));
01168         if (restore) fcb.GetRecord(old_block,old_rec);//store this for after the read.
01169         // Read records
01170         for (count=0; count<*numRec; count++) {
01171                 error = DOS_FCBRead(seg,offset,count);
01172                 if (error!=FCB_SUCCESS) break;
01173         }
01174         if (error==FCB_READ_PARTIAL) count++;   //partial read counts
01175         *numRec=count;
01176         Bit16u new_block;Bit8u new_rec;
01177         fcb.GetRecord(new_block,new_rec);
01178         if (restore) fcb.SetRecord(old_block,old_rec);
01179         /* Update the random record pointer with new position only when restore is false*/
01180         if(!restore) fcb.SetRandom(new_block*128+new_rec); 
01181         return error;
