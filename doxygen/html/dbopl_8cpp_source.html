00383 Bits Operator::TemplateVolume(  ) {
00384         Bit32s vol = volume;
00385         Bit32s change;
00386         switch ( yes ) {
00387         case OFF:
00388                 return ENV_MAX;

00390                 change = RateForward( attackAdd );

00406                                 volume = ENV_MAX;
