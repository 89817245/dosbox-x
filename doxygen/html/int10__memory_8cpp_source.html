00056 void INT10_LoadFont(PhysPt font,bool reload,Bitu count,Bitu offset,Bitu map,Bitu height) {
00057         PhysPt ftwhere=PhysMake(0xa000,map_offset[map & 0x7]+(Bit16u)(offset*32));
00058         IO_Write(0x3c4,0x2);IO_Write(0x3c5,0x4);        //Enable plane 2
00059         IO_Write(0x3c4,0x4);IO_Write(0x3c5,0x6);        //disable odd/even memory write
00060         IO_Write(0x3ce,0x6);Bitu old_6=IO_Read(0x3cf);
00061         IO_Write(0x3cf,0x0);    //Disable odd/even and a0000 addressing
00062         for (Bitu i=0;i<count;i++) {
00063                 MEM_BlockCopy(ftwhere,font,height);
00064                 ftwhere+=32;
00065                 font+=height;
00066         }
00067         IO_Write(0x3c4,0x2);IO_Write(0x3c5,0x3);        //Enable textmode planes (0,1)
00068         IO_Write(0x3c4,0x4);IO_Write(0x3c5,0x2);        //reenable odd/even memory write
00069         IO_Write(0x3ce,0x6);
00070         if (IS_VGA_ARCH) IO_Write(0x3cf,(Bit8u)old_6);  //odd/even and b8000 addressing
00071         else IO_Write(0x3cf,0x0e);
00072         /* Reload tables and registers with new values based on this height */
00073         if (reload) {
00074                 //Max scanline 
00075                 Bit16u base=real_readw(BIOSMEM_SEG,BIOSMEM_CRTC_ADDRESS);
00076                 IO_Write(base,0x9);
00077                 IO_Write(base+1,(IO_Read(base+1) & 0xe0)|(height-1));
00078                 // Vertical display end bios says, but should stay the same?
00079                 // Not on EGA.
00080         Bitu rows = CurMode->sheight/height;
00081                 if (machine==MCH_EGA) {
00082                         Bitu displayend = rows*height - 1;
00083                         IO_Write(base,0x12);
00084                         IO_Write(base+1,(Bit8u)(displayend & 0xff));
00085                         IO_Write(base,0x7);
00086                         // Note: IBM EGA registers can't be read
00087                         Bitu v_overflow = IO_Read(base+1) & ~0x2;
00088                         if (displayend & 0x100) v_overflow |= 0x2;
00089                         IO_Write(base+1,(Bit8u)v_overflow);
00090                 }
00091                 //Rows setting in bios segment
00092                 real_writeb(BIOSMEM_SEG,BIOSMEM_NB_ROWS,rows-1);
00093                 real_writeb(BIOSMEM_SEG,BIOSMEM_CHAR_HEIGHT,(Bit8u)height);
00094                 //TODO Reprogram cursor size?
00095         }
