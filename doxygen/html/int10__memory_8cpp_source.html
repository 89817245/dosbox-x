00071         IO_Write(0x3ce,0x04);IO_Write(0x3cf,0x02); // select plane 2 for reading

00076         for (Bit16u i=0;i<count;i++) {

00082                 while (Bitu chr=(Bitu)mem_readb(font++)) {

00085                 }

00091         IO_Write(0x3ce,0x04);IO_Write(0x3cf,0x00); // select plane 0 for reading
